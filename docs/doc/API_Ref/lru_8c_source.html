91 daos_lru_cache_create(int bits, uint32_t feats,
92  struct daos_llink_ops *ops,
93  struct daos_lru_cache **lcache)
94 {
95  struct daos_lru_cache *lru_cache = NULL;
96  int rc = 0;
97 
98  D_DEBUG(DB_TRACE, "Creating a new LRU cache of size (2^%d)\n",
99  bits);
100 
101  if (ops == NULL ||
102  ops->lop_cmp_keys == NULL ||
103  ops->lop_alloc_ref == NULL ||
104  ops->lop_free_ref == NULL) {
105  D_ERROR("Error missing ops/mandatory-ops for LRU cache\n");
106  return -DER_INVAL;
107  }
108 
109  D_ALLOC_PTR(lru_cache);
110  if (lru_cache == NULL)
111  return -DER_NOMEM;
112 
113  rc = d_hash_table_create_inplace(feats, max(4, bits - 3),
114  NULL, &lru_ops,
115  &lru_cache->dlc_htable);
116  if (rc)
117  D_GOTO(exit, rc = -DER_NOMEM);
118 
119  if (bits >= 0)
120  lru_cache->dlc_csize = (1 << bits);
121  else /* disable LRU */
122  lru_cache->dlc_csize = 0;
123 
124  lru_cache->dlc_ops = ops;
125 
126  D_INIT_LIST_HEAD(&lru_cache->dlc_idle_list);
127  D_INIT_LIST_HEAD(&lru_cache->dlc_busy_list);
128 
129  *lcache = lru_cache;
130 exit:
131  if (rc != 0)
132  D_FREE_PTR(lru_cache);
133 
134  return rc;
135 }

155 daos_lru_cache_evict(struct daos_lru_cache *lcache,
156  daos_lru_cond_cb_t cond, void *args)
157 {
158  struct daos_llink *llink;
159  struct daos_llink *tmp;
160  unsigned int cntr;
161 
162  cntr = 0;
163  d_list_for_each_entry(llink, &lcache->dlc_busy_list, ll_qlink) {
164  if (cond == NULL || cond(llink, args))
165  /* will be evicted later in daos_lru_ref_release */
166  daos_lru_ref_evict(llink);
167  }
168  D_DEBUG(DB_TRACE, "Marked %d busy items as evicted\n", cntr);
169 
170  cntr = 0;
171  d_list_for_each_entry_safe(llink, tmp, &lcache->dlc_idle_list,
172  ll_qlink) {
173  if (cond == NULL || cond(llink, args)) {
174  d_list_del_init(&llink->ll_qlink);
175  d_hash_rec_delete_at(&lcache->dlc_htable,
176  &llink->ll_hlink);
177  lcache->dlc_idle_nr--;
178  cntr++;
179  }
180  }
181  D_DEBUG(DB_TRACE, "Evicted %d items from idle list\n", cntr);
182 }

185 lru_fast_search(struct daos_lru_cache *lcache, d_list_t *head,
186  void *key, unsigned int key_size)
187 {
188  struct daos_llink *llink;
189 
190  if (d_list_empty(head))
191  return NULL;
192 
193  llink = d_list_entry(head->next, struct daos_llink, ll_qlink);
194  if (llink->ll_evicted)
195  return NULL;
196 
197  if (llink->ll_ops->lop_cmp_keys(key, key_size, llink)) {
198  D_DEBUG(DB_TRACE, "Found item on the %s list.\n",
199  head == &lcache->dlc_busy_list ? "busy" : "idle");
200 
201  llink->ll_ref++; /* +1 for caller */
202  return llink;
203  }

241 daos_lru_ref_hold(struct daos_lru_cache *lcache, void *key,
242  unsigned int key_size, void *create_args,
243  struct daos_llink **rlink)
244 {
245  struct daos_llink *llink;
246  int rc;
247 
248  D_ASSERT(lcache != NULL && key != NULL && key_size > 0);
249  if (lcache->dlc_ops->lop_print_key)
250  lcache->dlc_ops->lop_print_key(key, key_size);
251 
252  llink = lru_fast_search(lcache, &lcache->dlc_busy_list, key, key_size);
253  if (llink)
254  D_GOTO(found, rc = 0);
255 
256  llink = lru_fast_search(lcache, &lcache->dlc_idle_list, key, key_size);
257  if (llink)
258  D_GOTO(found, rc = 0);
259 
260  llink = lru_hash_search(lcache, key, key_size);
261  if (llink)
262  D_GOTO(found, rc = 0);
263 
264  if (!create_args)
265  D_GOTO(out, rc = -DER_NONEXIST);
266 
267  D_DEBUG(DB_TRACE, "Entry not found adding it to LRU\n");
268  /* llink does not exist create one */
269  rc = lcache->dlc_ops->lop_alloc_ref(key, key_size, create_args, &llink);
270  if (rc)
271  D_GOTO(out, rc);
272 
273  D_DEBUG(DB_TRACE, "Inserting into LRU Hash table\n");
274  llink->ll_evicted = 0;
275  llink->ll_ref = 1; /* 1 for caller */
276  llink->ll_ops = lcache->dlc_ops;
277  D_INIT_LIST_HEAD(&llink->ll_qlink);
278 
279  rc = d_hash_rec_insert(&lcache->dlc_htable, key, key_size,
280  &llink->ll_hlink, true);
281  D_ASSERT(rc == 0);
282 found:
283  if (llink->ll_ref == 2) /* 1 for hash, 1 for the first holder */
284  lru_mark_busy(lcache, llink);
285 
286  *rlink = llink;
287 out:
288  return rc;
289 }

291 void
292 daos_lru_ref_release(struct daos_lru_cache *lcache, struct daos_llink *llink)
293 {
294  D_ASSERT(lcache != NULL && llink != NULL && llink->ll_ref > 1);
295  D_DEBUG(DB_TRACE, "Releasing item %p, ref=%d\n", llink, llink->ll_ref);
296 
297  llink->ll_ref--;
298  if (llink->ll_ref == 1) { /* the last refcount */
299  D_DEBUG(DB_TRACE, "busy: %u, idle: %u\n",
300  lcache->dlc_busy_nr, lcache->dlc_idle_nr);
301 
302  D_ASSERT(lcache->dlc_busy_nr > 0);
303  lcache->dlc_busy_nr--;
304 
305  if (llink->ll_evicted) {
306  D_DEBUG(DB_TRACE, "Evict %p from LRU cache\n", llink);
307  d_list_del_init(&llink->ll_qlink);
308  /* be freed within hash callback */
309  d_hash_rec_delete_at(&lcache->dlc_htable,
310  &llink->ll_hlink);
311  } else {
312  D_DEBUG(DB_TRACE,
313  "Moving %p to the idle list\n", llink);
314  lcache->dlc_idle_nr++;
315  d_list_move(&llink->ll_qlink, &lcache->dlc_idle_list);
316  }
317  }
318 
319  while (lcache->dlc_idle_nr != 0 &&
320  (lcache->dlc_busy_nr + lcache->dlc_idle_nr >=
321  lcache->dlc_csize)) {
322  D_DEBUG(DB_TRACE, "Evicting from object cache :%d, %d\n",
323  lcache->dlc_idle_nr, lcache->dlc_busy_nr);
324 
326  D_ASSERT(!d_list_empty(&lcache->dlc_idle_list));
327  llink = container_of(lcache->dlc_idle_list.prev,
328  struct daos_llink, ll_qlink);
329 
330  d_list_del_init(&llink->ll_qlink);
331  d_hash_rec_delete_at(&lcache->dlc_htable, &llink->ll_hlink);
332  lcache->dlc_idle_nr--;
333  }
334  D_DEBUG(DB_TRACE, "Done releasing reference\n");
335 }

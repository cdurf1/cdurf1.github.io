203 daos_oclass_attr_find(daos_obj_id_t oid)
204 {
205  struct daos_obj_class *oc;
206  daos_oclass_id_t ocid;
207 
208  /* see daos_objid_generate */
209  ocid = daos_obj_id2class(oid);
210  for (oc = &daos_obj_classes[0]; oc->oc_id != DAOS_OC_UNKNOWN; oc++) {
211  if (oc->oc_id == ocid)
212  break;
213  }
214 
215  if (ocid == DAOS_OC_UNKNOWN) {
216  D_DEBUG(DB_PL, "Unknown object class %d for "DF_OID"\n",
217  ocid, DP_OID(oid));
218  return NULL;
219  }
220 
221  D_DEBUG(DB_PL, "Find class %s for oid "DF_OID"\n",
222  oc->oc_name, DP_OID(oid));
223  return &oc->oc_attr;
224 }

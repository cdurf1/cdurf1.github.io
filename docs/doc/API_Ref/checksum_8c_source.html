168  uint64_t len)
169 {
170 #if defined(__x86_64__)

177  *cur_crc64 = crc64_ecma_refl(*cur_crc64,

181  }
182  case DAOS_CS_CRC32:
183  {
184  uint32_t *cur_crc32;
185 
186  cur_crc32 = (uint32_t *)csum->dc_buf;
187  *cur_crc32 = crc32_iscsi((unsigned char *)buf,
188  (int)len,
189  *cur_crc32);
190  break;
191  }
192  default:
193  D_ERROR("Unknown checksum type\n");
194  return -DER_NOSYS;
195  }
196 
197 #else
198  /* accumulates a partial checksum of the input data */
199  int rc;
200 
201  rc = mchecksum_update(csum->checksum, buf, len);
202  if (rc < 1)
203  return -DER_NOSYS;
204 #endif
205  return 0;

63 pl_map_create(struct pool_map *pool_map, struct pl_map_init_attr *mia,
64  struct pl_map **pl_mapp)
65 {
66  struct pl_map_dict *dict = pl_maps;
67  struct pl_map *map;
68  int rc;
69 
70  for (dict = &pl_maps[0]; dict->pd_type != PL_TYPE_UNKNOWN; dict++) {
71  if (dict->pd_type == mia->ia_type)
72  break;
73  }
74 
75  if (dict->pd_type == PL_TYPE_UNKNOWN) {
76  D_DEBUG(DB_PL,
77  "Unknown placement map type %d\n", dict->pd_type);
78  return -EINVAL;
79  }
80 
81  D_DEBUG(DB_PL, "Create a %s placement map\n", dict->pd_name);
82 
83  rc = dict->pd_ops->o_create(pool_map, mia, &map);
84  if (rc != 0)
85  return rc;
86 
87  rc = D_SPIN_INIT(&map->pl_lock, PTHREAD_PROCESS_PRIVATE);
88  if (rc != 0) {
89  dict->pd_ops->o_destroy(map);
90  return rc;
91  }
92 
93  map->pl_ref = 1; /* for the caller */
94  map->pl_connects = 0;
95  map->pl_type = mia->ia_type;
96  map->pl_ops = dict->pd_ops;
97 
98  *pl_mapp = map;
99  return 0;
100 }

374 int
375 pl_map_update(uuid_t uuid, struct pool_map *pool_map, bool connect)
376 {
377  d_list_t *link;
378  struct pl_map *map;
379  struct pl_map_init_attr mia;
380  int rc;
381 
382  D_RWLOCK_WRLOCK(&pl_rwlock);
383  if (!pl_htable.ht_ops) {
384  /* NB: this hash table is created on demand, it will never
385  * be destroyed.
386  */
387  rc = d_hash_table_create_inplace(D_HASH_FT_NOLOCK,
388  PL_HTABLE_BITS, NULL,
389  &pl_hash_ops, &pl_htable);
390  if (rc)
391  D_GOTO(out, rc);
392 
393  link = NULL;
394  } else {
395  /* already created */
396  link = d_hash_rec_find(&pl_htable, uuid, sizeof(uuid_t));
397  }
398 
399  if (!link) {
400  pl_map_attr_init(pool_map, PL_TYPE_RING, &mia);
401  rc = pl_map_create(pool_map, &mia, &map);
402  if (rc != 0)
403  D_GOTO(out, rc);
404  } else {
405  struct pl_map *tmp;
406 
407  tmp = container_of(link, struct pl_map, pl_link);
408  if (pl_map_version(tmp) >= pool_map_get_version(pool_map)) {
409  d_hash_rec_decref(&pl_htable, link);
410  if (connect)
411  tmp->pl_connects++;
412  D_GOTO(out, rc = 0);
413  }
414 
415  pl_map_attr_init(pool_map, PL_TYPE_RING, &mia);
416  rc = pl_map_create(pool_map, &mia, &map);
417  if (rc != 0) {
418  d_hash_rec_decref(&pl_htable, link);
419  D_GOTO(out, rc);
420  }
421 
422  /* transfer the pool connection count */
423  map->pl_connects = tmp->pl_connects;
424  /* evict the old placement map for this pool */
425  d_hash_rec_delete_at(&pl_htable, link);
426  d_hash_rec_decref(&pl_htable, link);
427  }
428 
429  if (connect)
430  map->pl_connects++;
431 
432  /* insert the new placement map into hash table */
433  uuid_copy(map->pl_uuid, uuid);
434  rc = d_hash_rec_insert(&pl_htable, uuid, sizeof(uuid_t),
435  &map->pl_link, true);
436  D_ASSERT(rc == 0);
437  pl_map_decref(map); /* hash table has held the refcount */
438  out:
439  D_RWLOCK_UNLOCK(&pl_rwlock);
440  return rc;
441 }

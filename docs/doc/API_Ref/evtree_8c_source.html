93

420 static int
421 evt_uncover_entries(struct evt_entry_list *ent_list, d_list_t *covered)
422 {
423  struct evt_rect *this_rect;
424  struct evt_rect *next_rect;
425  struct evt_entry *this_ent;
426  struct evt_entry *next_ent;
427  struct evt_entry *temp_ent;
428  d_list_t *current;
429  d_list_t *next;
430  d_list_t unused;
431  daos_size_t diff;
432  int flag_bit;
433  bool insert;
434 
435  D_INIT_LIST_HEAD(&unused); /* list of entries that can be reclaimed */
436 
437  /* reset the linked list. We'll reconstruct it */
438  D_INIT_LIST_HEAD(&ent_list->el_list);
439 
440  insert = true;
441  /* Now uncover entries */
442  current = covered->next;
443  /* Some compilers can't tell that this_ent will be initialized */
444  this_ent = d_list_entry(current, struct evt_entry,
445  en_link);
446  next = current->next;
447 
448  while (next != covered) {
449  if (insert) {
450  this_ent = d_list_entry(current, struct evt_entry,
451  en_link);
452  d_list_move_tail(current, &ent_list->el_list);
453  }
454 
455  insert = true;
456 
457  /* Find next uncovered rectangle */
458  next_ent = evt_find_next_uncovered(this_ent, covered, &next,
459  &unused, &flag_bit);
460  if (next_ent == NULL)
461  return 0;
462 
463  this_rect = &this_ent->en_sel_rect;
464  next_rect = &next_ent->en_sel_rect;
465  current = next;
466  next = current->next;
467  /* NB: Three possibilities
468  * 1. No intersection. Current entry is inserted in entirety
469  * 2. Partial intersection, next is earlier. Next is truncated
470  * 3. Partial intersection, next is later. Current is truncated
471  * 4. Current entry contains next_entry. Current is split
472  * in two and both are truncated.
473  */
474  if (next_rect->rc_off_lo >= this_rect->rc_off_hi + 1) {
475  /* Case #1, entry already inserted, nothing to do */
476  } else if (next_rect->rc_epc_lo < this_rect->rc_epc_lo) {
477  /* Case #2, next rect is partially under this rect,
478  * Truncate left end of next_rec, reinsert.
479  */
480  diff = this_rect->rc_off_hi + 1 - next_rect->rc_off_lo;
481  next_rect->rc_off_lo = this_rect->rc_off_hi + 1;
482  next_ent->en_offset += diff;
483  next_ent->en_addr = (char *)next_ent->en_addr +
484  diff * next_ent->en_inob;
485  /* current now points at next_ent. Remove it and
486  * reinsert it in the list in case truncation moved
487  * it to a new position
488  */
489  d_list_del(current);
490  next = evt_insert_sorted(next_ent, covered, next);
491  /* Reset the flag bit */
492  next_ent->en_inob |= flag_bit;
493 
494  /* Now we need to rerun this iteration without
495  * inserting this_ent again
496  */
497  insert = false;
498  } else if (next_rect->rc_off_hi >= this_rect->rc_off_hi) {
499  /* Case #3, truncate entry */
500  this_rect->rc_off_hi = next_rect->rc_off_lo - 1;
501  } else {
502  /* Case #4, split, insert tail into sorted list */
503  temp_ent = evt_get_unused_entry(ent_list, &unused);
504  if (temp_ent == NULL)
505  return -DER_NOMEM;
506  evt_split_entry(this_ent, next_ent, temp_ent);
507  /* Current points at next_ent */
508  next = evt_insert_sorted(temp_ent, covered, next);
509  }
510  }
511 
512  d_list_move_tail(current, &ent_list->el_list);
513 
514  return 0;
515 }

1399  */
1400  mbr_changed = evt_node_rect_update(tcx, nm_cur,
1401  trace->tr_at, mbr);
1402  if (!mbr_changed || level == 0)
1403  D_GOTO(out, 0);
1404 
1405  /* continue to merge MBR with upper level node */
1406  mbr = evt_node_mbr_get(tcx, nm_cur);
1407  level--;
1408  continue;
1409  }
1410 
1411  if (!evt_node_is_full(tcx, nm_cur)) {
1412  bool changed;
1413 
1414  rc = evt_node_insert(tcx, nm_cur, &entry, &changed);
1415  if (rc != 0)
1416  D_GOTO(failed, rc);
1417 
1418  /* NB: mbr_changed could have been set while splitting
1419  * the child node.
1420  */
1421  mbr_changed |= changed;
1422  if (!mbr_changed || level == 0)
1423  D_GOTO(out, 0);
1424 
1425  /* continue to merge MBR with upper level node */
1426  mbr = evt_node_mbr_get(tcx, nm_cur);
1427  level--;
1428  continue;
1429  }
1430  /* Try to split */
1431 
1432  D_DEBUG(DB_TRACE, "Split node at level %d\n", level);
1433 
1434  leaf = evt_node_is_leaf(tcx, nm_cur);
1435  rc = evt_node_alloc(tcx, leaf ? EVT_NODE_LEAF : 0, &nm_new);
1436  if (rc != 0)
1437  D_GOTO(failed, rc);
1438 
1439  rc = evt_node_split(tcx, leaf, nm_cur, nm_new);
1440  if (rc != 0) {
1441  D_DEBUG(DB_TRACE, "Failed to split node: %d\n", rc);
1442  D_GOTO(failed, rc);
1443  }
1444

1446  * new created node.
1447  */
1448  nm_ins = evt_select_node(tcx, &entry.en_rect, nm_cur, nm_new);
1449  rc = evt_node_insert(tcx, nm_ins, &entry, NULL);
1450  if (rc != 0)
1451  D_GOTO(failed, rc);
1452

1454  * - If the current node is not root, insert it to its parent
1455  * - If the current node is root, create a new root
1456  */
1457  entry.en_mmid = umem_id_t2u(nm_new);
1458  entry.en_rect = *evt_node_mbr_get(tcx, nm_new);
1459  if (level != 0) { /* not root */
1460  level--;
1461  /* After splitting, MBR of the current node has been
1462  * changed (half of its entries are moved out, and
1463  * probably added a new entry), so we need to update
1464  * its MBR stored in its parent.
1465  */
1466  trace = &tcx->tc_trace[level];
1467  mbr_changed = evt_node_rect_update(tcx,
1468  trace->tr_node, trace->tr_at,
1469  evt_node_mbr_get(tcx, nm_cur));
1470  /* continue to insert the new node to its parent */
1471  continue;
1472  }
1473

1475  tcx->tc_root->tr_depth + 1);
1476 
1477  D_ASSERT(evt_node_is_root(tcx, nm_cur));
1478  evt_node_unset(tcx, nm_cur, EVT_NODE_ROOT);
1479 
1480  rc = evt_node_alloc(tcx, EVT_NODE_ROOT, &nm_new);
1481  if (rc != 0)
1482  D_GOTO(failed, rc);
1483 
1484  rc = evt_node_insert(tcx, nm_new, &entry, NULL);
1485  if (rc != 0)
1486  D_GOTO(failed, rc);
1487 
1488  evt_tcx_set_dep(tcx, tcx->tc_depth + 1);
1489  tcx->tc_trace->tr_node = nm_new;
1490  tcx->tc_trace->tr_at = 0;
1491 
1492  tcx->tc_root->tr_node = nm_new;
1493  tcx->tc_root->tr_depth++;
1494 
1495  /* continue the loop and insert the original root node into
1496  * the new root node.
1497  */
1498  entry.en_rect = *evt_node_mbr_get(tcx, nm_cur);
1499  entry.en_mmid = umem_id_t2u(nm_cur);
1500  }
1501  out:
1502  return 0;
1503  failed:
1504  D_ERROR("Failed to insert entry to level %d: %d\n", level, rc);
1505  return rc;

1794 evt_find_ent_list(struct evt_context *tcx, enum evt_find_opc find_opc,
1795  struct evt_rect *rect, struct evt_entry_list *ent_list)
1796 {

1798  int level;
1799  int at;
1800  int i;
1801  int rc = 0;
1802 
1803  D_DEBUG(DB_TRACE, "Searching rectangle "DF_RECT" opc=%d\n",
1804  DP_RECT(rect), find_opc);
1805  if (tcx->tc_root->tr_depth == 0)
1806  return 0; /* empty tree */
1807 
1808  evt_tcx_reset_trace(tcx);
1809 
1810  level = at = 0;
1811  nd_mmid = tcx->tc_root->tr_node;
1812  while (1) {
1813  struct evt_rect *mbr;
1814  struct evt_node *node;
1815  bool leaf;
1816 
1817  node = evt_tmmid2ptr(tcx, nd_mmid);
1818  leaf = evt_node_is_leaf(tcx, nd_mmid);
1819  mbr = evt_node_mbr_get(tcx, nd_mmid);

1822  D_DEBUG(DB_TRACE,
1823  "Checking "DF_RECT"("TMMID_PF"), l=%d, a=%d, f=%d\n",
1824  DP_RECT(mbr), TMMID_P(nd_mmid), level, at, leaf);
1825

1827  struct evt_entry *ent;
1828  struct evt_rect *rtmp;
1829  int time_overlap;
1830  int range_overlap;
1831 
1832  rtmp = evt_node_rect_at(tcx, nd_mmid, i);
1833  D_DEBUG(DB_TRACE, " rect[%d]="DF_RECT"\n",
1834  i, DP_RECT(rtmp));
1835

1837  &time_overlap);
1838  switch (range_overlap) {
1839  default:
1840  D_ASSERT(0);
1841  case RT_OVERLAP_NO:
1842  continue; /* skip, no overlap */
1843 
1844  case RT_OVERLAP_SAME:
1845  case RT_OVERLAP_INCLUDED:
1846  case RT_OVERLAP_INCLUDES:
1847  case RT_OVERLAP_PARTIAL:
1848  break; /* overlapped */
1849  }
1850 
1851  switch (time_overlap) {
1852  default:
1853  D_ASSERT(0);
1854  case RT_OVERLAP_NO:
1855  case RT_OVERLAP_UNDER:
1856  continue; /* skip, no overlap */
1857  case RT_OVERLAP_OVER:
1858  case RT_OVERLAP_SAME:
1859  break; /* overlapped */
1860  }
1861 
1862  if (!leaf) {
1863  /* break the internal loop and enter the
1864  * child node.
1865  */
1866  D_DEBUG(DB_TRACE, "Enter the next level\n");
1867  break;
1868  }
1869  D_DEBUG(DB_TRACE, "Found overlapped leaf rect\n");
1870 
1871  /* early check */
1872  switch (find_opc) {
1873  default:
1874  D_ASSERTF(0, "%d\n", find_opc);
1875  case EVT_FIND_OVERWRITE:
1876  if (time_overlap != RT_OVERLAP_SAME)
1877  continue; /* not same epoch, skip */

1879  * in same epoch to avoid breaking rebuild.
1880  * Without some sequence number and client

1882  * There can be a race between rebuild and

1886  * change while we decide how to handle this
1887  * properly.
1888  */
1889  if (range_overlap != RT_OVERLAP_SAME) {
1890  D_DEBUG(DB_IO, "Same epoch partial "
1891  "overwrite not supported:"
1892  DF_RECT" overlaps with "DF_RECT
1893  "\n", DP_RECT(rect),
1894  DP_RECT(rtmp));
1895  rc = -DER_NO_PERM;
1896  goto out;
1897  }
1898  break; /* we can update the record in place */
1899  case EVT_FIND_SAME:
1900  if (range_overlap != RT_OVERLAP_SAME)
1901  continue;
1902  if (time_overlap != RT_OVERLAP_SAME)
1903  continue;
1904  break;
1905  case EVT_FIND_FIRST:
1906  case EVT_FIND_ALL:
1907  break;
1908  }
1909 
1910  ent = evt_ent_list_alloc(ent_list, true);
1911  if (ent == NULL)
1912  D_GOTO(out, rc = -DER_NOMEM);
1913 
1914  evt_fill_entry(tcx, nd_mmid, i, rect, ent);
1915  switch (find_opc) {
1916  default:
1917  D_ASSERTF(0, "%d\n", find_opc);
1918  case EVT_FIND_OVERWRITE:
1919  case EVT_FIND_FIRST:
1920  case EVT_FIND_SAME:
1921  /* store the trace and return for clip or
1922  * iteration.
1923  * NB: clip is not implemented yet.
1924  */
1925  evt_tcx_set_trace(tcx, level, nd_mmid, i);
1926  D_GOTO(out, rc = 0);
1927 
1928  case EVT_FIND_ALL:
1929  break;
1930  }
1931  }
1932 
1933  if (i < node->tn_nr) {
1934  /* overlapped with a non-leaf node, dive into it. */
1935  evt_tcx_set_trace(tcx, level, nd_mmid, i);
1936  nd_mmid = *evt_node_child_at(tcx, nd_mmid, i);
1937  at = 0;
1938  level++;
1939 
1940  } else {
1941  struct evt_trace *trace;
1942 
1943  if (level == 0) { /* done with the root */
1944  D_DEBUG(DB_TRACE, "Found total %d rects\n",
1945  ent_list ? ent_list->el_ent_nr : 0);
1946  return 0; /* succeed and return */
1947  }
1948 
1949  level--;
1950  trace = evt_tcx_trace(tcx, level);
1951  nd_mmid = trace->tr_node;
1952  at = trace->tr_at + 1;
1953  D_ASSERT(at <= tcx->tc_order);
1954  }
1955  }
1956 out:
1957  if (rc != 0)
1958  evt_ent_list_fini(ent_list);
1959  return rc;
1960 }

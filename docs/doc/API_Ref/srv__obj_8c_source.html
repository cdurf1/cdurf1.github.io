771 int
772 fill_key(vos_iter_entry_t *key_ent, struct obj_key_enum_in *oei,
773  struct obj_key_enum_out *oeo, unsigned int *kds_idx,
774  unsigned int *iovs_idx)
775 {
776  daos_iov_t *iovs = oeo->oeo_sgl.sg_iovs;
777  daos_key_desc_t *kds = oeo->oeo_kds.ca_arrays;
778  unsigned int iovs_nr = oeo->oeo_sgl.sg_nr;
779  unsigned int kds_nr = oei->oei_nr;
780 
781  while (*iovs_idx < iovs_nr) {
782  if (iovs[*iovs_idx].iov_len + key_ent->ie_key.iov_len >=
783  iovs[*iovs_idx].iov_buf_len) {
784  (*iovs_idx)++;
785  continue;
786  }
787 
788  D_ASSERT(*kds_idx < kds_nr);
789  kds[*kds_idx].kd_key_len = key_ent->ie_key.iov_len;
790  kds[*kds_idx].kd_csum_len = 0;
791  oeo->oeo_kds.ca_count++;
792  (*kds_idx)++;
793 
794  memcpy(iovs[*iovs_idx].iov_buf + iovs[*iovs_idx].iov_len,
795  key_ent->ie_key.iov_buf, key_ent->ie_key.iov_len);
796  iovs[*iovs_idx].iov_len += key_ent->ie_key.iov_len;
797 
798  if (oeo->oeo_sgl.sg_nr_out < *iovs_idx + 1)
799  oeo->oeo_sgl.sg_nr_out = *iovs_idx + 1;
800 
801  break;
802  }
803 
804  /* if it reaches the end of iovs and kds, then return 1 to
805  * break the loop
806  */
807  if (*iovs_idx >= iovs_nr || *kds_idx >= kds_nr)
808  return 1;
809 
810  return 0;
811 }

861 static int
862 ds_iter_single_vos(void *data)
863 {

872  daos_handle_t ih;
873  daos_hash_out_t *probe_hash;
874  int type;
875  int rc;
876 
877  rc = ds_check_container(oei->oei_co_hdl, oei->oei_co_uuid,
878  &cont_hdl, &cont);
879  if (rc)
880  D_GOTO(out, rc);
881 
882  D_ASSERT(cont_hdl->sch_pool != NULL);
883  if (iter_arg->map_version == 0)
884  iter_arg->map_version = cont_hdl->sch_pool->spc_map_version;
885 
886  if (oei->oei_map_ver < iter_arg->map_version) {
887  D_DEBUG(DB_IO, "stale version req %d map_version %d\n",
888  oei->oei_map_ver, iter_arg->map_version);
889  }
890 
891  if (arg->opc == DAOS_OBJ_AKEY_RPC_ENUMERATE) {
892  type = VOS_ITER_AKEY;
893  } else if (arg->opc == DAOS_OBJ_DKEY_RPC_ENUMERATE) {
894  type = VOS_ITER_DKEY;
895  } else {
896  if (oei->oei_rec_type == DAOS_IOD_ARRAY)
897  type = VOS_ITER_RECX;
898  else
899  type = VOS_ITER_SINGLE;
900  }
901 
902  /* prepare iterate parameters */
903  memset(&param, 0, sizeof(param));
904  param.ip_hdl = cont->sc_hdl;
905  param.ip_oid = oei->oei_oid;
906 
907  if (type == VOS_ITER_RECX || type == VOS_ITER_SINGLE) {
908  if (oei->oei_dkey.iov_len == 0 ||
909  oei->oei_akey.iov_len == 0)
910  D_GOTO(out_cont_hdl, rc = -DER_PROTO);
911  param.ip_dkey = oei->oei_dkey;
912  param.ip_akey = oei->oei_akey;
913 
914  param.ip_epr.epr_lo = 0;
915  param.ip_epr.epr_hi = oei->oei_epoch;
916  param.ip_epc_expr = VOS_IT_EPC_RE;
917  } else {
918  param.ip_epr.epr_lo = param.ip_epr.epr_hi = oei->oei_epoch;
919  if (type == VOS_ITER_AKEY) {
920  if (oei->oei_dkey.iov_len == 0)
921  D_GOTO(out_cont_hdl, rc = -DER_PROTO);
922  param.ip_dkey = oei->oei_dkey;
923  } else {
924  if (oei->oei_akey.iov_len > 0)
925  param.ip_akey = oei->oei_akey;
926  }
927  }
928 
929  D_DEBUG(DB_TRACE, ""DF_UOID" iterate type %d tag %d\n",
930  DP_UOID(oei->oei_oid), type, dss_get_module_info()->dmi_tid);
931 
932  rc = vos_iter_prepare(type, &param, &ih);
933  if (rc != 0) {
934  if (rc == -DER_NONEXIST) {
935  daos_hash_set_eof(&oeo->oeo_anchor);
936  rc = 0;
937  } else {
938  D_ERROR("Failed to prepare d-key iterator: %d\n", rc);
939  }
940  D_GOTO(out_cont_hdl, rc);
941  }
942 
943  if (daos_hash_is_zero(&oei->oei_anchor))
944  probe_hash = NULL;
945  else
946  probe_hash = &oei->oei_anchor;
947 
948  rc = vos_iter_probe(ih, probe_hash);
949  if (rc != 0) {
950  if (rc == -DER_NONEXIST || rc == -DER_AGAIN) {
951  daos_hash_set_eof(&oeo->oeo_anchor);
952  rc = 0;
953  }
954  D_GOTO(out_iter_fini, rc);
955  }
956 
957  while (iter_arg->key_nr < oei->oei_nr) {
958  rc = vos_iter_fetch(ih, &key_ent, &oeo->oeo_anchor);
959  if (rc != 0)
960  break;
961 
962  /* fill the key to iov if there are enough space */
963  if (type == VOS_ITER_AKEY || type == VOS_ITER_DKEY)
964  rc = fill_key(&key_ent, oei, oeo, &iter_arg->key_nr,
965  &iter_arg->iovs_idx);
966  else
967  rc = fill_rec(&key_ent, oei, oeo, &iter_arg->key_nr);
968 
969  if (rc != 0) {
970  if (rc == 1)
971  rc = vos_iter_next(ih);
972  break;
973  }
974  vos_iter_next(ih);
975  }
976 
977  if (rc == 0) /* anchor for the next call */
978  rc = vos_iter_fetch(ih, &key_ent, &oeo->oeo_anchor);
979 
980  if (rc == -DER_NONEXIST) {
981  daos_hash_set_eof(&oeo->oeo_anchor);
982  rc = 0;
983  }
984 out_iter_fini:
985  vos_iter_finish(ih);
986 out_cont_hdl:
987  if (!cont_hdl->sch_cont)
988  ds_cont_put(cont); /* -1 for rebuild container */
989  ds_cont_hdl_put(cont_hdl);
990 out:
991  return rc;
992 }

1016  if (oeo->oeo_eprs.ca_arrays == NULL)

1018 
1019  oeo->oeo_recxs.ca_count = 0;
1020  D_ALLOC(oeo->oeo_recxs.ca_arrays,
1021  oei->oei_nr * sizeof(daos_recx_t));

1023  D_GOTO(out, rc = -DER_NOMEM);
1024

1026  D_ALLOC(oeo->oeo_cookies.ca_arrays,
1027  oei->oei_nr * sizeof(uuid_t));
1028  if (oeo->oeo_cookies.ca_arrays == NULL)
1029  D_GOTO(out, rc = -DER_NOMEM);
1030

1032  D_ALLOC(oeo->oeo_vers.ca_arrays,
1033  oei->oei_nr * sizeof(uint32_t));
1034  if (oeo->oeo_vers.ca_arrays == NULL)
1035  D_GOTO(out, rc = -DER_NOMEM);
1036  } else {
1037  /* prepare buffer for enumerate */
1038  rc = ds_sgls_prep(&oeo->oeo_sgl, &oei->oei_sgl, 1);
1039  if (rc != 0)
1040  D_GOTO(out, rc);
1041 
1042  /* Prepare key desciptor buffer */
1043  oeo->oeo_kds.ca_count = 0;
1044  D_ALLOC(oeo->oeo_kds.ca_arrays,
1045  oei->oei_nr * sizeof(daos_key_desc_t));
1046  if (oeo->oeo_kds.ca_arrays == NULL)
1047  D_GOTO(out, rc = -DER_NOMEM);
1048  }
1049 
1050  task_arg.opc = opc_get(rpc->cr_opc);
1051  task_arg.u.iter_arg.oei = crt_req_get(rpc);
1052  task_arg.u.iter_arg.oeo = crt_reply_get(rpc);
1053  task_arg.u.iter_arg.iovs_idx = 0;
1054  task_arg.u.iter_arg.key_nr = 0;
1055  task_arg.u.iter_arg.status = 0;
1056  task_arg.u.iter_arg.map_version = 0;

1059  * reaching the end of the stream
1060  */

1062 
1063  while (task_arg.u.iter_arg.key_nr < oei->oei_nr &&

1065  if (tag == dss_get_module_info()->dmi_tid ||
1066  opc_get(rpc->cr_opc) != DAOS_OBJ_DKEY_RPC_ENUMERATE)
1067  rc = ds_iter_single_vos(&task_arg);
1068  else
1069  rc = dss_ult_create_execute(ds_iter_single_vos,
1070  &task_arg,
1071  NULL /* user callback */,
1072  NULL /* user cb args */,
1073  tag/* async */);
1074 
1075  if (rc != 0)
1076  D_GOTO(out, rc);
1077

1079  * go multiple tags.
1080  */

1082  if (opc_get(rpc->cr_opc) ==
1083  DAOS_OBJ_DKEY_RPC_ENUMERATE &&
1084  tag < dss_get_threads_number() - 1) {
1085  tag++;
1086  enum_anchor_reset_hkey(&oeo->oeo_anchor);
1087  enum_anchor_reset_hkey(&oei->oei_anchor);
1088  } else {
1089  break;
1090  }
1091  }
1092 
1093  /* Check if the buffer is full */
1094  if (opc_get(rpc->cr_opc) == DAOS_OBJ_DKEY_RPC_ENUMERATE ||
1095  opc_get(rpc->cr_opc) == DAOS_OBJ_AKEY_RPC_ENUMERATE) {
1096  if (task_arg.u.iter_arg.iovs_idx >=
1097  oeo->oeo_sgl.sg_nr)
1098  break;
1099  }
1100  }
1101 
1102  enum_anchor_set_tag(&oeo->oeo_anchor, tag);
1103  if (oei->oei_bulk != NULL) {
1104  daos_sg_list_t *sgl = &oeo->oeo_sgl;
1105 
1106  rc = ds_bulk_transfer(rpc, CRT_BULK_PUT, &oei->oei_bulk,
1107  DAOS_HDL_INVAL, &sgl, 1);
1108 
1109  /* If the keys will be replied by bulk, then let's empty
1110  * the sgl in the reply to avoid confusing
1111  */
1112  daos_sgl_fini(sgl, true);
1113  oeo->oeo_sgl.sg_iovs = NULL;
1114  oeo->oeo_sgl.sg_nr = 0;
1115  oeo->oeo_sgl.sg_nr_out = 0;
1116  }
1117 
1118 out:
1119  if (rc == 0)
1120  rc = task_arg.u.iter_arg.status;
1121 
1122  ds_eu_complete(rpc, rc, task_arg.u.iter_arg.map_version);
1123 }

185 /* If buf is NULL, then just calculate and return the length required. */
186 static size_t
187 rdb_tx_op_encode(struct rdb_tx_op *op, void *buf)
188 {
189  void *p = buf;
190 
191  /* opc */
192  if (buf != NULL)
193  *(uint8_t *)p = op->dto_opc;
194  p += sizeof(uint8_t);
195  /* kvs */
196  p += rdb_encode_iov(&op->dto_kvs, buf == NULL ? NULL : p);
197  /* key */
198  p += rdb_encode_iov(&op->dto_key, buf == NULL ? NULL : p);
199  if (op->dto_opc == RDB_TX_UPDATE) {
200  /* value */
201  p += rdb_encode_iov(&op->dto_value, buf == NULL ? NULL : p);
202  } else if (op->dto_opc == RDB_TX_CREATE_ROOT ||
203  op->dto_opc == RDB_TX_CREATE) {
204  /* attr */
205  if (buf != NULL)
206  *(struct rdb_kvs_attr *)p = *(op->dto_attr);
207  p += sizeof(struct rdb_kvs_attr);
208  } else {
209  D_ASSERT(op->dto_value.iov_buf == NULL);
210  D_ASSERT(op->dto_value.iov_buf_len == 0);
211  D_ASSERT(op->dto_value.iov_len == 0);
212  D_ASSERT(op->dto_attr == NULL);
213  }
214  return p - buf;
215 }

217 /* Returns the number of bytes processed or -DER_IO if the content is bad. */
218 static ssize_t
219 rdb_tx_op_decode(const void *buf, size_t len, struct rdb_tx_op *op)
220 {
221  struct rdb_tx_op o = {};
222  const void *p = buf;
223  ssize_t n;
224 
225  /* opc */
226  if (p + sizeof(uint8_t) > buf + len) {
227  D_ERROR("truncated opc: %zu < %zu\n", len, sizeof(uint8_t));
228  return -DER_IO;
229  }
230  o.dto_opc = *(const uint8_t *)p;
231  p += sizeof(uint8_t);
232  /* kvs */
233  n = rdb_decode_iov(p, buf + len - p, &o.dto_kvs);
234  if (n < 0) {
235  D_ERROR("failed to decode kvs\n");
236  return n;
237  }
238  p += n;
239  /* key */
240  n = rdb_decode_iov(p, buf + len - p, &o.dto_key);
241  if (n < 0) {
242  D_ERROR("failed to decode key\n");
243  return n;
244  }
245  p += n;
246  if (o.dto_opc == RDB_TX_UPDATE) {
247  /* value */
248  n = rdb_decode_iov(p, buf + len - p, &o.dto_value);
249  if (n < 0) {
250  D_ERROR("failed to decode value\n");
251  return n;
252  }
253  p += n;
254  } else if (o.dto_opc == RDB_TX_CREATE_ROOT ||
255  o.dto_opc == RDB_TX_CREATE) {
256  /* attr */
257  if (p + sizeof(struct rdb_kvs_attr) > buf + len) {
258  D_ERROR("truncated attr: %zu < %zu\n", buf + len - p,
259  sizeof(struct rdb_kvs_attr));
260  return -DER_IO;
261  }
262  o.dto_attr = (struct rdb_kvs_attr *)p;
263  p += sizeof(struct rdb_kvs_attr);
264  }
265  *op = o;
266  return p - buf;
267 }

269 /* Append an update operation to tx->dt_entry. */
270 static int
271 rdb_tx_append(struct rdb_tx *tx, struct rdb_tx_op *op)
272 {
273  size_t len;
274  int rc;
275 
276  D_ASSERTF((tx->dt_entry == NULL && tx->dt_entry_cap == 0 &&
277  tx->dt_entry_len == 0) ||
278  (tx->dt_entry != NULL && tx->dt_entry_cap > 0 &&
279  tx->dt_entry_len <= tx->dt_entry_cap),
280  "entry=%p cap=%zu len=%zu\n", tx->dt_entry, tx->dt_entry_cap,
281  tx->dt_entry_len);
282  if (op->dto_opc == RDB_TX_CREATE || op->dto_opc == RDB_TX_DESTROY ||
283  op->dto_opc == RDB_TX_UPDATE || op->dto_opc == RDB_TX_DELETE) {
284  if (op->dto_key.iov_len == 0)
285  return -DER_INVAL;
286  }
287 
288  rc = rdb_tx_leader_check(tx);
289  if (rc != 0)
290  return rc;
291 
292  /* Calculate and check the additional bytes required. */
293  len = rdb_tx_op_encode(op, NULL);
294  if (len > tx->dt_entry_cap - tx->dt_entry_len) {
295  size_t new_size = tx->dt_entry_cap;
296  void *new_buf;
297 
298  /* Not enough room; reallocate a larger buffer. */
299  do {
300  if (new_size == 0)
301  new_size = 4096;
302  else
303  new_size *= 2;
304  } while (len > new_size - tx->dt_entry_len);
305  D_ALLOC(new_buf, new_size);
306  if (new_buf == NULL)
307  return -DER_NOMEM;
308  if (tx->dt_entry_len > 0)
309  memcpy(new_buf, tx->dt_entry, tx->dt_entry_len);
310  if (tx->dt_entry != NULL)
311  D_FREE(tx->dt_entry);
312  tx->dt_entry = new_buf;
313  tx->dt_entry_cap = new_size;
314  }
315 
316  /* Now do the actual encoding. */
317  rdb_tx_op_encode(op, tx->dt_entry + tx->dt_entry_len);
318  tx->dt_entry_len += len;
319  return 0;
320 }

463 static int
464 rdb_tx_apply_op(struct rdb *db, struct rdb_tx_op *op, d_list_t *destroyed)
465 {
466  struct rdb_kvs *kvs = NULL;
467  rdb_path_t victim_path;
468  volatile int rc;
469 
470  D_DEBUG(DB_ANY, DF_DB": "DF_TX_OP"\n", DP_DB(db), DP_TX_OP(op));
471 
472  if (op->dto_opc != RDB_TX_CREATE_ROOT &&
473  op->dto_opc != RDB_TX_DESTROY_ROOT) {
474  /* Look up the KVS. */
475  rc = rdb_kvs_lookup(db, &op->dto_kvs, &kvs);
476  if (rc != 0)
477  return rc;
478  }
479 
480  /* If destroying a KVS, prepare a path to it. */
481  if (op->dto_opc == RDB_TX_DESTROY_ROOT) {
482  rc = rdb_path_init(&victim_path);
483  if (rc != 0)
484  return rc;
485  rc = rdb_path_push(&victim_path, &rdb_path_root_key);
486  if (rc != 0) {
487  rdb_path_fini(&victim_path);
488  return rc;
489  }
490  } else if (op->dto_opc == RDB_TX_DESTROY) {
491  rc = rdb_path_clone(&op->dto_kvs, &victim_path);
492  if (rc != 0) {
493  rdb_kvs_put(db, kvs);
494  return rc;
495  }
496  rc = rdb_path_push(&victim_path, &op->dto_key);
497  if (rc != 0) {
498  rdb_path_fini(&victim_path);
499  rdb_kvs_put(db, kvs);
500  return rc;
501  }
502  }
503 
504  TX_BEGIN(db->d_pmem) {
505  switch (op->dto_opc) {
506  case RDB_TX_CREATE_ROOT:
507  rc = rdb_create_tree(db->d_attr, &rdb_attr_root,
508  op->dto_attr->dsa_class,
509  0 /* feats */,
510  op->dto_attr->dsa_order,
511  NULL /* child */);
512  break;
513  case RDB_TX_DESTROY_ROOT:
514  rc = rdb_destroy_tree(db->d_attr, &rdb_attr_root);
515  break;
516  case RDB_TX_CREATE:
517  rc = rdb_create_tree(kvs->de_hdl, &op->dto_key,
518  op->dto_attr->dsa_class,
519  0 /* feats */,
520  op->dto_attr->dsa_order,
521  NULL /* child */);
522  break;
523  case RDB_TX_DESTROY:
524  rc = rdb_destroy_tree(kvs->de_hdl, &op->dto_key);
525  break;
526  case RDB_TX_UPDATE:
527  rc = dbtree_update(kvs->de_hdl, &op->dto_key,
528  &op->dto_value);
529  break;
530  case RDB_TX_DELETE:
531  rc = dbtree_delete(kvs->de_hdl, &op->dto_key, NULL);
532  break;
533  default:
534  D_ERROR(DF_DB": unknown update operation %u\n",
535  DP_DB(db), op->dto_opc);
536  rc = -DER_IO;
537  }
538  if (rc != 0)
539  pmemobj_tx_abort(rc);
540  } TX_ONCOMMIT {
541  if (op->dto_opc == RDB_TX_DESTROY_ROOT ||
542  op->dto_opc == RDB_TX_DESTROY) {
543  struct rdb_kvs *victim;
544  int rc_tmp;
545 
546  /*
547  * Look up and save victim in destroyed, so that
548  * we can evict it only if the upper-level PMDK TX
549  * commits successfully.
550  */
551  rc_tmp = rdb_kvs_lookup(db, &victim_path, &victim);
552  if (rc_tmp == 0) {
553  D_DEBUG(DB_ANY, DF_DB": add to destroyed %p\n",
554  DP_DB(db), victim);
555  d_list_add_tail(&victim->de_list, destroyed);
556  }
557  }
558  } TX_ONABORT {
559  rc = umem_tx_errno(rc);
560  } TX_FINALLY {
561  if (op->dto_opc == RDB_TX_DESTROY_ROOT ||
562  op->dto_opc == RDB_TX_DESTROY)
563  rdb_path_fini(&victim_path);
564  if (kvs != NULL)
565  rdb_kvs_put(db, kvs);
566  } TX_END
567 
568  return rc;
569 }

580  * Apply an entry and return the error only if a non-deterministic error
581  * happens. Ask callers to provide memory for destroyed to avoid fiddling with
582  * volatiles.
583  */
584 int
585 rdb_tx_apply(struct rdb *db, uint64_t index, const void *buf, size_t len,
586  void *result, d_list_t *destroyed)
587 {
588  daos_iov_t value;
589  volatile int rc;
590 
591  D_DEBUG(DB_ANY, DF_DB": applying entry "DF_U64": buf=%p len="DF_U64
592  "\n", DP_DB(db), index, buf, len);
593 
594  daos_iov_set(&value, &index, sizeof(index));
595 
596  TX_BEGIN(db->d_pmem) {
597  const void *p = buf;
598 
599  while (p < buf + len) {
600  struct rdb_tx_op op;
601  ssize_t n;
602 
603  n = rdb_tx_op_decode(p, buf + len - p, &op);
604  if (n < 0) {
605  /* Perhaps due to storage corruptions. */
606  D_ERROR(DF_DB": invalid entry format: buf=%p "
607  "len="DF_U64" p=%p\n", DP_DB(db), buf,
608  len, p);
609  pmemobj_tx_abort(n);
610  }
611  rc = rdb_tx_apply_op(db, &op, destroyed);
612  if (rc != 0) {
613  if (!rdb_tx_deterministic_error(rc))
614  D_ERROR(DF_DB": failed to apply entry "
615  DF_U64" op %u <%td, %zd>: %d\n",
616  DP_DB(db), index, op.dto_opc,
617  p - buf, n, rc);
618  pmemobj_tx_abort(rc);
619  }
620  p += n;
621  }
622  rc = dbtree_update(db->d_attr, &rdb_attr_applied, &value);
623  if (rc != 0)
624  pmemobj_tx_abort(rc);
625  } TX_ONABORT {
626  rc = umem_tx_errno(rc);
627  } TX_FINALLY {
628  struct rdb_kvs *kvs;
629  struct rdb_kvs *tmp;
630 
631  /*
632  * If rc == 0, then evict the KVS objects that have been
633  * destroyed. Otherwise, just release them.
634  */
635  d_list_for_each_entry_safe(kvs, tmp, destroyed, de_list) {
636  d_list_del_init(&kvs->de_list);
637  if (rc == 0) {
638  D_DEBUG(DB_ANY, DF_DB": evicting %p\n",
639  DP_DB(db), kvs);
640  rdb_kvs_evict(db, kvs);
641  }
642  rdb_kvs_put(db, kvs);
643  }
644  } TX_END
645 
646  if (rc != 0 && !rdb_tx_deterministic_error(rc))
647  return rc;
648 
649  if (rc != 0) {
650  volatile int rc_tmp;
651 
652  /* For deterministic errors, update rdb_attr_applied. */
653  TX_BEGIN(db->d_pmem) {
654  rc_tmp = dbtree_update(db->d_attr, &rdb_attr_applied,
655  &value);
656  if (rc_tmp != 0)
657  pmemobj_tx_abort(rc_tmp);
658  } TX_ONABORT {
659  rc_tmp = umem_tx_errno(rc_tmp);
660  } TX_END
661  if (rc_tmp != 0)
662  return rc_tmp;
663  }
664  /*
665  * Report the deterministic error to the result buffer, if there is
666  * one, and consider this entry applied.
667  */
668  if (result != NULL)
669  *(int *)result = rc;
670  return 0;
671 }

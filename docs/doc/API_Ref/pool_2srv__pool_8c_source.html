319  int ndomains, const int *domains, d_rank_list_t **ranksp)
320 {

323  d_rank_list_t *ranks;

326 
327  if (nreplicas <= 0)
328  return -DER_INVAL;
329 
330  /* Determine the number of selectable targets. */
331  selectable = target_addrs->rl_nr;
332  if (daos_rank_list_find((d_rank_list_t *)target_addrs, 0 /* rank */,

337  */
338  if (selectable > 1)

345  if (ranks == NULL)
346  return -DER_NOMEM;
347 
348  /* TODO: Choose ranks according to failure domains. */
349  j = 0;
350  for (i = 0; i < target_addrs->rl_nr; i++) {
351  if (j == ranks->rl_nr)
352  break;
353  if (i == i_rank_zero && selectable > 1)
354  /* This is rank 0 and it's not the only rank. */
355  continue;
356  D_DEBUG(DB_MD, "ranks[%d]: %u\n", j, target_addrs->rl_ranks[i]);
357  ranks->rl_ranks[j] = target_addrs->rl_ranks[i];
358  j++;
359  }
360  D_ASSERTF(j == ranks->rl_nr, "%d == %u\n", j, ranks->rl_nr);
361 
362  *ranksp = ranks;
363  return 0;

654 static void
655 pool_svc_step_down(struct pool_svc *svc)
656 {
657  d_rank_t rank;
658  int rc;
659 
660  D_ASSERT(svc->ps_state != POOL_SVC_DOWN);
661  D_DEBUG(DB_MD, DF_UUID": stepping down from "DF_U64"\n",
662  DP_UUID(svc->ps_uuid), svc->ps_term);
663 
664  /* Stop accepting new leader references. */
665  svc->ps_state = POOL_SVC_DRAINING;

668  * TODO: Abort all in-flight RPCs we sent, after aborting bcasts is
669  * implemented.
670  */

1175 static void
1176 pool_svc_stop(struct pool_svc *svc)
1177 {
1178  ABT_mutex_lock(svc->ps_mutex);
1179 
1180  if (svc->ps_stop) {
1181  D_DEBUG(DF_DSMS, DF_UUID": already stopping\n",
1182  DP_UUID(svc->ps_uuid));
1183  ABT_mutex_unlock(svc->ps_mutex);
1184  return;
1185  }
1186  D_DEBUG(DF_DSMS, DF_UUID": stopping pool service\n",
1187  DP_UUID(svc->ps_uuid));
1188  svc->ps_stop = true;
1189 
1190  if (svc->ps_state == POOL_SVC_UP ||
1191  svc->ps_state == POOL_SVC_UP_EMPTY)
1192  /*
1193  * The service has stepped up. If it is still the leader of
1194  * svc->ps_term, the following rdb_resign() call will trigger
1195  * the matching pool_svc_step_down_cb() callback in
1196  * svc->ps_term; otherwise, the callback must already be
1197  * pending. Either way, the service shall eventually enter the
1198  * POOL_SVC_DOWN state.
1199  */
1200  rdb_resign(svc->ps_db, svc->ps_term);
1201  while (svc->ps_state != POOL_SVC_DOWN)
1202  ABT_cond_wait(svc->ps_state_cv, svc->ps_mutex);
1203 
1204  ABT_mutex_unlock(svc->ps_mutex);
1205 
1206  ABT_mutex_lock(pool_svc_hash_lock);
1207  d_hash_rec_delete_at(&pool_svc_hash, &svc->ps_entry);
1208  ABT_mutex_unlock(pool_svc_hash_lock);
1209 }

1427  in->pri_ntgts != in->pri_tgt_ranks->rl_nr)
1428  D_GOTO(out, rc = -DER_PROTO);
1429  if (in->pri_ndomains != in->pri_domains.ca_count)

1489

1494  if (rc != 0)

1498  /*
1499  * The DB is no longer empty. Since the previous
1500  * pool_svc_step_up_cb() call didn't finish stepping up due to
1501  * an empty DB, and there hasn't been a pool_svc_step_down_cb()
1502  * call yet, we should call pool_svc_step_up() to finish
1503  * stepping up.
1504  */
1505  D_DEBUG(DF_DSMS, DF_UUID": trying to finish stepping up\n",
1506  DP_UUID(in->pri_op.pi_uuid));
1507  rc = pool_svc_step_up(svc);
1508  if (rc != 0) {
1509  D_ASSERT(rc != DER_UNINIT);
1510  /* TODO: Ask rdb to step down. */
1511  D_GOTO(out_svc, rc);
1512  }
1513  svc->ps_state = POOL_SVC_UP;
1514  }
1515 
1516 out_mutex:
1517  ABT_mutex_unlock(svc->ps_mutex);
1518 out_svc:
1519  ds_pool_set_hint(svc->ps_db, &out->pro_op.po_hint);
1520  pool_svc_put(svc);
1521 out:
1522  out->pro_op.po_rc = rc;
1523  D_DEBUG(DF_DSMS, DF_UUID": replying rpc %p: %d\n",
1524  DP_UUID(in->pri_op.pi_uuid), rpc, rc);
1525  crt_reply_send(rpc);
1526 }

1528 static int
1529 permitted(const struct pool_attr *attr, uint32_t uid, uint32_t gid,
1530  uint64_t capas)
1531 {
1532  int shift;
1533  uint32_t capas_permitted;
1534 
1535  /*
1536  * Determine which set of capability bits applies. See also the
1537  * comment/diagram for ds_pool_attr_mode in src/pool/srv_layout.h.
1538  */
1539  if (uid == attr->pa_uid)
1540  shift = DAOS_PC_NBITS * 2; /* user */
1541  else if (gid == attr->pa_gid)
1542  shift = DAOS_PC_NBITS; /* group */
1543  else
1544  shift = 0; /* other */
1545 
1546  /* Extract the applicable set of capability bits. */
1547  capas_permitted = (attr->pa_mode >> shift) & DAOS_PC_MASK;
1548 
1549  /* Only if all requested capability bits are permitted... */
1550  return (capas & capas_permitted) == capas;
1551 }

1710 {
1711  struct pool_connect_in *in = crt_req_get(rpc);
1712  struct pool_connect_out *out = crt_reply_get(rpc);
1713  struct pool_svc *svc;
1714  struct rdb_tx tx;
1715  daos_iov_t key;
1716  daos_iov_t value;
1717  struct pool_attr attr;
1718  struct pool_hdl hdl;
1719  daos_iov_t iv_iov;
1720  unsigned int iv_ns_id;

1723  int rc;
1724 
1725  D_DEBUG(DF_DSMS, DF_UUID": processing rpc %p: hdl="DF_UUID"\n",
1726  DP_UUID(in->pci_op.pi_uuid), rpc, DP_UUID(in->pci_op.pi_hdl));
1727 
1728  rc = pool_svc_lookup_leader(in->pci_op.pi_uuid, &svc,
1729  &out->pco_op.po_hint);
1730  if (rc != 0)
1731  D_GOTO(out, rc);
1732 
1733  /* sp_iv_ns will be destroyed when pool is destroyed,
1734  * see pool_free_ref()
1735  */
1736  D_ASSERT(svc->ps_pool != NULL);
1737  if (svc->ps_pool->sp_iv_ns == NULL) {
1738  rc = ds_iv_ns_create(rpc->cr_ctx, NULL,
1739  &iv_ns_id, &iv_iov,
1740  &svc->ps_pool->sp_iv_ns);
1741  if (rc)
1742  D_GOTO(out_svc, rc);
1743  } else {
1744  rc = ds_iv_global_ns_get(svc->ps_pool->sp_iv_ns, &iv_iov);
1745  if (rc)
1746  D_GOTO(out_svc, rc);
1747  }

1750  if (rc != 0)
1751  D_GOTO(out_svc, rc);
1752 
1753  rc = rdb_tx_begin(svc->ps_db, svc->ps_term, &tx);
1754  if (rc != 0)
1755  D_GOTO(out_svc, rc);
1756 
1757  ABT_rwlock_wrlock(svc->ps_lock);
1758 
1759  /* Check existing pool handles. */
1760  daos_iov_set(&key, in->pci_op.pi_hdl, sizeof(uuid_t));
1761  daos_iov_set(&value, &hdl, sizeof(hdl));
1762  rc = rdb_tx_lookup(&tx, &svc->ps_handles, &key, &value);
1763  if (rc == 0) {
1764  if (hdl.ph_capas == in->pci_capas) {
1765  /*
1766  * The handle already exists; only do the pool map
1767  * transfer.
1768  */
1769  skip_update = 1;
1770  } else {
1771  /* The existing one does not match the new one. */
1772  D_ERROR(DF_UUID": found conflicting pool handle\n",
1773  DP_UUID(in->pci_op.pi_uuid));
1774  D_GOTO(out_lock, rc = -DER_EXIST);
1775  }
1776  } else if (rc != -DER_NONEXIST) {
1777  D_GOTO(out_lock, rc);
1778  }
1779 
1780  rc = pool_attr_read(&tx, svc, &attr);
1781  if (rc != 0)
1782  D_GOTO(out_map_version, rc);
1783 
1784  if (!permitted(&attr, in->pci_uid, in->pci_gid, in->pci_capas)) {
1785  D_ERROR(DF_UUID": refusing connect attempt for uid %u gid %u "
1786  DF_X64"\n", DP_UUID(in->pci_op.pi_uuid), in->pci_uid,
1787  in->pci_gid, in->pci_capas);
1788  D_GOTO(out_map_version, rc = -DER_NO_PERM);
1789  }
1790 
1791  out->pco_mode = attr.pa_mode;
1792

1794  * Transfer the pool map to the client before adding the pool handle,
1795  * so that we don't need to worry about rolling back the transaction
1796  * when the tranfer fails. The client has already been authenticated
1797  * and authorized at this point. If an error occurs after the transfer
1798  * completes, then we simply return the error and the client will throw
1799  * its pool_buf away.
1800  */
1801  rc = transfer_map_buf(&tx, svc, rpc, in->pci_map_bulk,
1802  &out->pco_map_buf_size);
1803  if (rc != 0)
1804  D_GOTO(out_map_version, rc);
1805 
1806  if (skip_update)

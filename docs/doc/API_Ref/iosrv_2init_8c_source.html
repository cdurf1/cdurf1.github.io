140  mod = strsep(&run, ",");

145  mod = "pool";

147  strcmp(mod, "co") == 0)
148  mod = "cont";
149  else if (strcmp(mod, "management") == 0)
150  mod = "mgmt";

152  mod = "vos_srv";
153 
154  mod_facs = 0;
155  rc = dss_module_load(mod, &mod_facs);
156  if (rc != 0) {
157  D_ERROR("Failed to load module %s: %d\n",
158  mod, rc);
159  break;
160  }
161 
162  if (facs != NULL)
163  *facs |= mod_facs;
164 
165  mod = strsep(&run, ",");
166  }
167 
168  free(sep);

330 static int
331 parse(int argc, char **argv)
332 {
333  struct option opts[] = {
334  { "modules", required_argument, NULL, 'm' },
335  { "cores", required_argument, NULL, 'c' },
336  { "group", required_argument, NULL, 'g' },
337  { "storage", required_argument, NULL, 's' },
338  { "attach_info", optional_argument, NULL, 'a' },
339  { "help", no_argument, NULL, 'h' },
340  { NULL, 0, NULL, 0}
341  };
342  int rc = 0;
343  int c;
344 
345  /* load all of modules by default */
346  sprintf(modules, "%s", MODULE_LIST);
347  while ((c = getopt_long(argc, argv, "c:m:g:s:a::h", opts, NULL)) !=
348  -1) {
349  switch (c) {
350  case 'm':
351  if (strlen(optarg) > MAX_MODULE_OPTIONS) {
352  rc = -DER_INVAL;
353  usage(argv[0], stderr);
354  break;
355  }
356  snprintf(modules, sizeof(modules), "%s", optarg);
357  break;
358  case 'c': {
359  unsigned int nr;
360  char *end;
361 
362  nr = strtoul(optarg, &end, 10);
363  if (end == optarg || nr == ULONG_MAX) {
364  rc = -DER_INVAL;
365  break;
366  }
367  nr_threads = nr;
368  break;
369  }
370  case 'g':
371  server_group_id = optarg;
372  break;
373  case 's':
374  storage_path = optarg;
375  break;
376  case 'h':
377  usage(argv[0], stdout);
378  break;
379  case 'a':
380  save_attach_info = true;
381  attach_info_path = optarg;
382  break;
383  default:
384  usage(argv[0], stderr);
385  rc = -DER_INVAL;
386  }
387  if (rc < 0)
388  return rc;
389  }
390 
391  return 0;
392 }

215 /*
216  * Execute the messages on msg ring, call all registered pollers.
217  *
218  * \param[IN] ctxt Per-xstream NVMe context
219  *
220  * \returns Executed message count
221  */
222 size_t dss_nvme_poll(struct dss_nvme_context *ctxt)
223 {
224  struct daos_msg *msg;
225  struct daos_nvme_poller *poller;
226  size_t count;
227  uint64_t now = d_timeus_secdiff(0);
228 
229  /* NVMe context setup was skipped */
230  if (ctxt->dnc_msg_ring == NULL)
231  return 0;
232 
233  /* Process one msg on the msg ring */
234  count = spdk_ring_dequeue(ctxt->dnc_msg_ring, (void **)&msg, 1);
235  if (count > 0) {
236  msg->dm_fn(msg->dm_arg);
237  free(msg);
238  }
239 
240  /* Call all registered poller one by one */
241  d_list_for_each_entry(poller, &ctxt->dnc_pollers, dnp_link) {
242  if (poller->dnp_period_us != 0 && poller->dnp_expire_us < now)
243  continue;
244 
245  poller->dnp_fn(poller->dnp_arg);
246 
247  if (poller->dnp_period_us != 0)
248  poller->dnp_expire_us = now + poller->dnp_period_us;
249  }
250 
251  return count;
252 }

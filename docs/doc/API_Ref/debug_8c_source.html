132 static void
133 debug_fac_load_env(void)
134 {
135  char *fac_env;
136  char *fac_str;
137  char *cur;
138  int i;
139 
140  fac_env = getenv(DD_FAC_ENV);
141  if (fac_env == NULL)
142  return;
143 
144  D_STRNDUP(fac_str, fac_env, DAOS_FAC_MAX_LEN);
145  if (fac_str == NULL) {
146  D_PRINT_ERR("D_STRNDUP of fac mask failed");
147  return;
148  }
149 
150  /* Disable all facilities. The first one is ignored because NULL is
151  * always enabled.
152  */
153  for (i = 1; i < NUM_DBG_FAC_ENTRIES; i++)
154  debug_fac_dict[i].df_enabled = 0;
155 
156  cur = strtok(fac_str, DD_SEP);
157  while (cur != NULL) {
158  /* skip 1 because it's NULL and enabled always */
159  for (i = 1; i < NUM_DBG_FAC_ENTRIES; i++) {
160  if (debug_fac_dict[i].df_name != NULL &&
161  strncasecmp(cur, debug_fac_dict[i].df_name,
162  debug_fac_dict[i].df_name_size)
163  == 0) {
164  debug_fac_dict[i].df_enabled = 1;
165  break;
166  } else if (strncasecmp(cur, DD_FAC_ALL,
167  strlen(DD_FAC_ALL)) == 0) {
168  debug_fac_dict[i].df_enabled = 1;
169  }
170  }
171  cur = strtok(NULL, DD_SEP);
172  }
173  D_FREE(fac_str);
174 }

206 int
207 daos_debug_init(char *logfile)
208 {
209  int i;
210  int rc;
211  uint64_t allocd_dbg_bit;
212 
213  D_MUTEX_LOCK(&dd_lock);
214  if (dd_ref > 0) {
215  dd_ref++;
216  D_MUTEX_UNLOCK(&dd_lock);
217  return 0;
218  }
219 
220  if (getenv(D_LOG_FILE_ENV)) /* honor the env variable first */
221  logfile = getenv(D_LOG_FILE_ENV);
222  else if (logfile == NULL)
223  logfile = DAOS_LOG_DEFAULT;
224 
225  /* load other env variables */
226  debug_fac_load_env();
227 
228  rc = d_log_init_adv("DAOS", logfile, DLOG_FLV_FAC,
229  DLOG_INFO, DLOG_CRIT);
230  if (rc != 0) {
231  D_PRINT_ERR("Failed to init DAOS debug log: %d\n", rc);
232  goto failed_unlock;
233  }
234 
235  for (i = 0; i < NUM_DBG_FAC_ENTRIES; i++) {
236  if (!debug_fac_dict[i].df_enabled) {
237  /* redirect disabled facility to NULL */
238  *debug_fac_dict[i].df_idp = d_null_logfac;
239  continue;
240  }
241 
242  rc = debug_fac_register(&debug_fac_dict[i]);
243  if (rc != 0) {
244  D_PRINT_ERR("Failed to add DAOS facility %s: %d\n",
245  debug_fac_dict[i].df_name, rc);
246  goto failed_fini;
247  }
248  }
249 
250  /* Register DAOS debug bits with gurt used with DD_MASK env */
251  for (i = 0; i < NUM_DBG_BIT_ENTRIES; i++) {
252  /* register DAOS debug bit masks */
253  rc = d_log_dbg_bit_alloc(&allocd_dbg_bit,
254  daos_bit_dict[i].db_name,
255  daos_bit_dict[i].db_lname);
256  if (rc < 0) {
257  D_PRINT_ERR("Error allocating daos debug bit for %s",
258  daos_bit_dict[i].db_name);
259  return -DER_UNINIT;
260  }
261 
262  *daos_bit_dict[i].db_bit = allocd_dbg_bit;
263  }
264 
265  /* Register DAOS debug bit groups */
266  rc = d_log_dbg_grp_alloc(DB_GRP1, "daos_default");
267  if (rc < 0) {
268  D_PRINT_ERR("Error allocating daos debug group\n");
269  return -DER_UNINIT;
270  }
271 
272  /* Sync DAOS debug env with libgurt */
273  d_log_sync_mask();
274 
275  dd_ref = 1;
276  D_MUTEX_UNLOCK(&dd_lock);
277 
278  return 0;
279 
280 failed_fini:
281  debug_fini_locked();
282 
283 failed_unlock:
284  D_MUTEX_UNLOCK(&dd_lock);
285  return rc;
286 }

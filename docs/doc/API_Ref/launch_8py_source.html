32 def filelist(directory):
33  """
34  Create a list of test files contained in the provided path.
35  This is meant to deal primarily with the structure of tests
36  in the daos repo and would need to be changed to deal with
37  random directory trees of tests.
38  """
39 
40  test_files = []
41  test_pattern = "*.py"
42 
43  for path, dirs, files in os.walk(directory):
44  if not (path == directory or path == os.path.join(directory, 'util')):
45  for f in files:
46  if fnmatch.fnmatch(f, test_pattern):
47  test_files.append(os.path.join(path,f))
48  return test_files;

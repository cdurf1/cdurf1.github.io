478 void show_result(double now, double then, int vsize, char *test_name)
479 {
480  double duration, agg_duration;
481  double first_start;
482  double last_end;
483  double duration_max;
484  double duration_min;
485  double duration_sum;
486 
487  duration = now - then;
488 
489  if (ts_ctx.tsc_mpi_size > 1) {
490  MPI_Reduce(&then, &first_start, 1, MPI_DOUBLE,
491  MPI_MIN, 0, MPI_COMM_WORLD);
492  MPI_Reduce(&now, &last_end, 1, MPI_DOUBLE,
493  MPI_MAX, 0, MPI_COMM_WORLD);
494  } else {
495  first_start = then;
496  last_end = now;
497  }
498 
499  agg_duration = last_end - first_start;
500 
501  if (ts_ctx.tsc_mpi_size > 1) {
502  MPI_Reduce(&duration, &duration_max, 1, MPI_DOUBLE,
503  MPI_MAX, 0, MPI_COMM_WORLD);
504  MPI_Reduce(&duration, &duration_min, 1, MPI_DOUBLE,
505  MPI_MIN, 0, MPI_COMM_WORLD);
506  MPI_Reduce(&duration, &duration_sum, 1, MPI_DOUBLE,
507  MPI_SUM, 0, MPI_COMM_WORLD);
508  } else {
509  duration_max = duration_min = duration_sum = duration;
510  }
511 
512  if (ts_ctx.tsc_mpi_rank == 0) {
513  unsigned long total;
514  double bandwidth;
515  double latency;
516  double rate;
517 
518  total = ts_ctx.tsc_mpi_size *
519  ts_obj_p_cont * ts_dkey_p_obj *
520  ts_akey_p_dkey * ts_recx_p_akey;
521 
522  rate = total / agg_duration;
523  latency = (agg_duration * 1000 * 1000) / total;
524  bandwidth = (rate * vsize) / (1024 * 1024);
525

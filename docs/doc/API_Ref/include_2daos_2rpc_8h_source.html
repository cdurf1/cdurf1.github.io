148 static inline int
149 daos_rpc_register(struct daos_rpc *rpcs, struct daos_rpc_handler *handlers,
150  int mod_id)
151 {
152  struct daos_rpc *rpc;
153  uint32_t flags;
154  int rc;
155 
156  if (rpcs == NULL)
157  return 0;
158 
159  /* walk through the handler list and register each individual RPC */
160  for (rpc = rpcs; rpc->dr_opc != 0; rpc++) {
161  crt_opcode_t opcode;
162 
163  flags = 0;
164  if (rpc->dr_flags & DAOS_RPC_NO_REPLY)
165  flags |= CRT_RPC_FEAT_NO_REPLY;
166 
167  opcode = DAOS_RPC_OPCODE(rpc->dr_opc, mod_id, rpc->dr_ver);
168  if (handlers != NULL) {
169  struct daos_rpc_handler *handler;
170 
171  handler = daos_rpc_handler_find(handlers, rpc->dr_opc);
172  if (handler == NULL) {
173  D_ERROR("failed to find handler for opc %d\n",
174  rpc->dr_opc);
175  return rc;
176  }
177  if (handler->dr_corpc_ops.co_aggregate == NULL)
178  rc = crt_rpc_srv_register(opcode, flags,
179  rpc->dr_req_fmt,
180  handler->dr_hdlr);
181  else
182  rc = crt_corpc_register(opcode, rpc->dr_req_fmt,
183  handler->dr_hdlr,
184  &handler->dr_corpc_ops);
185  } else {
186  rc = crt_rpc_register(opcode, flags, rpc->dr_req_fmt);
187  }
188  if (rc)
189  return rc;
190  }
191  return 0;
192 }

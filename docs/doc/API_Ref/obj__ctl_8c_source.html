288 static int
289 ctl_cmd_run(char opc, char *args)
290 {
291  struct dts_io_credit *cred;
292  char *dkey = NULL;
293  char *akey = NULL;
294  char *val = NULL;
295  char *str;
296  char buf[CTL_BUF_LEN];
297  int rc;
298  bool opened = false;
299 
300  if (args) {
301  strcpy(buf, args);
302  str = daos_str_trimwhite(buf);
303  } else {
304  str = NULL;
305  }
306 
307  cred = dts_credit_take(&ctl_ctx);
308  D_ASSERT(cred);
309 
310  ctl_abits = 0;
311  memset(&ctl_oid, 0, sizeof(ctl_oid));
312  memset(&cred->tc_sgl, 0, sizeof(cred->tc_sgl));
313  memset(&cred->tc_iod, 0, sizeof(cred->tc_iod));
314  memset(&cred->tc_recx, 0, sizeof(cred->tc_recx));
315 
316  while (str && !isspace(*str) && *str != '\0') {
317  if (str[1] != CTL_SEP_VAL)
318  D_GOTO(out, rc = -1);
319 
320  switch (str[0]) {
321  case 'e':
322  case 'E':
323  ctl_abits |= CTL_ARG_EPOCH;
324  ctl_epoch = strtoul(&str[2], NULL, 0);
325  break;
326  case 'o':
327  case 'O':
328  ctl_abits |= CTL_ARG_OID;
329  ctl_oid.id_pub.lo = strtoul(&str[2], NULL, 0);
330  if (!daos_mode)
331  break;
332 
333  daos_obj_id_generate(&ctl_oid.id_pub, 0,
334  DAOS_OC_TINY_RW);
335  break;
336  case 'd':
337  case 'D':
338  ctl_abits |= CTL_ARG_DKEY;
339  dkey = &str[2];
340  break;
341  case 'a':
342  case 'A':
343  ctl_abits |= CTL_ARG_AKEY;
344  akey = &str[2];
345  break;
346  case 'v':
347  case 'V':
348  ctl_abits |= CTL_ARG_VAL;
349  val = &str[2];
350  break;
351  }
352 
353  str = strchr(str, CTL_SEP);
354  if (str != NULL) {
355  *str = '\0';
356  str++;
357  }
358  }
359 
360  if (ctl_abits & CTL_ARG_DKEY) {
361  strcpy(cred->tc_dbuf, dkey);
362  daos_iov_set(&cred->tc_dkey, cred->tc_dbuf,
363  strlen(cred->tc_dbuf) + 1);
364  }
365 
366  if (ctl_abits & CTL_ARG_AKEY) {
367  strcpy(cred->tc_abuf, akey);
368  daos_iov_set(&cred->tc_iod.iod_name, cred->tc_abuf,
369  strlen(cred->tc_abuf) + 1);
370 
371  cred->tc_iod.iod_type = DAOS_IOD_SINGLE;
372  cred->tc_iod.iod_size = -1; /* overwrite by CTL_ARG_VAL */
373  cred->tc_iod.iod_nr = 1; /* one recx */
374  cred->tc_iod.iod_recxs = &cred->tc_recx;
375  cred->tc_recx.rx_nr = 1;
376  }
377 
378  if (ctl_abits & CTL_ARG_VAL) {
379  cred->tc_iod.iod_size = strlen(val) + 1;
380  strcpy(cred->tc_vbuf, val);
381  daos_iov_set(&cred->tc_val, cred->tc_vbuf,
382  strlen(cred->tc_vbuf) + 1);
383  } else {
384  memset(cred->tc_vbuf, 0, ctl_ctx.tsc_cred_vsize);
385  daos_iov_set(&cred->tc_val, cred->tc_vbuf,
386  ctl_ctx.tsc_cred_vsize);
387  }
388  cred->tc_sgl.sg_nr = 1;
389  cred->tc_sgl.sg_iovs = &cred->tc_val;
390 
391  switch (opc) {
392  case 'u':
393  if (ctl_abits != CTL_ARG_ALL) {
394  ctl_print_usage();
395  D_GOTO(out, rc = -1);
396  } else {
397  ctl_obj_open(&opened);
398  }
399 
400  rc = ctl_update(cred);
401  break;
402  case 'f':
403  if (ctl_abits != (CTL_ARG_ALL & ~CTL_ARG_VAL)) {
404  ctl_print_usage();
405  D_GOTO(out, rc = -1);
406  } else {
407  ctl_obj_open(&opened);
408  }
409 
410  rc = ctl_fetch(cred);
411  if (rc == 0) {
412  D_PRINT("%s\n", strlen(cred->tc_vbuf) ?
413  cred->tc_vbuf : "<NULL>");
414  }
415  break;
416  case 'p':
417  if (!(ctl_abits & CTL_ARG_EPOCH) ||
418  !(ctl_abits & CTL_ARG_OID)) {
419  ctl_print_usage();
420  D_GOTO(out, rc = -1);
421  } else {
422  ctl_obj_open(&opened);
423  }
424 
425  rc = ctl_punch(cred);
426  break;
427  case 'l':
428  if (!(ctl_abits & CTL_ARG_OID)) {
429  ctl_print_usage();
430  D_GOTO(out, rc = -1);
431  } else {
432  if (!(ctl_abits & CTL_ARG_EPOCH))
433  ctl_epoch = DAOS_EPOCH_MAX;
434  ctl_obj_open(&opened);
435  }
436 
437  if (daos_mode)
438  rc = ctl_daos_list(cred);
439  else
440  rc = ctl_vos_list(cred);
441  break;
442  case 'h':
443  ctl_print_usage();
444  rc = 0;
445  break;
446  case 'q':
447  printf("quiting ...\n");
448  rc = -ESHUTDOWN;
449  break;
450  default:
451  D_GOTO(out, rc = -1);
452  }
453  if (rc && rc != -ESHUTDOWN)
454  D_GOTO(out, rc = -2);
455 out:
456  if (opened)
457  daos_obj_close(ctl_oh, NULL);
458 
459  switch (rc) {
460  case -2: /* real failure */
461  D_PRINT("Operation failed, rc=%d\n", rc);
462  break;
463 
464  case -1: /* invalid input */
465  D_PRINT("Invalid command or parameter string: %c, %s\n",
466  opc, args);
467  rc = 0; /* ignore input error */
468  break;
469  }
470  return rc;
471 }

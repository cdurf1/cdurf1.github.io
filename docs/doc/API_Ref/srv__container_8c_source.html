728 /* Close an array of handles, possibly belonging to different containers. */
729 static int
730 cont_close_hdls(struct cont_svc *svc, struct cont_tgt_close_rec *recs,
731  int nrecs, crt_context_t ctx)
732 {
733  int i;
734  int rc;
735 
736  D_ASSERTF(nrecs > 0, "%d\n", nrecs);
737  D_DEBUG(DF_DSMS, DF_CONT": closing %d recs: recs[0].hdl="DF_UUID
738  " recs[0].hce="DF_U64"\n", DP_CONT(svc->cs_pool_uuid, NULL),
739  nrecs, DP_UUID(recs[0].tcr_hdl), recs[0].tcr_hce);
740 
741  rc = cont_close_bcast(ctx, svc, recs, nrecs);
742  if (rc != 0)

746  * Use one TX per handle to avoid calling ds_cont_epoch_fini_hdl() more
747  * than once in a TX, in which case we would be attempting to query
748  * uncommitted updates. This could be optimized by adding container
749  * UUIDs into recs[i] and sorting recs[] by container UUIDs. Then we
750  * could maintain a list of deleted LREs and a list of deleted LHEs for
751  * each container while looping, and use the lists to update the GHCE
752  * once for each container. This approach enables us to commit only
753  * once (or when a TX becomes too big).
754  */
755  for (i = 0; i < nrecs; i++) {
756  struct rdb_tx tx;
757 
758  rc = rdb_tx_begin(svc->cs_db, cont_svc_term(svc), &tx);

760  break;
761  rc = cont_close_one_hdl(&tx, svc, ctx, recs[i].tcr_hdl);
762  if (rc != 0) {
763  rdb_tx_end(&tx);
764  break;
765  }
766  rc = rdb_tx_commit(&tx);
767  rdb_tx_end(&tx);
768  if (rc != 0)
769  break;
770  }
771 
772 out:
773  D_DEBUG(DF_DSMS, DF_CONT": leaving: %d\n",
774  DP_CONT(svc->cs_pool_uuid, NULL), rc);
775  return rc;
776 }

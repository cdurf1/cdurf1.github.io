88  handle = dlopen(name, RTLD_LAZY | RTLD_GLOBAL);
89  if (handle == NULL) {

92  }
93

95  D_ALLOC_PTR(lmod);

102  dlerror();
103

105  sprintf(name, "%s_module", modname);
106  smod = (struct dss_module *)dlsym(handle, name);
107 
108  /* check for errors */
109  err = dlerror();
110  if (err != NULL) {
111  D_ERROR("failed to load %s: %s\n", modname, err);
112  D_GOTO(err_lmod, rc = -DER_INVAL);
113  }

121  }

127  D_GOTO(err_hdl, rc = -DER_INVAL);

131  dss_register_key(smod->sm_key);
132  /* register client RPC handlers */

134  smod->sm_mod_id);
135  if (rc) {

137  modname, rc);

139  }

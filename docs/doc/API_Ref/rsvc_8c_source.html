81 rsvc_client_choose(struct rsvc_client *client, crt_endpoint_t *ep)
82 {
83  int chosen;
84 
85  D_DEBUG(DB_MD, DF_CLI"\n", DP_CLI(client));
86  if (client->sc_leader_known && client->sc_leader_aliveness > 0) {
87  chosen = client->sc_leader_index;
88  } else {
89  chosen = client->sc_next;
90  /* The hintless search is a round robin of all replicas. */
91  client->sc_next++;
92  client->sc_next %= client->sc_ranks->rl_nr;
93  }
94  D_ASSERTF(chosen >= 0 && chosen < client->sc_ranks->rl_nr, "%d\n",
95  chosen);
96  ep->ep_rank = client->sc_ranks->rl_ranks[chosen];
97  ep->ep_tag = 0;
98 }

100 /* Process an error without any leadership hint. */
101 static void
102 rsvc_client_process_error(struct rsvc_client *client, int rc,
103  const crt_endpoint_t *ep)
104 {
105  int leader_index = client->sc_leader_index;
106 
107  if (client->sc_leader_known && client->sc_leader_aliveness > 0 &&
108  ep->ep_rank == client->sc_ranks->rl_ranks[leader_index]) {
109  if (rc == -DER_NOTLEADER)
110  client->sc_leader_aliveness = 0;
111  else
112  client->sc_leader_aliveness--;
113  if (client->sc_leader_aliveness == 0) {
114  /*
115  * Gave up this leader. Start the hintless
116  * search.
117  */
118  client->sc_next = client->sc_leader_index + 1;
119  client->sc_next %= client->sc_ranks->rl_nr;
120  }
121  }
122 }

124 /* Process a result with a leadership hint. */
125 static void
126 rsvc_client_process_hint(struct rsvc_client *client,
127  const struct rsvc_hint *hint, bool from_leader,
128  const crt_endpoint_t *ep)
129 {
130  bool found;
131 
132  D_ASSERT(hint->sh_flags & RSVC_HINT_VALID);
133 
134  if (from_leader && hint->sh_rank != ep->ep_rank) {
135  D_ERROR("empty or invalid hint from leader rank %u: hint.term="
136  DF_U64" hint.rank=%u\n", ep->ep_rank, hint->sh_term,
137  hint->sh_rank);
138  return;
139  }
140 
141  if (client->sc_leader_known) {
142  if (hint->sh_term < client->sc_leader_term) {
143  D_DEBUG(DB_MD, "stale hint from rank %u: hint.term="
144  DF_U64" hint.rank=%u\n", ep->ep_rank,
145  hint->sh_term, hint->sh_rank);
146  return;
147  } else if (hint->sh_term == client->sc_leader_term) {
148  return;
149  }
150  }
151 
152  /* Got new leadership info. Cache it. */
153  found = daos_rank_list_find(client->sc_ranks, hint->sh_rank,
154  &client->sc_leader_index);
155  if (!found) {
156  int rc;
157 
158  D_DEBUG(DB_MD, "unknown replica from rank %u: hint.term="DF_U64
159  " hint.rank=%u\n", ep->ep_rank, hint->sh_term,
160  hint->sh_rank);
161  /* Append the unknown rank to tolerate user mistakes. */
162  rc = daos_rank_list_append(client->sc_ranks, hint->sh_rank);
163  if (rc != 0) {
164  D_DEBUG(DB_MD, "failed to append new rank: %d\n", rc);
165  return;
166  }
167  client->sc_leader_index = client->sc_ranks->rl_nr - 1;
168  }
169  client->sc_leader_term = hint->sh_term;
170  client->sc_leader_known = true;
171  /*
172  * If from_leader, set the aliveness to 2 so that upon a crt error
173  * we'll give the leader another try before turning to others. (If node
174  * failures were more frequent than message losses, then 1 should be
175  * used instead.)
176  */
177  client->sc_leader_aliveness = from_leader ? 2 : 1;
178  D_DEBUG(DB_MD, "new hint from rank %u: hint.term="DF_U64
179  " hint.rank=%u\n", ep->ep_rank, hint->sh_term, hint->sh_rank);
180 }

196 int
197 rsvc_client_complete_rpc(struct rsvc_client *client, const crt_endpoint_t *ep,
198  int rc_crt, int rc_svc, const struct rsvc_hint *hint)
199 {
200  D_DEBUG(DB_MD, DF_CLI"\n", DP_CLI(client));
201  /*
202  * Enumerate all cases of <rc_crt, rc_svc, hint>. Keep them at the same
203  * indentation level, please.
204  */
205  if (rc_crt != 0) {
206  D_DEBUG(DB_MD, "no reply from rank %u: rc_crt=%d\n",
207  ep->ep_rank, rc_crt);
208  rsvc_client_process_error(client, rc_crt, ep);
209  return RSVC_CLIENT_RECHOOSE;
210  } else if (rc_svc == -DER_NOTLEADER &&
211  (hint == NULL || !(hint->sh_flags & RSVC_HINT_VALID))) {
212  D_DEBUG(DB_MD, "non-leader reply without hint from rank %u\n",
213  ep->ep_rank);
214  rsvc_client_process_error(client, rc_svc, ep);
215  return RSVC_CLIENT_RECHOOSE;
216  } else if (rc_svc == -DER_NOTLEADER) {
217  D_DEBUG(DB_MD, "non-leader reply with hint from rank %u: "
218  "hint.term="DF_U64" hint.rank=%u\n", ep->ep_rank,
219  hint->sh_term, hint->sh_rank);
220  rsvc_client_process_error(client, rc_svc, ep);
221  rsvc_client_process_hint(client, hint, false /* !from_leader */,
222  ep);
223  return RSVC_CLIENT_RECHOOSE;
224  } else if (hint == NULL || !(hint->sh_flags & RSVC_HINT_VALID)) {
225  /* This may happen if the service wasn't found. */
226  D_DEBUG(DB_MD, "\"leader\" reply without hint from rank %u: "
227  "rc_svc=%d\n", ep->ep_rank, rc_svc);
228  return RSVC_CLIENT_PROCEED;
229  } else {
230  D_DEBUG(DB_MD, "leader reply with hint from rank %u: hint.term="
231  DF_U64" hint.rank=%u rc_svc=%d\n", ep->ep_rank,
232  hint->sh_term, hint->sh_rank, rc_svc);
233  rsvc_client_process_hint(client, hint, true /* from_leader */,
234  ep);
235  return RSVC_CLIENT_PROCEED;
236  }

295 ssize_t
296 rsvc_client_decode(void *buf, size_t len, struct rsvc_client *client)
297 {
298  struct rsvc_client_buf *p = buf;
299  bool swap = false;
300 
301  /* OK to access the struct? */
302  if (len < sizeof(*p)) {
303  D_ERROR("truncated buffer: %zu < %zu\n", len, sizeof(*p));
304  return -DER_IO;
305  }
306  /* Magic matches? */
307  if (p->scb_magic != rsvc_client_buf_magic) {
308  if (p->scb_magic == D_SWAP32(rsvc_client_buf_magic)) {
309  swap = true;
310  } else {
311  D_ERROR("bad buffer magic: %x\n", p->scb_magic);
312  return -DER_IO;
313  }
314  }
315  /* Swap the struct if needed. */
316  if (swap)
317  rsvc_client_buf_swap(p);
318  /* OK to access the ranks? */
319  if (p->scb_nranks == 0) {
320  D_ERROR("zero nranks\n");
321  return -DER_IO;
322  }
323  if (len < sizeof(*p) + sizeof(*p->scb_ranks) * p->scb_nranks) {
324  D_ERROR("truncated buffer: %zu < %zu\n", len,
325  sizeof(*p) + sizeof(*p->scb_ranks) * p->scb_nranks);
326  return -DER_IO;
327  }
328  /* Swap the ranks if needed. */
329  if (swap)
330  rsvc_client_buf_swap_ranks(p);
331  /* Copy the data. */
332  client->sc_ranks = daos_rank_list_alloc(p->scb_nranks);
333  if (client->sc_ranks == NULL)
334  return -DER_NOMEM;
335  memcpy(client->sc_ranks->rl_ranks, p->scb_ranks,
336  sizeof(*p->scb_ranks) * p->scb_nranks);
337  client->sc_leader_known = p->scb_leader_known == 0 ? false : true;
338  client->sc_leader_aliveness = p->scb_leader_aliveness;
339  client->sc_leader_term = p->scb_leader_term;
340  client->sc_leader_index = p->scb_leader_index;
341  client->sc_next = p->scb_next;
342  return sizeof(*p) + sizeof(*p->scb_ranks) * p->scb_nranks;
343 }

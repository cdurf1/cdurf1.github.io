246 daos_proc_iod(crt_proc_t proc, daos_iod_t *dvi)
247 {
248  crt_proc_op_t proc_op;
249  int rc;
250  int i;
251  uint32_t existing_flags = 0;
252 
253  if (proc == NULL || dvi == NULL) {
254  D_ERROR("Invalid parameter, proc: %p, data: %p.\n",
255  proc, dvi);
256  return -DER_INVAL;
257  }
258 
259  rc = daos_proc_iovec(proc, &dvi->iod_name);
260  if (rc != 0)
261  return rc;
262 
263  rc = daos_proc_csum_buf(proc, &dvi->iod_kcsum);
264  if (rc != 0)
265  return rc;
266 
267  rc = crt_proc_memcpy(proc, &dvi->iod_type, sizeof(daos_iod_type_t));
268  if (rc != 0)
269  return -DER_HG;
270 
271  rc = crt_proc_uint64_t(proc, &dvi->iod_size);
272  if (rc != 0)
273  return -DER_HG;
274 
275  rc = crt_proc_uint32_t(proc, &dvi->iod_nr);
276  if (rc != 0)
277  return -DER_HG;
278 
279  if (dvi->iod_nr == 0) {
280  D_ERROR("invalid I/O descriptor, iod_nr = 0\n");
281  return -DER_HG;
282  }
283 
284  rc = crt_proc_get_op(proc, &proc_op);
285  if (rc != 0)
286  return -DER_HG;
287 
288  if (proc_op == CRT_PROC_ENCODE) {
289  if (dvi->iod_type == DAOS_IOD_ARRAY && dvi->iod_recxs != NULL)
290  existing_flags |= IOD_REC_EXIST;
291  if (dvi->iod_csums != NULL)
292  existing_flags |= IOD_CSUM_EXIST;
293  if (dvi->iod_eprs != NULL)
294  existing_flags |= IOD_EPRS_EXIST;
295  }
296 
297  rc = crt_proc_uint32_t(proc, &existing_flags);
298  if (rc != 0)
299  return -DER_HG;
300 
301  if (proc_op == CRT_PROC_DECODE) {
302  if (existing_flags & IOD_REC_EXIST) {
303  D_ALLOC(dvi->iod_recxs,
304  dvi->iod_nr * sizeof(*dvi->iod_recxs));
305  if (dvi->iod_recxs == NULL)
306  D_GOTO(free, rc = -DER_NOMEM);
307  }
308 
309  if (existing_flags & IOD_CSUM_EXIST) {
310  D_ALLOC(dvi->iod_csums,
311  dvi->iod_nr * sizeof(*dvi->iod_csums));
312  if (dvi->iod_csums == NULL)
313  D_GOTO(free, rc = -DER_NOMEM);
314  }
315 
316  if (existing_flags & IOD_EPRS_EXIST) {
317  D_ALLOC(dvi->iod_eprs,
318  dvi->iod_nr * sizeof(*dvi->iod_eprs));
319  if (dvi->iod_eprs == NULL)
320  D_GOTO(free, rc = -DER_NOMEM);
321  }
322  }
323 
324  if (existing_flags & IOD_REC_EXIST) {
325  for (i = 0; i < dvi->iod_nr; i++) {
326  rc = daos_proc_recx(proc, &dvi->iod_recxs[i]);
327  if (rc != 0) {
328  if (proc_op == CRT_PROC_DECODE)
329  D_GOTO(free, rc);
330  return rc;
331  }
332  }
333  }
334 
335  if (existing_flags & IOD_CSUM_EXIST) {
336  for (i = 0; i < dvi->iod_nr; i++) {
337  rc = daos_proc_csum_buf(proc, &dvi->iod_csums[i]);
338  if (rc != 0) {
339  if (proc_op == CRT_PROC_DECODE)
340  D_GOTO(free, rc);
341  return rc;
342  }
343  }
344  }
345 
346  if (existing_flags & IOD_EPRS_EXIST) {
347  for (i = 0; i < dvi->iod_nr; i++) {
348  rc = daos_proc_epoch_range(proc, &dvi->iod_eprs[i]);
349  if (rc != 0) {
350  if (proc_op == CRT_PROC_DECODE)
351  D_GOTO(free, rc);
352  return rc;
353  }
354  }
355  }
356 
357  if (proc_op == CRT_PROC_FREE) {
358 free:
359  if (dvi->iod_recxs != NULL)
360  D_FREE(dvi->iod_recxs);
361  if (dvi->iod_csums != NULL)
362  D_FREE(dvi->iod_csums);
363  if (dvi->iod_eprs != NULL)
364  D_FREE(dvi->iod_eprs);
365  }
366 
367  return rc;
368 }

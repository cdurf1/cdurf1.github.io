149 int
150 umem_tx_errno(int err)
151 {
152  if (err == 0) {
153  err = pmemobj_tx_errno();
154  if (err == ENOMEM) /* pmdk returns ENOMEM for out of space */
155  err = ENOSPC;
156  }
157 
158  if (err == 0) {
159  D_ERROR("Transaction aborted for unknown reason\n");
160  return -DER_UNKNOWN;
161  }
162 
163  if (err < 0) {
164  if (err < -DER_ERR_GURT_BASE)
165  return err; /* aborted by DAOS */
166 
167  D_ERROR("pmdk returned negative errno %d\n", err);
168  err = -err;
169  }
170  return daos_errno2der(err);
171 }

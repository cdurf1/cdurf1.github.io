81 daos_eq_lib_init()
82 {
83  bool singleton = false;
84  uint32_t flags;
85  int rc;
86 
87  D_MUTEX_LOCK(&daos_eq_lock);
88  if (eq_ref > 0) {
89  eq_ref++;
90  D_GOTO(unlock, rc = 0);
91  }
92 
93  /*
94  * If DAOS_SINGLETON_CLI ENV set as non-zero value, then daos client
95  * works in singleton mode that without cient-side rank but with the
96  * benefit of independent with PMIx.
97  */
98  d_getenv_bool("DAOS_SINGLETON_CLI", &singleton);
99  flags = singleton ? CRT_FLAG_BIT_SINGLETON : 0;
100  rc = crt_init(NULL, flags);
101  if (rc != 0) {
102  D_ERROR("failed to initialize crt: %d\n", rc);
103  D_GOTO(unlock, rc);
104  }
105 
106  /* use a global shared context for all eq for now */
107  rc = crt_context_create(&daos_eq_ctx);
108  if (rc != 0) {
109  D_ERROR("failed to create client context: %d\n", rc);
110  D_GOTO(crt, rc);
111  }
112 
114  rc = tse_sched_init(&daos_sched_g, NULL, daos_eq_ctx);
115  if (rc != 0)
116  D_GOTO(crt, rc);
117 
118  eq_ref = 1;
119 
120 unlock:
121  D_MUTEX_UNLOCK(&daos_eq_lock);
122  return rc;
123 crt:
124  crt_finalize();

330 static int
331 daos_event_complete_locked(struct daos_eq_private *eqx,
332  struct daos_event_private *evx, int rc)
333 {
334  struct daos_event_private *parent_evx = evx->evx_parent;
335  struct daos_eq *eq = NULL;
336  daos_event_t *ev = daos_evx2ev(evx);
337 
338  if (eqx != NULL)
339  eq = daos_eqx2eq(eqx);
340 
341  evx->evx_status = DAOS_EVS_COMPLETED;
342  rc = daos_event_complete_cb(evx, rc);
343  ev->ev_error = rc;
344 
345  if (parent_evx != NULL) {
346  daos_event_t *parent_ev = daos_evx2ev(parent_evx);
347 
348  D_ASSERT(parent_evx->evx_nchild_running > 0);
349  parent_evx->evx_nchild_running--;
350 
351  D_ASSERT(parent_evx->evx_nchild_comp <
352  parent_evx->evx_nchild);
353  parent_evx->evx_nchild_comp++;
354 
355  if (parent_evx->evx_nchild_comp < parent_evx->evx_nchild) {
356  /* Not all children have completed yet */
357  parent_ev->ev_error = parent_ev->ev_error ?: rc;
358  return 0;
359  }
360 
361  /* If the parent is not launched yet, let's return */
362  if (parent_evx->evx_status == DAOS_EVS_READY)
363  return 0;
364 
365  /* If the parent was completed or aborted, we can return */
366  if (parent_evx->evx_status == DAOS_EVS_COMPLETED ||
367  parent_evx->evx_status == DAOS_EVS_ABORTED)
368  return 0;
369 
370  /* If the parent is not a barrier it will complete on its own */
371  if (!parent_evx->is_barrier)
372  return 0;
373 
374  /* Complete the barrier parent */
375  D_ASSERT(parent_evx->evx_status == DAOS_EVS_RUNNING);
376  parent_evx->evx_status = DAOS_EVS_COMPLETED;
377  rc = daos_event_complete_cb(parent_evx, rc);
378 
379  parent_ev->ev_error = parent_ev->ev_error ?: rc;
380  evx = parent_evx;
381  }
382 
383  if (eq != NULL) {
384  D_ASSERT(!d_list_empty(&evx->evx_link));
385  d_list_move_tail(&evx->evx_link, &eq->eq_comp);
386  eq->eq_n_comp++;
387  D_ASSERT(eq->eq_n_running > 0);
388  eq->eq_n_running--;
389  }
390 
391  return 0;
392 }

394 int
395 daos_event_launch(struct daos_event *ev)
396 {
397  struct daos_event_private *evx = daos_ev2evx(ev);
398  struct daos_eq_private *eqx = NULL;
399  int rc = 0;
400 
401  if (evx->evx_status != DAOS_EVS_READY) {
402  D_ERROR("Event status should be INIT: %d\n", evx->evx_status);
403  return -DER_NO_PERM;
404  }
405 
406  if (evx->evx_nchild > evx->evx_nchild_running + evx->evx_nchild_comp) {
407  D_ERROR("Launch all children before launch the parent.\n");
408  rc = -DER_NO_PERM;
409  goto out;
410  }
411 
412  if (!daos_handle_is_inval(evx->evx_eqh)) {
413  eqx = daos_eq_lookup(evx->evx_eqh);
414  if (eqx == NULL) {
415  D_ERROR("Can't find eq from handle %"PRIu64"\n",
416  evx->evx_eqh.cookie);
417  return -DER_NONEXIST;
418  }
419 
420  D_MUTEX_LOCK(&eqx->eqx_lock);
421  if (eqx->eqx_finalizing) {
422  D_ERROR("Event queue is in progress of finalizing\n");
423  rc = -DER_NONEXIST;
424  goto out;
425  }
426  }
427 
428  daos_event_launch_locked(eqx, evx);
429 
430  /*
431  * If all child events completed before a barrier parent was launched,
432  * complete the parent.
433  */
434  if (evx->is_barrier && evx->evx_nchild > 0 &&
435  evx->evx_nchild == evx->evx_nchild_comp) {
436  D_ASSERT(evx->evx_nchild_running == 0);
437  daos_event_complete_locked(eqx, evx, rc);
438  }
439  out:
440  if (eqx != NULL)
441  D_MUTEX_UNLOCK(&eqx->eqx_lock);
442 
443  if (eqx != NULL)
444  daos_eq_putref(eqx);
445 
446  return rc;
447 }

498 static int
499 ev_progress_cb(void *arg)
500 {
501  struct ev_progress_arg *epa = (struct ev_progress_arg *)arg;
502  struct daos_event_private *evx = epa->evx;
503  struct daos_eq_private *eqx = epa->eqx;
504 
505  tse_sched_progress(evx->evx_sched);
506 
508  if (evx->evx_status == DAOS_EVS_READY)
509  return 1;
510 
512  if (evx->evx_status != DAOS_EVS_COMPLETED &&
513  evx->evx_status != DAOS_EVS_ABORTED)
514  return 0;
515 
517  if (evx->evx_nchild_running > 0)
518  return 0;
519 
520  /*
521  * Change status of event to INIT only if event is not in EQ and get
522  * out.
523  */
524  if (daos_handle_is_inval(evx->evx_eqh)) {
525  evx->evx_status = DAOS_EVS_READY;
526  return 1;
527  }
528 
530  D_MUTEX_LOCK(&eqx->eqx_lock);
531 
532  /*
533  * if the EQ was finalized from under us, just update the event status
534  * and return.
535  */
536  if (eqx->eqx_finalizing) {
537  evx->evx_status = DAOS_EVS_READY;
538  D_ASSERT(d_list_empty(&evx->evx_link));
539  D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
540  return 1;
541  }
542 
543  /*
544  * Check again if the event is still in completed/aborted state, then
545  * remove it from the event queue.
546  */
547  if (evx->evx_status == DAOS_EVS_COMPLETED ||
548  evx->evx_status == DAOS_EVS_ABORTED) {
549  struct daos_eq *eq = daos_eqx2eq(eqx);
550 
551  evx->evx_status = DAOS_EVS_READY;
552  D_ASSERT(eq->eq_n_comp > 0);
553  eq->eq_n_comp--;
554  d_list_del_init(&evx->evx_link);
555  }
556 
557  D_ASSERT(evx->evx_status == DAOS_EVS_READY);
558  D_MUTEX_UNLOCK(&eqx->eqx_lock);
559 
560  return 1;
561 }

640 static int
641 eq_progress_cb(void *arg)
642 {
643  struct eq_progress_arg *epa = (struct eq_progress_arg *)arg;
644  struct daos_event *ev;
645  struct daos_eq *eq;
646  struct daos_event_private *evx;
647  struct daos_event_private *tmp;
648 
649  eq = daos_eqx2eq(epa->eqx);
650 
651  tse_sched_progress(&epa->eqx->eqx_sched);
652 
653  D_MUTEX_LOCK(&epa->eqx->eqx_lock);
654  d_list_for_each_entry_safe(evx, tmp, &eq->eq_comp, evx_link) {
655  D_ASSERT(eq->eq_n_comp > 0);
656 
658  if (evx->evx_nchild_running > 0)
659  continue;
660 
661  eq->eq_n_comp--;
662 
663  d_list_del_init(&evx->evx_link);
664  D_ASSERT(evx->evx_status == DAOS_EVS_COMPLETED ||
665  evx->evx_status == DAOS_EVS_ABORTED);
666  evx->evx_status = DAOS_EVS_READY;
667 
668  if (epa->events != NULL) {
669  ev = daos_evx2ev(evx);
670  epa->events[epa->count++] = ev;
671  }
672 
673  D_ASSERT(epa->count <= epa->n_events);
674  if (epa->count == epa->n_events)
675  break;
676  }
677 
678  /* exit once there are completion events */
679  if (epa->count > 0) {
680  D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
681  return 1;
682  }
683 
684  /* no completion event, eq::eq_comp is empty */
685  if (epa->eqx->eqx_finalizing) { /* no new event is coming */
686  D_ASSERT(d_list_empty(&eq->eq_running));
687  D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
688  return -DER_NONEXIST;
689  }
690 
691  /* wait only if there are running events? */
692  if (epa->wait_running && d_list_empty(&eq->eq_running)) {
693  D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
694  return 1;
695  }
696 
697  D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
698 
700  return 0;
701 }

1019 daos_event_fini(struct daos_event *ev)
1020 {

1023  struct daos_eq *eq = NULL;

1028  if (eqx == NULL)

1030  eq = daos_eqx2eq(eqx);

1035  struct daos_event_private *tmp;
1036

1038  struct daos_event_private, evx_link);
1039  D_ASSERTF(tmp->evx_status == DAOS_EVS_READY ||
1040  tmp->evx_status == DAOS_EVS_COMPLETED ||

1045  tmp->evx_status != DAOS_EVS_COMPLETED &&

1059  tmp->evx_parent = NULL;
1060  }

1063  if (evx->evx_parent != NULL) {

1085  eq->eq_n_running--;

1089  eq->eq_n_comp--;

1091  }

1098 }

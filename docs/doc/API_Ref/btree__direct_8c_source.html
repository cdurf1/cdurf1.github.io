416 static int
417 sk_btr_kv_operate(enum sk_btr_opc opc, char *str, bool verbose)
418 {
419  int count = 0;
420  umem_id_t rec_mmid;
421  int rc;
422 
423  if (daos_handle_is_inval(sk_toh)) {
424  D_ERROR("Can't find opened tree\n");
425  return -1;
426  }
427 
428  while (str != NULL && !isspace(*str) && *str != '\0') {
429  char *val = NULL;
430  char *key = str;
431  daos_iov_t key_iov;
432  daos_iov_t val_iov;
433 
434  if (opc == BTR_OPC_UPDATE) {
435  val = strchr(str, SK_SEP_VAL);
436  if (val == NULL) {
437  D_ERROR("Invalid parameters %s (errno %d)\n",
438  str, errno);
439  return -1;
440  }
441  *val = 0;
442  str = ++val;
443  }
444 
445  str = strchr(str, SK_SEP);
446  if (str != NULL) {
447  *str = '\0';
448  str++;
449  }
450 
451  daos_iov_set(&key_iov, key, strlen(key) + 1);
452  switch (opc) {
453  default:
454  return -1;
455  case BTR_OPC_UPDATE:
456  daos_iov_set(&val_iov, val, strlen(val) + 1);
457  rc = dbtree_update(sk_toh, &key_iov, &val_iov);
458  if (rc != 0) {
459  D_ERROR("Failed to update %s:%s\n", key, val);
460  return -1;
461  }
462  break;
463 
464  case BTR_OPC_DELETE:
465  rc = dbtree_delete(sk_toh, &key_iov, NULL);
466  if (rc != 0) {
467  D_ERROR("Failed to delete %s\n", key);
468  return -1;
469  }
470  if (verbose)
471  D_PRINT("Deleted key %s\n", key);
472 
473  if (dbtree_is_empty(sk_toh) && verbose)
474  D_PRINT("Tree is empty now\n");
475  break;
476 
477  case BTR_OPC_DELETE_RETAIN:
478  rc = dbtree_delete(sk_toh, &key_iov, &rec_mmid);
479  if (rc != 0) {
480  D_ERROR("Failed to delete %s\n", key);
481  return -1;
482  }
483 
485  rc = btr_rec_verify_delete(&rec_mmid, &key_iov);
486  if (rc != 0) {
487  D_ERROR("Failed to verify and delete rec\n");
488  return -1;
489  }
490 
491  if (verbose)
492  D_PRINT("Deleted key %s\n", key);
493  if (dbtree_is_empty(sk_toh) && verbose)
494  D_PRINT("Tree is empty now\n");
495  break;
496 
497  case BTR_OPC_LOOKUP:
498  D_DEBUG(DB_TEST, "Looking for %s\n", key);
499 
500  daos_iov_set(&val_iov, NULL, 0); /* get address */
501  rc = dbtree_lookup(sk_toh, &key_iov, &val_iov);
502  if (rc != 0) {
503  D_ERROR("Failed to lookup %s\n", key);
504  return -1;
505  }
506 
507  if (verbose) {
508  D_PRINT("Found key %s, value %s\n",
509  key, (char *)val_iov.iov_buf);
510  }
511  break;
512  }
513  count++;
514  }
515  if (verbose)
516  D_PRINT("%s %d record(s)\n", btr_opc2str(opc), count);
517  return 0;
518 }

545 static int
546 sk_btr_iterate(char *args)
547 {
548  daos_handle_t ih;
549  int i;
550  int d;
551  int del;
552  int rc;
553  int opc;
554  char *err;
555 
556  if (daos_handle_is_inval(sk_toh)) {
557  D_ERROR("Can't find opened tree\n");
558  return -1;
559  }
560 
561  rc = dbtree_iter_prepare(sk_toh, BTR_ITER_EMBEDDED, &ih);
562  if (rc != 0) {
563  err = "initialize";
564  goto failed;
565  }
566 
567  if (args[0] == 'b')
568  opc = BTR_PROBE_LAST;
569  else
570  opc = BTR_PROBE_FIRST;
571 
572  if (strlen(args) >= 3 && args[1] == ':')
573  del = atoi(&args[2]);
574  else
575  del = 0;
576 
577  for (i = d = 0;; i++) {
578  char *key;
579  daos_iov_t key_iov;
580  daos_iov_t val_iov;
581 
582  if (i == 0 || (del != 0 && d <= del)) {
583  rc = dbtree_iter_probe(ih, opc, NULL, NULL);
584  if (rc == -DER_NONEXIST)
585  break;
586 
587  if (rc != 0) {
588  err = "probe";
589  goto failed;
590  }
591 
592  if (del != 0) {
593  if (d == del)
594  del = d = 0; /* done */
595  else
596  d++;
597  }
598  }
599 
600  daos_iov_set(&key_iov, NULL, 0);
601  daos_iov_set(&val_iov, NULL, 0);
602  rc = dbtree_iter_fetch(ih, &key_iov, &val_iov, NULL);
603  if (rc != 0) {
604  err = "fetch";
605  goto failed;
606  }
607 
608  key = key_iov.iov_buf;
609 
610  if (d != 0) { /* delete */
611  rc = dbtree_iter_delete(ih, NULL);
612  if (rc != 0) {
613  err = "delete";
614  goto failed;
615  }
616 
617  } else { /* iterate */
618  D_PRINT("%s: %s\n", key, (char *)val_iov.iov_buf);
619 
620  if (opc == BTR_PROBE_LAST)
621  rc = dbtree_iter_prev(ih);
622  else
623  rc = dbtree_iter_next(ih);
624 
625  if (rc == -DER_NONEXIST)
626  break;
627 
628  if (rc != 0) {
629  err = "move";
630  goto failed;
631  }
632  }
633  }
634 
635  D_PRINT("%s iterator: total %d, deleted %d\n",
636  opc == BTR_PROBE_FIRST ? "forward" : "backward", i, d);
637  dbtree_iter_finish(ih);
638  return 0;
639  failed:
640  D_PRINT("Iterator %s failed: %d\n", err, rc);
641  dbtree_iter_finish(ih);
642  return -1;
643 }

880  sk_btr_mix_keys(kv, key_nr);
881  for (i = 0; i < key_nr;) {
882  int j;

884  D_PRINT("Batch lookup %d records.\n", key_nr - i);

892  return -1;
893  }

895 
896  D_PRINT("Batch delete %d records.\n",
897  min(key_nr - i, DEL_BATCH));

900  key = kv[i].key.iov_buf;

904  if (rc != 0) {

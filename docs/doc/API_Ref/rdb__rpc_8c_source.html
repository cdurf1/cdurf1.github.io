60  rc = crt_proc_get_op(proc, &proc_op);
61  if (rc != 0)
62  return -DER_HG;
63  rc = crt_proc_uint32_t(proc, &e->term);
64  if (rc != 0)

68  return -DER_HG;

291 void
292 rdb_recvd(void *arg)
293 {
294  struct rdb *db = arg;
295 
296  D_DEBUG(DB_ANY, DF_DB": recvd starting\n", DP_DB(db));
297  for (;;) {
298  struct rdb_raft_rpc *rrpc = NULL;
299  bool stop;
300 
301  ABT_mutex_lock(db->d_mutex);
302  for (;;) {
303  stop = db->d_stop;
304  if (!d_list_empty(&db->d_replies)) {
305  rrpc = d_list_entry(db->d_replies.next,
306  struct rdb_raft_rpc,
307  drc_entry);
308  d_list_del_init(&rrpc->drc_entry);
309  break;
310  }
311  if (stop)
312  break;
313  ABT_cond_wait(db->d_replies_cv, db->d_mutex);
314  }
315  ABT_mutex_unlock(db->d_mutex);
316  if (rrpc == NULL) {
317  D_ASSERT(stop);
318  /* The queue is empty and we are asked to stop. */
319  break;
320  }
321  /*
322  * The queue has pending replies. If we are asked to stop, skip
323  * the processing but still free the RPCs until the queue
324  * become empty.
325  */
326  if (!stop)
327  rdb_raft_process_reply(db, rrpc->drc_node,
328  rrpc->drc_rpc);
329  rdb_raft_free_request(db, rrpc->drc_rpc);
330  rdb_free_raft_rpc(rrpc);
331  ABT_thread_yield();
332  }
333  D_DEBUG(DB_ANY, DF_DB": recvd stopping\n", DP_DB(db));
334 }

336 static void
337 rdb_raft_rpc_cb(const struct crt_cb_info *cb_info)
338 {
339  struct rdb_raft_rpc *rrpc = cb_info->cci_arg;
340  struct rdb *db = rrpc->drc_db;
341  crt_opcode_t opc = opc_get(cb_info->cci_rpc->cr_opc);
342  int rc = cb_info->cci_rc;
343 
344  D_DEBUG(DB_MD, DF_DB": opc=%u rank=%u rtt=%f\n", DP_DB(db), opc,
345  rrpc->drc_rpc->cr_ep.ep_rank, ABT_get_wtime() - rrpc->drc_sent);
346  ABT_mutex_lock(db->d_mutex);
347  if (rc != 0 || db->d_stop) {
348  if (rc != -DER_CANCELED)
349  D_ERROR(DF_DB": RPC %x to rank %u failed: %d\n",
350  DP_DB(rrpc->drc_db), opc,
351  rrpc->drc_rpc->cr_ep.ep_rank, rc);
352  /*
353  * Drop this RPC, assuming that raft will make a new one. If we
354  * are stopping, rdb_recvd() might have already stopped. Hence,
355  * we shall not add any new items to db->d_replies.
356  */
357  d_list_del_init(&rrpc->drc_entry);
358  ABT_mutex_unlock(db->d_mutex);
359  rdb_raft_free_request(db, rrpc->drc_rpc);
360  rdb_free_raft_rpc(rrpc);
361  return;
362  }
363  /* Move this RPC to db->d_replies for rdb_recvd(). */
364  d_list_move_tail(&rrpc->drc_entry, &db->d_replies);
365  ABT_cond_broadcast(db->d_replies_cv);
366  ABT_mutex_unlock(db->d_mutex);
367 }

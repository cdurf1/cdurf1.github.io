233  if (rc != 0)
234  break;
235

237  rc = rebuild_out->ro_status;
238  break;
239  }
240

254  }
255

259  break;
260  }
261  }
262 
263  if (targets)
264  D_FREE(targets);
265 
266  if (target_failed) {
267  /* Remote target has failed, no need retry */
268  D_DEBUG(DB_REBUILD, "target %d has been failed\n",
269  tgt_id);
270  break;
271  }
272  ABT_thread_yield();
273  }
274 out:
275  if (rpc)
276  crt_req_decref(rpc);
277  if (oids != NULL)
278  D_FREE(oids);
279  if (uuids != NULL)
280  D_FREE(uuids);
281  if (shards != NULL)
282  D_FREE(shards);
283  if (arg != NULL)
284  D_FREE_PTR(arg);
285 
286  return rc;

499 static int
500 placement_check(uuid_t co_uuid, daos_unit_oid_t oid, void *data)
501 {
502  struct rebuild_scan_arg *arg = data;
503  struct rebuild_tgt_pool_tracker *rpt = arg->rpt;
504  struct pl_obj_layout *layout = NULL;
505  struct pl_map *map = NULL;
506  struct daos_obj_md md;
507  uint32_t tgt_rebuild;
508  unsigned int shard_rebuild;
509  d_rank_t myrank;
510  int rc;
511 
512  if (rpt->rt_abort)
513  return 1;
514 
515  dc_obj_fetch_md(oid.id_pub, &md);
516  while (1) {
517  struct pool_map *poolmap;
518 
519  map = pl_map_find(rpt->rt_pool_uuid, oid.id_pub);
520  if (map == NULL) {
521  D_ERROR(DF_UOID"Cannot find valid placement map"
522  DF_UUID"\n", DP_UOID(oid),
523  DP_UUID(rpt->rt_pool_uuid));
524  D_GOTO(out, rc = -DER_INVAL);
525  }
526 
527  poolmap = rebuild_pool_map_get(rpt->rt_pool);
528  if (pl_map_version(map) >= pool_map_get_version(poolmap)) {
529  /* very likely we just break out from here, unless
530  * there is a cascading failure.
531  */
532  rebuild_pool_map_put(poolmap);
533  break;
534  }
535  pl_map_decref(map);
536 
537  pl_map_update(rpt->rt_pool_uuid, poolmap, false);
538  rebuild_pool_map_put(poolmap);
539  }
540 
541  crt_group_rank(rpt->rt_pool->sp_group, &myrank);
542 
543  rc = pl_obj_find_rebuild(map, &md, NULL, arg->tgp_failed->tg_ver,
544  &tgt_rebuild, &shard_rebuild);
545  if (rc <= 0) /* No need rebuild */
546  D_GOTO(out, rc);
547 
548  D_DEBUG(DB_REBUILD, "rebuild obj "DF_UOID"/"DF_UUID"/"DF_UUID
549  " on %d for shard %d\n", DP_UOID(oid), DP_UUID(co_uuid),
550  DP_UUID(rpt->rt_pool_uuid), tgt_rebuild, shard_rebuild);
551 
552  /* During rebuild test, it will manually exclude some target to
553  * trigger the rebuild, then later add it back, so some objects
554  * might exist on some illegal target, so they might use its "own"
555  * target as the spare target, let's skip these object now.
556  * When we have better support from CART exclude/addback, myrank
557  * should always equal to tgt_rebuild. XXX
558  */
559  if (myrank != tgt_rebuild) {
560  rc = rebuild_object_insert(arg, tgt_rebuild, shard_rebuild,
561  rpt->rt_pool_uuid, co_uuid, oid);
562  if (rc)
563  D_GOTO(out, rc);
564  } else {
565  D_DEBUG(DB_REBUILD, "skip "DF_UOID", not send it to its own.\n",
566  DP_UOID(oid));
567  rc = 0;
568  }
569 out:
570  if (layout != NULL)
571  pl_obj_layout_free(layout);
572 
573  if (map != NULL)
574  pl_map_decref(map);
575 
576  return rc;
577 }

724 rebuild_tgt_scan_handler(crt_rpc_t *rpc)

727  struct rebuild_scan_out *ro;

729  struct umem_attr uma;
730  struct rebuild_tgt_pool_tracker *rpt = NULL;
731  int rc;
732 
733  rsi = crt_req_get(rpc);
734  D_ASSERT(rsi != NULL);
735 
736  D_DEBUG(DB_REBUILD, "%d scan rebuild for "DF_UUID" ver %d/%d\n",
737  dss_get_module_info()->dmi_tid, DP_UUID(rsi->rsi_pool_uuid),
738  rsi->rsi_pool_map_ver, rsi->rsi_rebuild_ver);
739 
740  /* check if the rebuild is already started */
741  rpt = rpt_lookup(rsi->rsi_pool_uuid, rsi->rsi_rebuild_ver);
742  if (rpt != NULL) {
743  /* Rebuild should never skip the version */
744  D_ASSERTF(rsi->rsi_rebuild_ver == rpt->rt_rebuild_ver,
745  "rsi_rebuild_ver %d != rt_rebuild_ver %d\n",
746  rsi->rsi_rebuild_ver, rpt->rt_rebuild_ver);
747 
748  D_DEBUG(DB_REBUILD, DF_UUID" already started.\n",
749  DP_UUID(rsi->rsi_pool_uuid));
750 
751  /* Ignore the rebuild trigger request if it comes from
752  * an old or same leader.
753  */
754  if (rsi->rsi_leader_term <= rpt->rt_leader_term)
755  D_GOTO(out, rc = 0);
756 
757  if (rpt->rt_pool->sp_iv_ns != NULL &&
758  rpt->rt_pool->sp_iv_ns->iv_master_rank !=
759  rsi->rsi_master_rank) {
760  D_DEBUG(DB_REBUILD, DF_UUID" master rank"
761  " %d -> %d term "DF_U64" -> "DF_U64"\n",
762  DP_UUID(rpt->rt_pool_uuid),
763  rpt->rt_pool->sp_iv_ns->iv_master_rank,
764  rsi->rsi_master_rank,
765  rpt->rt_leader_term,
766  rsi->rsi_leader_term);
767  /* Update master rank */
768  rc = ds_pool_iv_ns_update(rpt->rt_pool,
769  rsi->rsi_master_rank,
770  &rsi->rsi_ns_iov,
771  rsi->rsi_ns_id);
772  if (rc)
773  D_GOTO(out, rc);
774 
775  /* If this is the old leader, then also stop the rebuild
776  * tracking ULT.
777  */
778  ds_rebuild_leader_stop();
779  }
780 
781  rpt->rt_leader_term = rsi->rsi_leader_term;
782 
783  D_GOTO(out, rc = 0);
784  }
785 
786  if (daos_fail_check(DAOS_REBUILD_TGT_START_FAIL))
787  D_GOTO(out, rc = -DER_INVAL);
788 
789  rc = rebuild_tgt_prepare(rpc, &rpt);
790  if (rc)
791  D_GOTO(out, rc);
792 
793  rpt_get(rpt);
794  rc = dss_ult_create(rebuild_tgt_status_check, rpt, -1, NULL);
795  if (rc) {
796  rpt_put(rpt);
797  D_GOTO(out, rc);
798  }
799 
800  /* step-1: parameters for scanner */
801  D_ALLOC_PTR(scan_arg);
802  if (scan_arg == NULL)
803  D_GOTO(out, rc = -DER_NOMEM);
804 
805  rc = ABT_mutex_create(&scan_arg->scan_lock);
806  if (rc != ABT_SUCCESS) {
807  rc = dss_abterr2der(rc);
808  D_GOTO(out_arg, rc);
809  }
810 
811  /* step-2: Create the btree root for global object scan list */
812  memset(&uma, 0, sizeof(uma));
813  uma.uma_id = UMEM_CLASS_VMEM;
814  rc = dbtree_create(DBTREE_CLASS_NV, 0, 4, &uma, NULL,
815  &scan_arg->rebuild_tree_hdl);
816  if (rc != 0) {
817  D_ERROR("failed to create rebuild tree: %d\n", rc);
818  D_GOTO(out_lock, rc);
819  }
820 
821  rc = daos_rank_list_dup(&scan_arg->failed_ranks, rsi->rsi_tgts_failed);
822  if (rc != 0)
823  D_GOTO(out_tree, rc);
824 
825  rpt_get(rpt);
826  scan_arg->rpt = rpt;
827  /* step-3: start scann leader */
828  rc = dss_ult_create(rebuild_scan_leader, scan_arg, -1, NULL);
829  if (rc != 0) {
830  rpt_put(rpt);
831  D_GOTO(out_f_rankfs, rc);
832  }
833 
834  D_GOTO(out, rc);
835 out_f_rankfs:
836  daos_rank_list_free(scan_arg->failed_ranks);
837 out_tree:
838  dbtree_destroy(scan_arg->rebuild_tree_hdl);
839 out_lock:
840  ABT_mutex_free(&scan_arg->scan_lock);
841 out_arg:
842  D_FREE_PTR(scan_arg);
843 out:
844  if (rpt)
845  rpt_put(rpt);
846  ro = crt_reply_get(rpc);
847  ro->rso_status = rc;
848  if (rc) {
849  d_rank_list_t *fail_list;
850 
851  /* If it failed, tell the master the target can not
852  * start the rebuild, so master will put the target
853  * into DOWN state.
854  */
855  fail_list = d_rank_list_alloc(1);
856  if (rpt && rpt->rt_pool)
857  crt_group_rank(rpt->rt_pool->sp_group,
858  &fail_list->rl_ranks[0]);
859  else
860  crt_group_rank(NULL, &fail_list->rl_ranks[0]);
861 
862  ro->rso_ranks_list = fail_list;
863  if (rpt)
864  rpt->rt_abort = 1;
865  }
866 
867  dss_rpc_reply(rpc, DAOS_REBUILD_DROP_SCAN);
868 }

384

391  status->rebuilding = true;
392  }

396  "obj_count="DF_U64", rec_count="DF_U64"\n",
397  DP_UUID(rpt->rt_pool_uuid), status->scanning,

456 static void
457 rebuild_status_check(struct ds_pool *pool, uint32_t map_ver,
458  struct rebuild_global_pool_tracker *rgt)
459 {
460  double begin = ABT_get_wtime();
461  double last_print = 0;
462  double last_query = 0;
463  unsigned int total;
464  int rc;
465 
466  /* FIXME add group later */
467  rc = crt_group_size(NULL, &total);
468  if (rc)
469  return;
470 
471  while (1) {
472  struct daos_rebuild_status *rs = &rgt->rgt_status;
473  char sbuf[RBLD_SBUF_LEN];
474  struct pool_target *targets;
475  unsigned int failed_tgts_cnt;
476  double now;
477  char *str;
478 
479  now = ABT_get_wtime();
480  if (now - last_query < RBLD_BCAST_INTV) {
481  /* Yield to other ULTs */
482  ABT_thread_yield();
483  continue;
484  }
485 
486  last_query = now;
487 
488  rc = pool_map_find_failed_tgts(pool->sp_map, &targets,
489  &failed_tgts_cnt);
490  if (rc != 0) {
491  D_ERROR("failed to create failed tgt list rc %d\n",
492  rc);
493  return;
494  }
495 
496  if (targets != NULL) {
497  int i;
498 
499  for (i = 0; i < failed_tgts_cnt; i++) {
500  D_DEBUG(DB_REBUILD, "target %d failed\n",
501  targets[i].ta_comp.co_rank);
502  setbit(rgt->rgt_scan_bits,
503  targets[i].ta_comp.co_rank);
504  setbit(rgt->rgt_pull_bits,
505  targets[i].ta_comp.co_rank);
506  }
507  D_FREE(targets);
508  }
509 
510  if (!rgt->rgt_done && rgt->rgt_scan_done) {
511  struct rebuild_iv iv;
512 
513  memset(&iv, 0, sizeof(iv));
514  uuid_copy(iv.riv_pool_uuid, rgt->rgt_pool_uuid);
515  iv.riv_master_rank = pool->sp_iv_ns->iv_master_rank;
516  iv.riv_global_scan_done = 1;
517  iv.riv_ver = rgt->rgt_rebuild_ver;
518  iv.riv_leader_term = rgt->rgt_leader_term;
519 
520  /* Notify others the global scan is done, then
521  * each target can reliablly report its pull status
522  */
523  rc = rebuild_iv_update(pool->sp_iv_ns,
524  &iv, CRT_IV_SHORTCUT_NONE,
525  CRT_IV_SYNC_LAZY);
526  if (rc)
527  D_WARN("rebuild master iv update failed: %d\n",
528  rc);
529  }
530 
531  /* query the current rebuild status */
532  if (rgt->rgt_done)
533  rs->rs_done = 1;
534 
535  if (rs->rs_done)
536  str = rs->rs_errno ? "failed" : "completed";
537  else if (rs->rs_obj_nr == 0 && rs->rs_rec_nr == 0)
538  str = "scanning";
539  else
540  str = "pulling";
541 
542  snprintf(sbuf, RBLD_SBUF_LEN,
543  "Rebuild [%s] (pool "DF_UUID" ver=%u, obj="DF_U64
544  ", rec= "DF_U64", done %d status %d duration=%d"
545  " secs)\n", str, DP_UUID(pool->sp_uuid), map_ver,
546  rs->rs_obj_nr, rs->rs_rec_nr, rs->rs_done,
547  rs->rs_errno, (int)(now - begin));
548 
549  D_DEBUG(DB_REBUILD, "%s", sbuf);
550  if (rs->rs_done || rebuild_gst.rg_abort) {
551  D_PRINT("%s", sbuf);
552  break;
553  }
554 
555  /* print something at least for each 10 secons */
556  if (now - last_print > 10) {
557  last_print = now;
558  D_PRINT("%s", sbuf);
559  }
560 
561  ABT_thread_yield();
562  }

724  d_rank_list_t *tgts_failed, d_rank_list_t *svc_list,
725  uint32_t map_ver, daos_iov_t *map_buf)

808  }
809 
810  rc = ds_rebuild_schedule(pool->sp_uuid,
811  pool_map_get_version(pool->sp_map),
812  &fail_rank_list, svc_list);
813  if (rc != 0) {
814  D_ERROR("rebuild fails rc %d\n", rc);
815  break;
816  }
817  }
818  }
819

825  }

1028 #define REBUILD_MAX_INFLIGHT 10
1029 static void
1030 rebuild_ults(void *arg)
1031 {
1032  struct rebuild_task *task;
1033  struct rebuild_task *task_tmp;
1034  int rc;
1035 
1036  while (!d_list_empty(&rebuild_gst.rg_queue_list) ||
1037  !d_list_empty(&rebuild_gst.rg_running_list)) {
1038  if (rebuild_gst.rg_abort) {
1039  D_DEBUG(DB_REBUILD, "abort rebuild\n");
1040  break;
1041  }
1042 
1043  if (d_list_empty(&rebuild_gst.rg_queue_list) ||
1044  rebuild_gst.rg_inflight >= REBUILD_MAX_INFLIGHT) {
1045  ABT_thread_yield();
1046  continue;
1047  }
1048 
1049  d_list_for_each_entry_safe(task, task_tmp,
1050  &rebuild_gst.rg_queue_list, dst_list) {
1051  if (pool_is_rebuilding(task->dst_pool_uuid))
1052  continue;
1053 
1054  rc = dss_ult_create(rebuild_one_ult, task, -1, NULL);
1055  if (rc == 0) {
1056  rebuild_gst.rg_inflight++;
1057  d_list_move(&task->dst_list,
1058  &rebuild_gst.rg_running_list);
1059  } else {
1060  D_ERROR(DF_UUID" create ult failed: %d\n",
1061  DP_UUID(task->dst_pool_uuid), rc);
1062  }
1063  }
1064  ABT_thread_yield();
1065  }
1066 
1067  /* If there are still rebuild task in queue and running list, then
1068  * it is forced abort, let's delete the queue_list task, but leave
1069  * the running task there, either the new leader will tell these
1070  * running rebuild to update their leader or just abort the rebuild
1071  * task.
1072  */
1073  d_list_for_each_entry_safe(task, task_tmp, &rebuild_gst.rg_queue_list,
1074  dst_list) {
1075  d_list_del(&task->dst_list);
1076  daos_rank_list_free(task->dst_tgts_failed);
1077  daos_rank_list_free(task->dst_svc_list);
1078  D_FREE_PTR(task);
1079  }
1080 
1081  ABT_mutex_lock(rebuild_gst.rg_lock);
1082  ABT_cond_signal(rebuild_gst.rg_stop_cond);
1083  rebuild_gst.rg_rebuild_running = 0;
1084  ABT_mutex_unlock(rebuild_gst.rg_lock);
1085 }

1087 void
1088 ds_rebuild_leader_stop()
1089 {
1090  ABT_mutex_lock(rebuild_gst.rg_lock);
1091  if (!rebuild_gst.rg_rebuild_running) {
1092  ABT_mutex_unlock(rebuild_gst.rg_lock);
1093  return;
1094  }
1095 
1096  /* This will eliminate all of the queued rebuild task, then abort all
1097  * running rebuild. Note: this only abort the rebuild tracking ULT
1098  * (rebuild_one_ult), and the real rebuild process on each target
1099  * triggered by scan/object request are still running. Once the new
1100  * leader is elected, it will send those rebuild trigger req with new
1101  * term, then each target will only need update its leader information
1102  * and report the rebuild status to the new leader.
1103  * If the new leader never comes, then those rebuild process can still
1104  * finish, but those tracking ULT (rebuild_tgt_status_check) will keep
1105  * sending the status report to the stale leader, until it is aborted.
1106  */
1107  rebuild_gst.rg_abort = 1;
1108  if (rebuild_gst.rg_rebuild_running)
1109  ABT_cond_wait(rebuild_gst.rg_stop_cond,
1110  rebuild_gst.rg_lock);
1111  ABT_mutex_unlock(rebuild_gst.rg_lock);
1112  if (rebuild_gst.rg_stop_cond)
1113  ABT_cond_free(&rebuild_gst.rg_stop_cond);
1114 }

32 /*
33  * daos_iov_t encoding/decoding utilities
34  *
35  * These functions convert between a daos_iov_t object and a byte stream in a
36  * buffer. The format of such a byte stream is:
37  *
38  * size_head (rdb_iov_size_t)
39  * data
40  * size_tail (rdb_iov_size_t)
41  *
42  * size_head and size_tail are identical, both indicates the size of data,
43  * which equals iov_len of the corresponding daos_iov_t object. The two sizes
44  * allows decoding from the tail as well as from the head.
45  */
46 
47 typedef uint32_t rdb_iov_size_t;
48 
49 /* Maximal buf_len and len of an iov */
50 const daos_size_t rdb_iov_max = (rdb_iov_size_t)-1LL;
51 
52 /* If buf is NULL, then just calculate and return the length required. */
53 size_t
54 rdb_encode_iov(const daos_iov_t *iov, void *buf)
55 {
56  size_t len = sizeof(rdb_iov_size_t) * 2 + iov->iov_len;
57 
58  D_ASSERTF(iov->iov_len <= rdb_iov_max, DF_U64"\n", iov->iov_len);
59  D_ASSERTF(iov->iov_buf_len <= rdb_iov_max, DF_U64"\n",
60  iov->iov_buf_len);
61  if (buf != NULL) {
62  void *p = buf;
63 
64  /* iov_len (head) */
65  *(rdb_iov_size_t *)p = iov->iov_len;
66  p += sizeof(rdb_iov_size_t);
67  /* iov_buf */
68  memcpy(p, iov->iov_buf, iov->iov_len);
69  p += iov->iov_len;
70  /* iov_len (tail) */
71  *(rdb_iov_size_t *)p = iov->iov_len;
72  p += sizeof(rdb_iov_size_t);
73  D_ASSERTF(p - buf == len, "%td == %zu\n", p - buf, len);
74  }
75  return len;
76 }

78 /* Returns the number of bytes processed or -DER_IO if the content is bad. */
79 ssize_t
80 rdb_decode_iov(const void *buf, size_t len, daos_iov_t *iov)
81 {
82  daos_iov_t v = {};
83  const void *p = buf;
84 
85  /* iov_len (head) */
86  if (p + sizeof(rdb_iov_size_t) > buf + len) {
87  D_ERROR("truncated iov_len (head): %zu < %zu\n", len,
88  sizeof(rdb_iov_size_t));
89  return -DER_IO;
90  }
91  v.iov_len = *(const rdb_iov_size_t *)p;
92  if (v.iov_len > rdb_iov_max) {
93  D_ERROR("invalid iov_len (head): "DF_U64" > "DF_U64"\n",
94  v.iov_len, rdb_iov_max);
95  return -DER_IO;
96  }
97  v.iov_buf_len = v.iov_len;
98  p += sizeof(rdb_iov_size_t);
99  /* iov_buf */
100  if (v.iov_len != 0) {
101  if (p + v.iov_len > buf + len) {
102  D_ERROR("truncated iov_buf: %zu < %zu\n", buf + len - p,
103  v.iov_len);
104  return -DER_IO;
105  }
106  v.iov_buf = (void *)p;
107  p += v.iov_len;
108  }
109  /* iov_len (tail) */
110  if (p + sizeof(rdb_iov_size_t) > buf + len) {
111  D_ERROR("truncated iov_len (tail): %zu < %zu\n", buf + len - p,
112  sizeof(rdb_iov_size_t));
113  return -DER_IO;
114  }
115  if (*(const rdb_iov_size_t *)p != v.iov_len) {
116  D_ERROR("inconsistent iov_lens: "DF_U64" != %u\n",
117  v.iov_len, *(const rdb_iov_size_t *)p);
118  return -DER_IO;
119  }
120  p += sizeof(rdb_iov_size_t);
121  *iov = v;
122  return p - buf;
123 }

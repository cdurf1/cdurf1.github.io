350  int i;
351  int rc;
352 
353  crt_group_rank(NULL /* grp */, &rank);
354  crt_group_size(NULL /* grp */, &group_size);
355 
356  /* Build the rank list. */
357  ranks.rl_nr = in->tii_nreplicas;
358  if (ranks.rl_nr > group_size)
359  ranks.rl_nr = group_size;
360  D_ALLOC(ranks.rl_ranks, sizeof(*ranks.rl_ranks) * ranks.rl_nr);
361  if (ranks.rl_ranks == NULL) {
362  D_ERROR("failed to allocate ranks array\n");
363  D_GOTO(out, rc = -DER_NOMEM);
364  }
365  for (i = 0; i < ranks.rl_nr; i++)
366  ranks.rl_ranks[i] = i;
367 
368  D_WARN("initializing rank %u: nreplicas=%u\n", rank, in->tii_nreplicas);
369  rdb_file_path = rdbt_path(in->tii_uuid);
370  MUST(rdb_create(rdb_file_path, in->tii_uuid, 1 << 25, &ranks));
371  MUST(rdb_start(rdb_file_path, &rdbt_rdb_cbs, NULL /* arg */, &rdb_db));
372 out:
373  crt_reply_send(rpc);
374 }

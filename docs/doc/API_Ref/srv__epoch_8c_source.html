188 /*
189  * Decrement epoch \a dec, if not NULL, increment epoch \a inc, if not NULL,
190  * and then find the lowest epoch, in the \a type KVS. If after the update(s)
191  * the KVS will become empty, then \a emptyp returns true; otherwise, \a lowestp
192  * returns the lowest epoch. This is a little bit tricky because rdb does not
193  * support querying a TX's own uncommitted updates.
194  */
195 static int
196 ec_update_and_find_lowest(struct rdb_tx *tx, struct cont *cont,
197  enum ec_type type, const daos_epoch_t *dec,
198  const daos_epoch_t *inc, bool *emptyp,
199  daos_epoch_t *lowestp)
200 {
201  rdb_path_t *kvs = ec_type2kvs(cont, type);
202  daos_epoch_t lowest;
203  bool empty = false;
204  int rc;
205 
206  /* Decrement and set lowest and empty. */
207  if (dec != NULL) {
208  struct ec_decrement_iter_cb_arg arg;
209 
210  rc = ec_decrement(tx, cont, type, *dec);
211  if (rc != 0)
212  return rc;
213 
214  arg.eda_decremented = *dec;
215  arg.eda_found = false;
216  arg.eda_lowest = 0;
217  rc = rdb_tx_iterate(tx, kvs, false /* !backward */,
218  ec_decrement_iter_cb, &arg);
219  if (rc != 0) {
220  D_ERROR(DF_CONT": failed to iterate %s KVS: %d\n",
221  DP_CONT(cont->c_svc->cs_pool_uuid,
222  cont->c_uuid),
223  ec_type2name(type), rc);
224  return rc;
225  }
226  if (arg.eda_found)
227  lowest = arg.eda_lowest;
228  else
229  empty = true;
230  } else {
231  rc = ec_find_lowest(tx, cont, type, &lowest);
232  if (rc == -DER_NONEXIST)
233  empty = true;
234  else if (rc != 0)
235  return rc;
236  }
237 
238  /* Increment and set lowest and empty. */
239  if (inc != NULL) {
240  rc = ec_increment(tx, cont, type, *inc);
241  if (rc != 0)
242  return rc;
243 
244  if (empty || lowest > *inc)
245  lowest = *inc;
246  empty = false;
247  }
248 
249  if (emptyp != NULL)
250  *emptyp = empty;
251  if (lowestp != NULL && !empty)
252  *lowestp = lowest;
253  return 0;
254 }

460 /*
461  * Calculate GHCE afresh using attr->ea_ghpce and attr->ea_glhe, and if the
462  * result is higher than attr->ea_ghce, update attr->ea_ghce and CONT_GHCE.
463  *
464  * This function must be called by operations change GHPCE or GLHE. It may be
465  * called unnecessarily by other operations without compromising safety.
466  */
467 static int
468 update_ghce(struct rdb_tx *tx, struct cont *cont, struct epoch_attr *attr)
469 {
470  daos_epoch_t ghce;
471  daos_iov_t value;
472  int rc;
473 
474  /*
475  * GHCE shall be the highest epoch e satisfying both of these:
476  * - e <= GHPCE (committed by some handle)
477  * - e < GLHE (not held by any handle)
478  */
479  ghce = min(attr->ea_ghpce, attr->ea_glhe - 1);
480 
481  if (ghce < attr->ea_ghce) {
482  D_ERROR(DF_CONT": GHCE would decrease: "DF_EPOCH_ATTR"\n",
483  DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
484  DP_EPOCH_ATTR(attr));
485  return -DER_IO;
486  } else if (ghce == attr->ea_ghce) {
487  return 0;
488  }
489 
490  daos_iov_set(&value, &ghce, sizeof(ghce));
491  rc = rdb_tx_update(tx, &cont->c_attrs, &ds_cont_attr_ghce, &value);
492  if (rc != 0)
493  D_ERROR(DF_CONT": failed to update ghce: %d\n",
494  DP_CONT(cont->c_svc->cs_pool_uuid,
495  cont->c_uuid), rc);
496 
497  attr->ea_ghce = ghce;
498  return rc;
499 }

888  daos_iov_t key;
889  daos_iov_t value;

892  int rc;
893 
894  D_DEBUG(DF_DSMS, DF_CONT": processing rpc %p: epoch="DF_U64"\n",
895  DP_CONT(pool_hdl->sph_pool->sp_uuid, in->cei_op.ci_uuid), rpc,
896  in->cei_epoch);
897 
898  slip_flag = false;
899  islip = getenv("DAOS_IMPLICIT_PURGE");
900  if (islip != NULL)
901  slip_flag = true;
902 
903  /* Verify the container handle capabilities. */
904  if (!(hdl->ch_capas & DAOS_COO_RW))
905  D_GOTO(out, rc = -DER_NO_PERM);
906 
907  if (in->cei_epoch >= DAOS_EPOCH_MAX)
908  D_GOTO(out, rc = -DER_OVERFLOW);
909 
910  rc = read_epoch_attr(tx, cont, &attr);
911  if (rc != 0)
912  D_GOTO(out, rc);
913 
914  if (slip_flag)
915  glre = attr.ea_glre;
916 
917  if (check_epoch_invariant(cont, &attr, hdl) != 0)
918  D_GOTO(out, rc = -DER_IO);
919 
920  if (in->cei_epoch <= hdl->ch_hce)
921  /* Committing an already committed epoch is okay and a no-op. */
922  D_GOTO(out_state, rc = 0);
923  else if (in->cei_epoch < hdl->ch_lhe)
924  /* Committing an unheld epoch is not allowed. */
925  D_GOTO(out, rc = -DER_EP_RO);
926 
927  hdl->ch_hce = in->cei_epoch;
928  hdl->ch_lhe = hdl->ch_hce + 1;
929  if (!(hdl->ch_capas & DAOS_COO_NOSLIP))
930  hdl->ch_lre = hdl->ch_hce;
931 
932  D_DEBUG(DF_DSMS, "hce="DF_U64" hce'="DF_U64"\n", hce, hdl->ch_hce);
933  D_DEBUG(DF_DSMS, "lhe="DF_U64" lhe'="DF_U64"\n", lhe, hdl->ch_lhe);
934  D_DEBUG(DF_DSMS, "lre="DF_U64" lre'="DF_U64"\n", lre, hdl->ch_lre);
935 
936  daos_iov_set(&key, in->cei_op.ci_hdl, sizeof(uuid_t));
937  daos_iov_set(&value, hdl, sizeof(*hdl));
938  rc = rdb_tx_update(tx, &cont->c_svc->cs_hdls, &key, &value);
939  if (rc != 0)
940  D_GOTO(out_hdl, rc);
941 
942  rc = ec_update_and_find_lowest(tx, cont, EC_LHE, &lhe /* dec */,
943  &hdl->ch_lhe /* inc */,
944  NULL /* emptyp */, &attr.ea_glhe);
945  if (rc != 0)
946  D_GOTO(out_hdl, rc);
947 
948  if (!(hdl->ch_capas & DAOS_COO_NOSLIP)) {
949  rc = ec_update_and_find_lowest(tx, cont, EC_LRE, &lre /* dec */,
950  &hdl->ch_lre /* inc */,
951  NULL /* emptyp */,
952  &attr.ea_glre);
953  if (rc != 0)
954  D_GOTO(out_hdl, rc);
955  }
956 
957  if (hdl->ch_hce > attr.ea_ghpce) {
958  attr.ea_ghpce = hdl->ch_hce;
959  daos_iov_set(&value, &attr.ea_ghpce, sizeof(attr.ea_ghpce));
960  rc = rdb_tx_update(tx, &cont->c_attrs, &ds_cont_attr_ghpce,
961  &value);
962  if (rc != 0) {
963  D_ERROR(DF_CONT": failed to update ghpce: %d\n",
964  DP_CONT(cont->c_svc->cs_pool_uuid,
965  cont->c_uuid), rc);
966  D_GOTO(out_hdl, rc);
967  }
968  }
969 
970  rc = update_ghce(tx, cont, &attr);
971  if (rc != 0)
972  D_GOTO(out_hdl, rc);
973 
974  if (check_epoch_invariant(cont, &attr, hdl) != 0)
975  D_GOTO(out_hdl, rc = -DER_IO);
976 
977  if (slip_flag) {
978  rc = trigger_aggregation(glre, attr.ea_glre, attr.ea_ghce,
979  cont, rpc->cr_ctx);
980  if (rc != 0) {
981  D_ERROR("Trigger aggregation from commit failed %d\n",
982  rc);
983  D_GOTO(out_hdl, rc);
984  }
985  }
986 
987 out_state:
988  set_epoch_state(&attr, hdl, &out->ceo_epoch_state);
989 out_hdl:
990  if (rc != 0) {
991  hdl->ch_lre = lre;
992  hdl->ch_lhe = lhe;
993  hdl->ch_hce = hce;
994  }
995 out:
996  D_DEBUG(DF_DSMS, DF_CONT": replying rpc %p: %d\n",
997  DP_CONT(pool_hdl->sph_pool->sp_uuid, in->cei_op.ci_uuid), rpc,
998  rc);

360 static void
361 rdb_raft_queue_event(struct rdb *db, enum rdb_raft_event_type type,
362  uint64_t term)
363 {
364  D_ASSERTF(db->d_nevents >= 0 &&
365  db->d_nevents <= ARRAY_SIZE(db->d_events),
366  "%d\n", db->d_nevents);
367 
368  if (db->d_nevents > 0) {
369  struct rdb_raft_event *tail = &db->d_events[db->d_nevents - 1];
370 
371  switch (type) {
372  case RDB_RAFT_STEP_UP:
373  D_ASSERT(tail->dre_type == RDB_RAFT_STEP_DOWN);
374  D_ASSERTF(tail->dre_term < term, DF_U64" < "DF_U64"\n",
375  tail->dre_term, term);
376  break;
377  case RDB_RAFT_STEP_DOWN:
378  D_ASSERT(tail->dre_type == RDB_RAFT_STEP_UP);
379  D_ASSERT(tail->dre_term == term);
380  /*
381  * Since both of the matching events are still pending,
382  * cancel the UP and don't queue the DOWN, to avoid
383  * useless callbacks. This leaves us four possible
384  * states of the queue:
385  *
386  * - empty
387  * - UP(t)
388  * - DOWN(t)
389  * - DOWN(t), UP(t')
390  *
391  * where t' > t. The maximal queue size is therefore 2.
392  */
393  db->d_nevents--;
394  return;
395  default:
396  D_ASSERTF(0, "unknown event type: %d\n", type);
397  }
398  }
399 
400  /* Queue this new event. */
401  D_ASSERTF(db->d_nevents < ARRAY_SIZE(db->d_events), "%d\n",
402  db->d_nevents);
403  db->d_events[db->d_nevents].dre_term = term;
404  db->d_events[db->d_nevents].dre_type = type;
405  db->d_nevents++;
406  ABT_cond_broadcast(db->d_events_cv);
407 }

422 static void
423 rdb_raft_process_event(struct rdb *db, struct rdb_raft_event *event)
424 {
425  int rc;
426 
427  switch (event->dre_type) {
428  case RDB_RAFT_STEP_UP:
429  if (db->d_cbs == NULL || db->d_cbs->dc_step_up == NULL)
430  break;
431  rc = db->d_cbs->dc_step_up(db, event->dre_term, db->d_arg);
432  if (rc == 0)
433  break;
434  /* An error occurred. Step down if we are still that leader. */
435  if (raft_is_leader(db->d_raft) &&
436  raft_get_current_term(db->d_raft) ==
437  event->dre_term) {
438  D_DEBUG(DB_MD, DF_DB": stepping down from term "DF_U64
439  "\n", DP_DB(db), event->dre_term);
440  /* No need to generate a DOWN event. */
441  raft_become_follower(db->d_raft);
442  }
443  /*
444  * If there are pending events, then the next one must be the
445  * matching DOWN. (See the assertions in
446  * rdb_raft_queue_event().) Discard it to reduce just a little
447  * burden on the service code.
448  */
449  if (db->d_nevents > 0) {
450  struct rdb_raft_event next;
451 
452  rdb_raft_dequeue_event(db, &next);
453  D_ASSERTF(next.dre_type == RDB_RAFT_STEP_DOWN &&
454  next.dre_term == event->dre_term,
455  "%d "DF_U64" "DF_U64"\n", next.dre_type,
456  next.dre_term, event->dre_term);
457  }
458  break;
459  case RDB_RAFT_STEP_DOWN:
460  if (db->d_cbs == NULL || db->d_cbs->dc_step_down == NULL)
461  break;
462  db->d_cbs->dc_step_down(db, event->dre_term, db->d_arg);
463  break;
464  default:
465  D_ASSERTF(0, "unknown event type: %d\n", event->dre_type);
466  }
467 }

551 /*
552  * Check the current state against "state", which shall be a previously-saved
553  * state, and handle any changes and errors.
554  */
555 static int
556 rdb_raft_check_state(struct rdb *db, const struct rdb_raft_state *state,
557  int raft_rc)
558 {
559  bool leader = raft_is_leader(db->d_raft);
560  uint64_t term = raft_get_current_term(db->d_raft);
561  uint64_t committed;
562  int step_up_rc = 0;
563  int rc;
564 
565  /* Check the leader state. */
566  D_ASSERTF(term >= state->drs_term, DF_U64" >= "DF_U64"\n", term,
567  state->drs_term);
568  if (!state->drs_leader && leader) {
569  /* In this case, raft currently always returns zero. */
570  D_ASSERTF(raft_rc == 0, "%d\n", raft_rc);
571  step_up_rc = rdb_raft_step_up(db, term);
572  } else if (state->drs_leader && !leader) {
573  rdb_raft_step_down(db, state->drs_term);
574  }
575 
576  /*
577  * Check the commit state. We query the commit index here instead of at
578  * the beginning of this function, as the rdb_raft_step_up() call above
579  * may have increased it.
580  */
581  committed = raft_get_commit_idx(db->d_raft);
582  D_ASSERTF(committed >= state->drs_committed, DF_U64" >= "DF_U64"\n",
583  committed, state->drs_committed);
584  if (committed != state->drs_committed) {
585  ABT_cond_broadcast(db->d_committed_cv);
586  D_DEBUG(DB_ANY, DF_DB": committed to entry "DF_U64"\n",
587  DP_DB(db), committed);
588  }
589 
590  /* Check raft_rc and step_up_rc. */
591  if (raft_rc == 0) {
592  rc = step_up_rc;
593  } else {
594  D_ASSERTF(step_up_rc == 0, "%d\n", step_up_rc);
595  rc = rdb_raft_rc(raft_rc);
596  }
597  switch (rc) {
598  case -DER_NOMEM:
599  if (leader) {
600  raft_become_follower(db->d_raft);
601  leader = false;
602  /* If stepping up fails, don't step down. */
603  if (step_up_rc != 0)
604  break;
605  rdb_raft_step_down(db, state->drs_term);
606  }
607  break;
608  case -DER_SHUTDOWN:
609  case -DER_NOSPACE:
610  case -DER_IO:
611  db->d_cbs->dc_stop(db, rc, db->d_arg);
612  break;
613  }
614 
615  /* Wake up waiters who care about term and leadership changes. */
616  if (state->drs_term != term || state->drs_leader != leader)
617  ABT_cond_broadcast(db->d_applied_cv);
618 
619  return rc;
620 }

829 /* Daemon ULT for raft_periodic() */
830 static void
831 rdb_timerd(void *arg)
832 {
833  struct rdb *db = arg;
834  const double d_min = 0.5; /* min duration between beats (s) */
835  const double d_max = 1; /* max duration between beats (s) */
836  double d = 0; /* duration till next beat (s) */
837  double t; /* timestamp of beat (s) */
838  double t_prev; /* timestamp of previous beat (s) */
839  int rc;
840 
841  D_DEBUG(DB_ANY, DF_DB": timerd starting\n", DP_DB(db));
842  t = ABT_get_wtime();
843  t_prev = t;
844  do {
845  struct rdb_raft_state state;
846  double d_prev = t - t_prev;
847 
848  if (d_prev - d > 1 /* s */)
849  D_WARN(DF_DB": not scheduled for %f second\n",
850  DP_DB(db), d_prev - d);
851 
852  rdb_raft_save_state(db, &state);
853  rc = raft_periodic(db->d_raft, d_prev * 1000 /* ms */);
854  rc = rdb_raft_check_state(db, &state, rc);
855  if (rc != 0)
856  D_ERROR(DF_DB": raft_periodic() failed: %d\n",
857  DP_DB(db), rc);
858 
859  t_prev = t;
860  /* Wait for d in [d_min, d_max] before the next beat. */
861  d = d_min + (d_max - d_min) * rdb_raft_rand();
862  while ((t = ABT_get_wtime()) < t_prev + d && !db->d_stop)
863  ABT_thread_yield();
864  } while (!db->d_stop);
865  D_DEBUG(DB_ANY, DF_DB": timerd stopping\n", DP_DB(db));
866 }

1200  D_ASSERTF(rc == 0, "%d\n", rc);

1204  for (i = 0; i < nreplicas; i++) {
1205  raft_node_t *node;

1221  ABT_cond_free(&db->d_events_cv);

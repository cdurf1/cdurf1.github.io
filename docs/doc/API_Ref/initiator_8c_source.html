674 rebuild_cont_iter_cb(daos_handle_t ih, daos_iov_t *key_iov,
675  daos_iov_t *val_iov, void *data)
676 {

678  struct rebuild_iter_arg *arg = data;

680  struct rebuild_pool_tls *tls;

682  int rc;
683 
684  uuid_copy(arg->cont_uuid, *(uuid_t *)key_iov->iov_buf);
685  D_DEBUG(DB_REBUILD, "iter cont "DF_UUID"/%"PRIx64" %"PRIx64" start\n",
686  DP_UUID(arg->cont_uuid), ih.cookie, root->root_hdl.cookie);
687 
688  tls = rebuild_pool_tls_lookup(rpt->rt_pool_uuid,
689  rpt->rt_rebuild_ver);
690  D_ASSERT(tls != NULL);

692  if (daos_handle_is_inval(tls->rebuild_pool_hdl)) {

694  struct pool_map *map = rebuild_pool_map_get(rpt->rt_pool);

700  return rc;
701 
702  tls->rebuild_pool_hdl = ph;
703  }

706  0, tls->rebuild_pool_hdl, &coh);
707  if (rc)
708  return rc;
709  arg->cont_hdl = coh;
710 
711  while (!dbtree_is_empty(root->root_hdl)) {
712  rc = dbtree_iterate(root->root_hdl, false,
713  rebuild_obj_iter_cb, arg);
714  if (rc) {
715  if (tls->rebuild_pool_status == 0)
716  tls->rebuild_pool_status = rc;
717  D_ERROR("iterate cont "DF_UUID" failed: rc %d\n",
718  DP_UUID(arg->cont_uuid), rc);
719  break;
720  }
721  }
722 
723  rc = dc_cont_local_close(tls->rebuild_pool_hdl, coh);
724  if (rc)
725  return rc;
726 
727  D_DEBUG(DB_REBUILD, "iter cont "DF_UUID"/%"PRIx64" finish.\n",
728  DP_UUID(arg->cont_uuid), ih.cookie);
729 
730  /* Some one might insert new record to the tree let's reprobe */
731  rc = dbtree_iter_probe(ih, BTR_PROBE_EQ, key_iov, NULL);
732  if (rc) {
733  D_ASSERT(rc != -DER_NONEXIST);
734  return rc;
735  }
736 
737  rc = dbtree_iter_delete(ih, NULL);
738  if (rc)
739  return rc;
740 
741  /* re-probe the dbtree after delete */
742  rc = dbtree_iter_probe(ih, BTR_PROBE_FIRST, NULL, NULL);
743  if (rc == -DER_NONEXIST || rpt->rt_abort)
744  return 1;
745 
746  return rc;
747 }

801 /* Got the object list from scanner and rebuild the objects */
802 void
803 rebuild_obj_handler(crt_rpc_t *rpc)
804 {
805  struct rebuild_objs_in *rebuild_in;
806  struct rebuild_tgt_pool_tracker *rpt = NULL;
807  struct rebuild_pool_tls *tls;
808  struct rebuild_out *rebuild_out;
809  daos_unit_oid_t *oids;
810  unsigned int oids_count;
811  uuid_t *co_uuids;
812  unsigned int co_count;
813  uint32_t *shards;
814  unsigned int shards_count;
815  daos_handle_t btr_hdl;
816  unsigned int i;
817  int rc;
818 
819  rebuild_in = crt_req_get(rpc);
820  oids = rebuild_in->roi_oids.ca_arrays;
821  oids_count = rebuild_in->roi_oids.ca_count;
822  co_uuids = rebuild_in->roi_uuids.ca_arrays;
823  co_count = rebuild_in->roi_uuids.ca_count;
824  shards = rebuild_in->roi_shards.ca_arrays;
825  shards_count = rebuild_in->roi_shards.ca_count;
826 
827  if (co_count == 0 || oids_count == 0 || shards_count == 0 ||
828  oids_count != co_count || oids_count != shards_count) {
829  D_ERROR("oids_count %u co_count %u shards_count %u\n",
830  oids_count, co_count, shards_count);
831  D_GOTO(out, rc = -DER_INVAL);
832  }
833 
834  /* If rpt is NULL, it means the target is not prepared for
835  * rebuilding yet, i.e. it did not receive scan req to
836  * prepare rebuild yet (see rebuild_tgt_prepare()).
837  */
838  rpt = rpt_lookup(rebuild_in->roi_pool_uuid,
839  rebuild_in->roi_rebuild_ver);
840  if (rpt == NULL || rpt->rt_pool == NULL)
841  D_GOTO(out, rc = -DER_AGAIN);
842 
843  /* Initialize the local rebuild tree */
844  rc = rebuild_obj_hdl_get(rpt, &btr_hdl);
845  if (rc)
846  D_GOTO(out_put, rc);
847 
848  tls = rebuild_pool_tls_lookup(rpt->rt_pool_uuid,
849  rpt->rt_rebuild_ver);
850  D_ASSERT(tls != NULL);
851 
852  /* Insert these oids/conts into the local rebuild tree */
853  for (i = 0; i < oids_count; i++) {
854  rc = rebuild_cont_obj_insert(btr_hdl, co_uuids[i],
855  oids[i], shards[i]);
856  if (rc == 1) {
857  tls->rebuild_pool_obj_count++;
858  D_DEBUG(DB_REBUILD, "insert local "DF_UOID" "DF_UUID
859  " %u hdl %"PRIx64"\n", DP_UOID(oids[i]),
860  DP_UUID(co_uuids[i]), shards[i],
861  btr_hdl.cookie);
862  rc = 0;
863  } else if (rc == 0) {
864  D_DEBUG(DB_REBUILD, DF_UOID" "DF_UUID" %u exist.\n",
865  DP_UOID(oids[i]), DP_UUID(co_uuids[i]),
866  shards[i]);
867  } else if (rc < 0) {
868  break;
869  }
870  }
871  if (rc < 0)
872  D_GOTO(out_put, rc);
873 
874  /* Check and create task to iterate the local rebuild tree */
875  if (!rpt->rt_lead_puller_running) {
876  struct rebuild_iter_arg *arg;
877 
878  D_ALLOC_PTR(arg);
879  if (arg == NULL)
880  D_GOTO(out_put, rc = -DER_NOMEM);
881 
882  uuid_copy(arg->pool_uuid, rebuild_in->roi_pool_uuid);
883  arg->obj_cb = rebuild_obj_iterate_keys;
884  arg->root_hdl = btr_hdl;
885  rpt_get(rpt);
886  arg->rpt = rpt;
887 
888  D_ASSERT(rpt->rt_pullers != NULL);
889  rc = dss_ult_create(rebuild_puller, arg, -1, NULL);
890  if (rc) {
891  rpt_put(rpt);
892  D_FREE_PTR(arg);
893  D_GOTO(out_put, rc);
894  }
895  rpt->rt_lead_puller_running = 1;
896  }
897 out_put:
898  rpt_put(rpt);
899 out:
900  rebuild_out = crt_reply_get(rpc);
901  rebuild_out->ro_status = rc;
902  dss_rpc_reply(rpc, DAOS_REBUILD_DROP_OBJ);
903 }

718  D_FREE(vio->dvi_viod.iod_csums);
719  D_GOTO(out, -DER_NOMEM);
720  }
721 
722  vio->dvi_viod.iod_eprs = (daos_epoch_range_t *)ptmp;
723

747  &dei->del_recs[j].der_rec);

749  sizeof(daos_recx_t));

753

74 rdb_kvs_open_path(struct rdb *db, const rdb_path_t *path, daos_handle_t *hdl)
75 {
76  rdb_path_t p = *path;
77  struct rdb_kvs *kvs = NULL;
78  struct rdb_kvs_open_arg arg;
79  int rc;
80 
81  /* See if we can find a cache hit for a prefix of the path. */
82  while (rdb_path_pop(&p) == 0 && p.iov_len > 0) {
83  rc = rdb_kvs_lookup_internal(db, &p, false /* alloc */, &kvs);
84  if (rc == 0)
85  break;
86  else if (rc != -DER_NONEXIST)
87  return rc;
88  };
89 
90  /* Walk through the keys after "p". */
91  D_DEBUG(DB_ANY, "walking path %zu from kvs %p\n", p.iov_len, kvs);
92  p.iov_buf += p.iov_len;
93  p.iov_buf_len -= p.iov_len;
94  p.iov_len = path->iov_len - p.iov_len;
95  D_ASSERT(p.iov_len > 0);
96  arg.deo_base = kvs == NULL ? db->d_attr : kvs->de_hdl;
97  arg.deo_parent = DAOS_HDL_INVAL;
98  rc = rdb_path_iterate(&p, rdb_kvs_open_path_cb, &arg);
99  if (kvs != NULL)
100  rdb_kvs_put(db, kvs);
101  if (rc != 0) {
102  if (!daos_handle_is_inval(arg.deo_parent))
103  dbtree_close(arg.deo_parent);
104  return rc;
105  }
106 
107  D_DEBUG(DB_ANY, "got kvs handle "DF_U64"\n", arg.deo_parent.cookie);
108  *hdl = arg.deo_parent;
109  return 0;
110 }

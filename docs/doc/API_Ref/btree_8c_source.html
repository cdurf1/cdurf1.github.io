878 btr_split_at(struct btr_context *tcx, int level,
879  TMMID(struct btr_node) mmid_left,

906 static int
907 btr_node_split_and_insert(struct btr_context *tcx, struct btr_trace *trace,
908  struct btr_record *rec)
909 {
910  struct btr_record *rec_src;
911  struct btr_record *rec_dst;
912  struct btr_node *nd_left;
913  struct btr_node *nd_right;
914  TMMID(struct btr_node) mmid_left;
915  TMMID(struct btr_node) mmid_right;
916  char hkey_buf[DAOS_HKEY_MAX];
917  int split_at;
918  int level;
919  int rc;
920  bool leaf;
921  bool right;
922 
923  D_ASSERT(trace >= tcx->tc_trace);
924  level = trace - tcx->tc_trace;
925  mmid_left = trace->tr_node;
926 
927  rc = btr_node_alloc(tcx, &mmid_right);
928  if (rc != 0)
929  return rc;
930 
931  leaf = btr_node_is_leaf(tcx, mmid_left);
932  if (leaf)
933  btr_node_set(tcx, mmid_right, BTR_NODE_LEAF);
934 
935  split_at = btr_split_at(tcx, level, mmid_left, mmid_right);
936 
937  rec_src = btr_node_rec_at(tcx, mmid_left, split_at);
938  rec_dst = btr_node_rec_at(tcx, mmid_right, 0);
939 
940  nd_left = btr_mmid2ptr(tcx, mmid_left);
941  nd_right = btr_mmid2ptr(tcx, mmid_right);
942 
943  nd_right->tn_keyn = nd_left->tn_keyn - split_at;
944  nd_left->tn_keyn = split_at;
945 
946  if (leaf) {
947  D_DEBUG(DB_TRACE, "Splitting leaf node\n");
948 
949  btr_rec_copy(tcx, rec_dst, rec_src, nd_right->tn_keyn);
950  btr_node_insert_rec_only(tcx, trace, rec);
951 
952  /* insert the right node and the first key of the right
953  * node to its parent
954  */
955  if (tcx->tc_feats & BTR_FEAT_DIRECT_KEY)
956  rec->rec_node[0] = mmid_right;
957  else
958  btr_rec_copy_hkey(tcx, rec, rec_dst);
959  goto bubble_up;
960  }
961  /* non-leaf */
962 
963  right = btr_node_is_equal(tcx, trace->tr_node, mmid_right);
964  if (trace->tr_at == 0 && right) {
965  /* the new record is the first one on the right node */
966  D_DEBUG(DB_TRACE, "Bubble up the new key\n");
967  nd_right->tn_child = umem_id_u2t(rec->rec_mmid,
968  struct btr_node);
969  btr_rec_copy(tcx, rec_dst, rec_src, nd_right->tn_keyn);
970  goto bubble_up;
971  }
972 
973  D_DEBUG(DB_TRACE, "Bubble up the 1st key of the right node\n");
974 
975  nd_right->tn_child = umem_id_u2t(rec_src->rec_mmid, struct btr_node);
976  /* btr_split_at should ensure the right node has more than one record,
977  * because the first record of the right node will bubble up.
978  * (@src_rec[0] is this record at this point)
979  */
980  D_ASSERT(nd_right->tn_keyn > 1 || right);
981  nd_right->tn_keyn--;
982  /* insertion point has to be shifted if the new record is going to
983  * be inserted to the right node.
984  */
985  trace->tr_at -= right;
986 
987  /* Copy from @rec_src[1] because @rec_src[0] will bubble up.
988  * NB: call btr_rec_at instead of using array index, see btr_record.
989  */
990  btr_rec_copy(tcx, rec_dst, btr_rec_at(tcx, rec_src, 1),
991  nd_right->tn_keyn);
992 
993  /* backup it because the below btr_node_insert_rec_only may
994  * overwrite it.
995  */
996  btr_hkey_copy(tcx, &hkey_buf[0], &rec_src->rec_hkey[0]);
997 
998  btr_node_insert_rec_only(tcx, trace, rec);
999 
1000  btr_hkey_copy(tcx, &rec->rec_hkey[0], &hkey_buf[0]);
1001 
1002  bubble_up:
1003  D_DEBUG(DB_TRACE, "left keyn %d, right keyn %d\n",
1004  nd_left->tn_keyn, nd_right->tn_keyn);
1005 
1006  rec->rec_mmid = umem_id_t2u(mmid_right);
1007  if (level == 0)
1008  rc = btr_root_grow(tcx, mmid_left, rec);
1009  else
1010  rc = btr_node_insert_rec(tcx, trace - 1, rec);
1011 
1012  return rc;
1013 }

1060 static enum btr_probe_rc
1061 btr_probe(struct btr_context *tcx, int opc, daos_iov_t *key,
1062  daos_hash_out_t *anchor)
1063 {
1064  struct btr_record *rec;
1065  int start;
1066  int end;
1067  int at;
1068  int cmp;
1069  int level;
1070  bool nextl;
1071  char hkey_buf[DAOS_HKEY_MAX];
1072  char *hkey = &hkey_buf[0];
1073  TMMID(struct btr_node) nd_mmid;
1074 
1075  memset(&tcx->tc_traces[0], 0,
1076  sizeof(tcx->tc_traces[0]) * BTR_TRACE_MAX);
1077 
1078  /* depth could be changed by dbtree_delete/dbtree_iter_delete from
1079  * a different btr_context, so we always reinitialize both depth
1080  * and start point of trace for the context.
1081  */
1082  btr_context_set_depth(tcx, tcx->tc_tins.ti_root->tr_depth);
1083 
1084  if (btr_root_empty(tcx)) { /* empty tree */
1085  D_DEBUG(DB_TRACE, "Empty tree\n");
1086  return PROBE_RC_NONE;
1087  }
1088 
1089  if (opc & BTR_PROBE_EQ) {
1090  if (key != NULL) {
1091  btr_hkey_gen(tcx, key, hkey);
1092  } else {
1093  D_ASSERT(anchor != NULL);
1094  D_ASSERT(opc != BTR_PROBE_UPDATE);
1095  D_ASSERT((tcx->tc_feats & BTR_FEAT_DIRECT_KEY) == 0);
1096 
1097  btr_hkey_copy(tcx, hkey, &anchor->body[0]);
1098  }
1099  }
1100 
1101  nd_mmid = tcx->tc_tins.ti_root->tr_node;
1102  start = end = 0;
1103 
1104  for (nextl = true, level = 0 ;;) {
1105  if (nextl) { /* search a new level of the tree */
1106  nextl = false;
1107  start = 0;
1108  end = btr_mmid2ptr(tcx, nd_mmid)->tn_keyn - 1;
1109 
1110  D_DEBUG(DB_TRACE,
1111  "Probe level %d, node "TMMID_PF" keyn %d\n",
1112  level, TMMID_P(nd_mmid), end + 1);
1113  }
1114 
1115  if (opc & BTR_PROBE_EQ) {
1116  /* binary search */
1117  at = (start + end) / 2;
1118  if (tcx->tc_feats & BTR_FEAT_DIRECT_KEY) {
1119  rec = btr_node_direct_rec_at(tcx, nd_mmid, at);
1120  cmp = btr_key_cmp(tcx, rec, key);
1121  } else {
1122  rec = btr_node_rec_at(tcx, nd_mmid, at);
1123  cmp = btr_hkey_cmp(tcx, rec, hkey);
1124  }
1125 
1126  D_DEBUG(DB_TRACE, "compared record at %d, cmp %d\n",
1127  at, cmp);
1128 
1129  } else if (opc == BTR_PROBE_FIRST) {
1130  at = start = end = 0;
1131  cmp = BTR_CMP_GT;
1132 
1133  } else {
1134  D_ASSERT(opc == BTR_PROBE_LAST);
1135  at = start = end;
1136  cmp = BTR_CMP_LT;
1137  }
1138 
1139  if (cmp == BTR_CMP_ERR) {
1140  D_DEBUG(DB_TRACE, "compared record at %d, got "
1141  "BTR_CMP_ERR, return PROBE_RC_ERR.", at);
1142  return PROBE_RC_ERR;
1143  }
1144 
1145  D_ASSERTF((cmp == BTR_CMP_LT || cmp == BTR_CMP_GT ||
1146  cmp == BTR_CMP_EQ), "invalid key_cmp rc %d.\n", cmp);
1147  if (start < end && cmp != BTR_CMP_EQ) {
1148  /* continue the binary search in current level */
1149  if (cmp == BTR_CMP_LT)
1150  start = at + 1;
1151  else
1152  end = at - 1;
1153  continue;
1154  }
1155 
1156  if (btr_node_is_leaf(tcx, nd_mmid))
1157  break;
1158 
1159  /* NB: cmp is BTR_CMP_LT or BTR_CMP_EQ means search the record
1160  * in the right child, otherwise it is in the left child.
1161  */
1162  at += (cmp != BTR_CMP_GT);
1163  btr_trace_set(tcx, level, nd_mmid, at);
1164  btr_trace_debug(tcx, &tcx->tc_trace[level], "probe child\n");
1165 
1166  /* Search the next level. */
1167  nd_mmid = btr_node_child_at(tcx, nd_mmid, at);
1168  nextl = true;
1169  level++;
1170  }
1171  /* leaf node */
1172  D_ASSERT(level == tcx->tc_depth - 1);
1173  D_ASSERT(!TMMID_IS_NULL(nd_mmid));
1174 
1175  if (!(tcx->tc_feats & (BTR_FEAT_DIRECT_KEY | BTR_FEAT_UINT_KEY))) {
1176  if (cmp == BTR_CMP_EQ && key != NULL) {
1177  rec = btr_node_rec_at(tcx, nd_mmid, at);
1178  cmp = btr_key_cmp(tcx, rec, key);
1179  if (cmp == BTR_CMP_ERR) {
1180  D_DEBUG(DB_TRACE, "compared record at %d, got "
1181  "BTR_CMP_ERR, return PROBE_RC_ERR.",
1182  at);
1183  return PROBE_RC_ERR;
1184  }
1185  if (cmp != BTR_CMP_EQ)
1186  D_ERROR("Hash collision is poorly handled\n");
1187  }
1188  }
1189 
1190  if (opc == BTR_PROBE_UPDATE)
1191  at += (cmp == BTR_CMP_LT);
1192 
1193  btr_trace_set(tcx, level, nd_mmid, at);
1194  btr_trace_debug(tcx, &tcx->tc_trace[level], "probe finished\n");
1195 
1196  switch (opc) {
1197  default:
1198  D_ASSERT(0);
1199  case BTR_PROBE_FIRST:
1200  case BTR_PROBE_LAST:
1201  return PROBE_RC_OK;
1202 
1203  case BTR_PROBE_UPDATE:
1204  case BTR_PROBE_EQ:
1205  return cmp == BTR_CMP_EQ ? PROBE_RC_EQ : PROBE_RC_NONE;
1206 
1207  case BTR_PROBE_GE:
1208  if (cmp == BTR_CMP_EQ)
1209  return PROBE_RC_EQ;
1210  if (cmp == BTR_CMP_GT)
1211  return PROBE_RC_OK;
1212 
1213  return btr_probe_next(tcx) ? PROBE_RC_OK : PROBE_RC_NONE;
1214 
1215  case BTR_PROBE_LE:
1216  if (cmp == BTR_CMP_EQ)
1217  return PROBE_RC_EQ;
1218  if (cmp == BTR_CMP_LT)
1219  return PROBE_RC_OK;
1220 
1221  return btr_probe_prev(tcx) ? PROBE_RC_OK : PROBE_RC_NONE;
1222  }
1223 }

1226 btr_probe_next(struct btr_context *tcx)
1227 {
1228  struct btr_trace *trace;
1229  struct btr_node *nd;
1230  TMMID(struct btr_node) nd_mmid;
1231 
1232  if (btr_root_empty(tcx)) /* empty tree */
1233  return false;
1234 
1235  trace = &tcx->tc_trace[tcx->tc_depth - 1];
1236 
1237  btr_trace_debug(tcx, trace, "Probe the next\n");
1238  while (1) {
1239  bool leaf;
1240 
1241  nd_mmid = trace->tr_node;
1242  leaf = btr_node_is_leaf(tcx, nd_mmid);
1243 
1244  nd = btr_mmid2ptr(tcx, nd_mmid);
1245 
1246  /* NB: trace->tr_at might be larger than key number because
1247  * split can happen between two calls.
1248  */
1249  if (btr_node_is_root(tcx, nd_mmid) &&
1250  trace->tr_at >= nd->tn_keyn - leaf) {
1251  D_ASSERT(trace == tcx->tc_trace);
1252  D_DEBUG(DB_TRACE, "End\n");
1253  return false; /* done */
1254  }
1255 
1256  if (trace->tr_at >= nd->tn_keyn - leaf) {
1257  /* finish current level */
1258  trace--;
1259  continue;
1260  }
1261 
1262  trace->tr_at++;
1263  btr_trace_debug(tcx, trace, "trace back\n");
1264  break;

1266 
1267  while (trace < &tcx->tc_trace[tcx->tc_depth - 1]) {
1268  TMMID(struct btr_node) tmp;
1269 
1270  tmp = btr_node_child_at(tcx, trace->tr_node, trace->tr_at);
1271  trace++;
1272  trace->tr_at = 0;
1273  trace->tr_node = tmp;
1274  }
1275 
1276  btr_trace_debug(tcx, trace, "is the next\n");
1277  return true;
1278 }

1330 
1331  D_ASSERT(nd->tn_keyn != 0);
1332  trace->tr_at = nd->tn_keyn - leaf;
1333  }
1334

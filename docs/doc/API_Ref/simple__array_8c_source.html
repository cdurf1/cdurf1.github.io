394  } while (mpi_comp == 0 && daos_comp == 0);
395

399

401  ASSERT(count == 1, "Irecv test failed");

405 
407  rc = MPI_Irecv(&ep_rcv, 1, MPI_UINT64_T, MPI_ANY_SOURCE,
408  MPI_ANY_TAG, MPI_COMM_WORLD, &request);
409  ASSERT(rc == MPI_SUCCESS, "Irecv failed");
410  }
411 
413  if (daos_comp) {

415 
416  ASSERT(daos_comp == 1, "eq pool failed with %d",

434  ASSERT(ev.ev_error == 0,
435  "commit failed with %d", ev.ev_error);
436  *state = EP_COMMITTED;
438  epoch++;
439

445  break;

448  }

461  }
462  }

604 static int
605 ring_map_hash_build(struct pl_ring_map *rimap)
606 {
607  uint64_t range;
608  double stride;
609  double hash;
610  int i;
611  unsigned tg_per_dom;
612 
613  D_DEBUG(DB_PL, "Build consistent hash for ring map\n");
614  D_ALLOC(rimap->rmp_target_hashes,
615  rimap->rmp_target_nr * sizeof(*rimap->rmp_target_hashes));
616  if (rimap->rmp_target_hashes == NULL)
617  return -DER_NOMEM;
618 
619  D_ALLOC(rimap->rmp_ring_hashes,
620  rimap->rmp_ring_nr * sizeof(*rimap->rmp_ring_hashes));
621  if (rimap->rmp_ring_hashes == NULL)
622  return -DER_NOMEM;
623 
624  tg_per_dom = rimap->rmp_target_nr / rimap->rmp_domain_nr;
625  rimap->rmp_target_hbits = DOMAIN_BITS + TARGET_BITS +
626  daos_power2_nbits(tg_per_dom);
627  if (rimap->rmp_target_hbits > TARGET_HASH_BITS)
628  rimap->rmp_target_hbits = TARGET_HASH_BITS;
629 
630  range = 1ULL << rimap->rmp_target_hbits;
631 
632  D_DEBUG(DB_PL, "domanis %d, targets %d, hash range is 0-0x"DF_X64"\n",
633  rimap->rmp_domain_nr, rimap->rmp_target_nr, range);
634 
635  /* create consistent hash for targets */
636  stride = (double)range / rimap->rmp_target_nr;
637  rimap->rmp_stride = stride;
638 
639  hash = 0;
640  for (i = 0; i < rimap->rmp_target_nr; i++) {
641  rimap->rmp_target_hashes[i] = hash;
642  hash += stride;
643  }
644 
645  /* create consistent hash for rings */
646  range = 1ULL << RING_HASH_BITS;
647  stride = (double)range / rimap->rmp_ring_nr;
648  hash = 0;
649 
650  for (i = 0; i < rimap->rmp_ring_nr; i++) {
651  rimap->rmp_ring_hashes[i] = hash;
652  hash += stride;
653  }
654  return 0;
655 }

797  d_rank_t rank;

799  unsigned int i;

802  daos_obj_id2class(oid) == DAOS_OC_R1S_SPEC_RANK);
803

819  break;
820  }
821  if (i == rimap->rmp_target_nr)
822  return -DER_INVAL;

924  continue;

969 static bool
970 ring_remap_next_spare(struct pl_ring_map *rimap,
971  struct ring_obj_placement *rop, unsigned int *spare_idx)
972 {
973  unsigned int dist, max_dist, total_dist;
974 
975  D_ASSERTF(rop->rop_grp_size <= rimap->rmp_domain_nr,
976  "grp_size: %u > domain_nr: %u\n",
977  rop->rop_grp_size, rimap->rmp_domain_nr);
978 
979  /*
980  * Please be aware that we want to relocate the shard of
981  * non-replicated object (grp_size == 1) as well.
982  */
983  if (rop->rop_grp_size == rimap->rmp_domain_nr &&
984  rop->rop_grp_size > 1)
985  return false;
986 
987  /* Assume the ring consists of all pool targets. */
988  total_dist = rimap->rmp_target_nr;
989 
990  /* The max distance from spare index to rop->rop_begin */
991  max_dist = total_dist - rop->rop_grp_size * rop->rop_grp_nr;
992  /* Current distance from spare index to rop->rop_begin */
993  dist = (*spare_idx <= rop->rop_begin) ?
994  rop->rop_begin - *spare_idx :
995  rop->rop_begin + total_dist - *spare_idx;
996  /*
997  * Move the spare index forward, skip the domains where
998  * the original shards located on. Revise this when the
999  * rop_dist can be other values rather than 1.
1000  */
1001  if (!((dist + rop->rop_grp_size) % rimap->rmp_domain_nr))
1002  dist += rop->rop_grp_size;
1003  else
1004  dist++;
1005 
1006  if (dist > max_dist)
1007  return false;
1008 
1009  /* Convert distance to spare index */
1010  *spare_idx = (rop->rop_begin >= dist) ?
1011  rop->rop_begin - dist :
1012  total_dist - (dist - rop->rop_begin);
1013  return true;
1014 }

1039 static void
1040 ring_obj_remap_shards(struct pl_ring_map *rimap, struct daos_obj_md *md,
1041  struct pl_obj_layout *layout,
1042  struct ring_obj_placement *rop, d_list_t *remap_list)
1043 {
1044  struct ring_failed_shard *f_shard, *f_tmp;
1045  struct pl_target *plts;
1046  struct pl_obj_shard *l_shard;
1047  struct pool_target *spare_tgt, *tgts;
1048  d_list_t *current;
1049  unsigned int spare_idx;
1050  bool spare_avail = true;
1051 
1052  ring_remap_dump(remap_list, md, "before remap:");
1053 
1054  plts = ring_oid2ring(rimap, md->omd_id)->ri_targets;
1055  tgts = pool_map_targets(rimap->rmp_map.pl_poolmap);
1056  current = remap_list->next;
1057  spare_idx = rop->rop_begin;
1058 
1059  while (current != remap_list) {
1060  f_shard = d_list_entry(current, struct ring_failed_shard,
1061  rfs_list);
1062  l_shard = &layout->ol_shards[f_shard->rfs_shard_idx];
1063 
1064  /*
1065  * Select a spare target on the ring map for the current
1066  * failed shard, if there is no spare for former failed
1067  * shard, skip attempts for later ones.
1068  */
1069  spare_avail = spare_avail ?
1070  ring_remap_next_spare(rimap, rop, &spare_idx) : false;
1071 
1072  D_DEBUG(DB_PL, "obj:"DF_OID", select spare:%d grp_size:%u, "
1073  "grp_nr:%u, begin:%u, spare:%u\n", DP_OID(md->omd_id),
1074  spare_avail, rop->rop_grp_size, rop->rop_grp_nr,
1075  rop->rop_begin, spare_idx);
1076 
1077  if (!spare_avail)
1078  goto next;
1079 
1080  spare_tgt = &tgts[plts[spare_idx].pt_pos];
1081 
1082  /* The selected spare target is down as well */
1083  if (pool_target_unavail(spare_tgt)) {
1084  D_ASSERTF(spare_tgt->ta_comp.co_fseq !=
1085  f_shard->rfs_fseq, "same fseq %u!\n",
1086  f_shard->rfs_fseq);
1087 
1088  /*
1089  * The selected spare is down prior to current failed
1090  * one, then it can't be a valid spare, let's skip it
1091  * and try next spare on the ring.
1092  */
1093  if (spare_tgt->ta_comp.co_fseq < f_shard->rfs_fseq)
1094  continue;
1095 
1096  /*
1097  * Both failed target and spare target are down, skip
1098  * reuild or rw for this shard and all other failed
1099  * shards with higher fseq. The reuild for this shard
1100  * will be deferred till the later rebuild for this
1101  * down spare target.
1102  */
1103  if (f_shard->rfs_status == PO_COMP_ST_DOWN) {
1104  D_ASSERTF(spare_tgt->ta_comp.co_status !=
1105  PO_COMP_ST_DOWNOUT,
1106  "down fseq(%u) < downout fseq(%u)\n",
1107  f_shard->rfs_fseq,
1108  spare_tgt->ta_comp.co_fseq);
1109  spare_avail = false;
1110  goto next;
1111  }
1112 
1113  /* Current failed shard is in PO_COMP_ST_DOWNOUT */
1114  f_shard->rfs_fseq = spare_tgt->ta_comp.co_fseq;
1115  f_shard->rfs_status = spare_tgt->ta_comp.co_status;
1116 
1117  current = current->next;
1118  d_list_del_init(&f_shard->rfs_list);
1119  ring_remap_add_one(remap_list, f_shard);
1120 
1121  /* Continue with the failed shard has minimal fseq */
1122  if (current == remap_list) {
1123  current = &f_shard->rfs_list;
1124  } else {
1125  f_tmp = d_list_entry(current,
1126  struct ring_failed_shard,
1127  rfs_list);
1128  if (f_shard->rfs_fseq < f_tmp->rfs_fseq)
1129  current = &f_shard->rfs_list;
1130  }
1131  continue;
1132  }
1133 next:
1134  if (spare_avail) {
1135  /* The selected spare target is up and ready */
1136  l_shard->po_target = spare_tgt->ta_comp.co_id;
1137 
1138  /*
1139  * Mark the shard as 'rebuilding' so that read will
1140  * skip this shard.
1141  */
1142  if (f_shard->rfs_status == PO_COMP_ST_DOWN) {
1143  l_shard->po_rebuilding = 1;
1144  f_shard->rfs_rank = spare_tgt->ta_comp.co_rank;
1145  /*
1146  * It doesn't make sense to continue remap on
1147  * higher fseq anymore, rw should skip these
1148  * failed shards since they'll be rebuilt later
1149  * anyway.
1150  */
1151  spare_avail = false;
1152  }
1153  } else {
1154  l_shard->po_shard = -1;
1155  l_shard->po_target = -1;
1156  }
1157  current = current->next;
1158  }
1159 
1160  ring_remap_dump(remap_list, md, "after remap:");
1161 }

1302 #define SHARDS_ON_STACK_COUNT 256
1303 int
1304 ring_obj_find_rebuild(struct pl_map *map, struct daos_obj_md *md,
1305  struct daos_obj_shard_md *shard_md,
1306  uint32_t rebuild_ver, uint32_t *tgt_rank,
1307  uint32_t *shard_id)
1308 {
1309  struct ring_obj_placement rop;
1310  struct pl_ring_map *rimap = pl_map2rimap(map);
1311  struct pl_obj_layout *layout;
1312  struct pl_obj_layout layout_on_stack;
1313  struct pl_obj_shard shards_on_stack[SHARDS_ON_STACK_COUNT];
1314  d_list_t remap_list;
1315  struct ring_failed_shard *f_shard;
1316  struct pl_obj_shard *l_shard;
1317  unsigned int shards_count;
1318  int rc;
1319 
1320  /* Caller should guarantee the pl_map is uptodate */
1321  if (pl_map_version(map) < rebuild_ver) {
1322  D_ERROR("pl_map version(%u) < rebuild version(%u)\n",
1323  pl_map_version(map), rebuild_ver);
1324  return -DER_INVAL;
1325  }
1326 
1327  rc = ring_obj_placement_get(rimap, md, shard_md, &rop);
1328  if (rc)
1329  return rc;
1330 
1331  if (rop.rop_grp_size == 1) {
1332  D_DEBUG(DB_PL, "Not replicated object "DF_OID"\n",
1333  DP_OID(md->omd_id));
1334  return 0;
1335  }
1336 
1337  shards_count = rop.rop_grp_size * rop.rop_grp_nr;
1338  if (shards_count > SHARDS_ON_STACK_COUNT) {
1339  rc = pl_obj_layout_alloc(shards_count, &layout);
1340  if (rc)
1341  return rc;
1342  } else {
1343  layout = &layout_on_stack;
1344  layout->ol_nr = shards_count;
1345  layout->ol_shards = shards_on_stack;
1346  }
1347 
1348  D_INIT_LIST_HEAD(&remap_list);
1349  rc = ring_obj_layout_fill(map, md, &rop, layout, &remap_list);
1350  if (rc)
1351  goto out;
1352 
1353  d_list_for_each_entry(f_shard, &remap_list, rfs_list) {
1354  l_shard = &layout->ol_shards[f_shard->rfs_shard_idx];
1355 
1356  if (f_shard->rfs_fseq > rebuild_ver)
1357  break;
1358 
1359  if (f_shard->rfs_fseq < rebuild_ver) {
1360  if (f_shard->rfs_status != PO_COMP_ST_DOWNOUT)
1361  D_ERROR(""DF_OID" rebuild isn't done for "
1362  "fseq:%d(status:%d)? rbd_ver:%d\n",
1363  DP_OID(md->omd_id), f_shard->rfs_fseq,
1364  f_shard->rfs_status, rebuild_ver);
1365  continue;
1366  }

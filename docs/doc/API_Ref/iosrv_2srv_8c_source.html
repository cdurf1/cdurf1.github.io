171  if (ret != ABT_SUCCESS) {

287 dss_srv_handler(void *arg)
288 {

290  struct dss_thread_local_storage *dtc;
291  struct dss_module_info *dmi;
292  int rc;
293  bool signal_caller = true;
294 
296  rc = hwloc_set_cpubind(dss_topo, dx->dx_cpuset, HWLOC_CPUBIND_THREAD);
297  if (rc) {
298  D_ERROR("failed to set affinity: %d\n", errno);
299  goto signal;
300  }
301 
302  /* initialize xstream-local storage */
303  dtc = dss_tls_init(DAOS_SERVER_TAG);
304  if (dtc == NULL) {
305  D_ERROR("failed to initialize TLS\n");
306  goto signal;
307  }
308 
309  dmi = dss_get_module_info();
310  D_ASSERT(dmi != NULL);
311 
312  /* create private transport context */
313  rc = crt_context_create(&dmi->dmi_ctx);
314  if (rc != 0) {
315  D_ERROR("failed to create crt ctxt: %d\n", rc);
316  goto tls_fini;
317  }
318 
319  rc = crt_context_register_rpc_task(dmi->dmi_ctx,
320  dss_process_rpc,
321  dx->dx_pools);
322  if (rc != 0) {
323  D_ERROR("failed to register process cb %d\n", rc);
324  goto crt_destroy;
325  }
326 
328  rc = crt_context_idx(dmi->dmi_ctx, &dmi->dmi_tid);
329  if (rc != 0) {
330  D_ERROR("failed to get xtream index: rc %d\n", rc);
331  goto crt_destroy;
332  }
333 
334  /* Prepare the scheduler */
335  rc = tse_sched_init(&dmi->dmi_sched, NULL, dmi->dmi_ctx);
336  if (rc != 0) {
337  D_ERROR("failed to init the scheduler\n");
338  goto crt_destroy;
339  }
340 
341  /* Initialize NVMe context */
342  rc = dss_nvme_ctxt_init(&dmi->dmi_nvme_ctxt, dmi->dmi_tid);
343  if (rc != 0) {
344  D_ERROR("failed to init spdk context for xstream(%d) rc:%d\n",
345  dx->dx_idx, rc);
346  D_GOTO(tse_fini, rc);
347  }
348 
349  dx->dx_idx = dmi->dmi_tid;
350  dmi->dmi_xstream = dx;
351  ABT_mutex_lock(xstream_data.xd_mutex);
352  /* initialized everything for the ULT, notify the creater */
353  D_ASSERT(!xstream_data.xd_ult_signal);
354  xstream_data.xd_ult_signal = true;
355  xstream_data.xd_ult_init_rc = 0;
356  ABT_cond_signal(xstream_data.xd_ult_init);
357 
358  /* wait until all xstreams are ready, otherwise it is not safe
359  * to run lock-free dss_collective, althought this race is not
360  * realistically possible in the DAOS stack.
361  */
362  ABT_cond_wait(xstream_data.xd_ult_barrier, xstream_data.xd_mutex);
363  ABT_mutex_unlock(xstream_data.xd_mutex);
364 
365  signal_caller = false;
366  /* main service progress loop */
367  for (;;) {
368  ABT_bool state;
369 
370  rc = crt_progress(dmi->dmi_ctx, 0 /* no wait */, NULL, NULL);
371  if (rc != 0 && rc != -DER_TIMEDOUT) {
372  D_ERROR("failed to progress network context: %d\n",
373  rc);
374  /* XXX if it exists the loop here, then tls to be freed,
375  * other ULTs on this xstream might get freed TLS, so
376  * let's not break here until we have proper server stop
377  * procedures, i.e. this server xstream should wait until
378  * all ULTs DAOS-769 */
379  }
380 
381  dss_nvme_poll(&dmi->dmi_nvme_ctxt);
382 
383  rc = ABT_future_test(dx->dx_shutdown, &state);
384  D_ASSERTF(rc == ABT_SUCCESS, "%d\n", rc);
385  if (state == ABT_TRUE)
386  break;
387 
388  ABT_thread_yield();
389  }
390 
391  dss_nvme_ctxt_fini(&dmi->dmi_nvme_ctxt);
392 tse_fini:
393  tse_sched_fini(&dmi->dmi_sched);
394 crt_destroy:
395  crt_context_destroy(dmi->dmi_ctx, true);
396 tls_fini:
397  dss_tls_fini(dtc);
398 signal:
399  if (signal_caller) {
400  ABT_mutex_lock(xstream_data.xd_mutex);
401  /* initialized everything for the ULT, notify the creater */
402  D_ASSERT(!xstream_data.xd_ult_signal);
403  xstream_data.xd_ult_signal = true;
404  xstream_data.xd_ult_init_rc = rc;
405  ABT_cond_signal(xstream_data.xd_ult_init);
406  ABT_mutex_unlock(xstream_data.xd_mutex);
407  }

468 static int
469 dss_start_one_xstream(hwloc_cpuset_t cpus, int idx)
470 {
471  struct dss_xstream *dx;
472  int rc = 0;
473  int i;
474 
476  dx = dss_xstream_alloc(cpus);
477  if (dx == NULL)
478  return -DER_NOMEM;
479 
481  for (i = 0; i < DSS_POOL_CNT; i++) {
482  ABT_pool_access access;
483 
484  access = i == DSS_POOL_SHARE ?
485  ABT_POOL_ACCESS_MPSC : ABT_POOL_ACCESS_PRIV;
486 
487  rc = ABT_pool_create_basic(ABT_POOL_FIFO, access, ABT_TRUE,
488  &dx->dx_pools[i]);
489  if (rc != ABT_SUCCESS)
490  D_GOTO(out_pool, rc = dss_abterr2der(rc));
491  }
492 
493  rc = dss_sched_create(dx->dx_pools, DSS_POOL_CNT, &dx->dx_sched);
494  if (rc != 0) {
495  D_ERROR("create scheduler fails: %d\n", rc);
496  D_GOTO(out_pool, rc);
497  }
498 
500  rc = ABT_xstream_create_with_rank(dx->dx_sched, idx, &dx->dx_xstream);
501  if (rc != ABT_SUCCESS) {
502  D_ERROR("create xstream fails %d\n", rc);
503  D_GOTO(out_sched, rc = dss_abterr2der(rc));
504  }
505 
507  rc = ABT_thread_create(dx->dx_pools[DSS_POOL_SHARE],
508  dss_srv_handler, dx, ABT_THREAD_ATTR_NULL,
509  &dx->dx_progress);
510  if (rc != ABT_SUCCESS) {
511  D_ERROR("create xstream failed: %d\n", rc);
512  D_GOTO(out_xstream, rc = dss_abterr2der(rc));
513  }
514 
515  ABT_mutex_lock(xstream_data.xd_mutex);
516 
517  if (!xstream_data.xd_ult_signal)
518  ABT_cond_wait(xstream_data.xd_ult_init, xstream_data.xd_mutex);
519  xstream_data.xd_ult_signal = false;
520  rc = xstream_data.xd_ult_init_rc;
521  if (rc != 0) {
522  ABT_mutex_unlock(xstream_data.xd_mutex);
523  goto out_xstream;
524  }
526  d_list_add_tail(&dx->dx_list, &xstream_data.xd_list);
527  ABT_mutex_unlock(xstream_data.xd_mutex);
528 
529  return 0;
530 out_xstream:
531  ABT_xstream_join(dx->dx_xstream);
532  ABT_xstream_free(&dx->dx_xstream);
533  dss_xstream_free(dx);
534  return rc;
535 out_sched:
536  ABT_sched_free(&dx->dx_sched);
537 out_pool:
538  for (i = 0; i < DSS_POOL_CNT; i++) {
539  if (dx->dx_pools[i] != ABT_POOL_NULL)
540  ABT_pool_free(&dx->dx_pools[i]);
541  }
542  dss_xstream_free(dx);
543  return rc;
544 }

602 static int
603 dss_xstreams_init(int nr)
604 {
605  int rc;
606  int i;
607  int depth;
608  int ncores;
609 
610  depth = hwloc_get_type_depth(dss_topo, HWLOC_OBJ_CORE);
612  ncores = hwloc_get_nbobjs_by_type(dss_topo, HWLOC_OBJ_CORE);
613 
614  if (!nr || nr > ncores)
615  D_PRINT("(%d/%d) cores requested; use default (%d) cores\n",
616  nr, ncores, ncores);
618  dss_nxstreams = (nr > 0 && nr <= ncores) ? nr : ncores;
619 
620  /* initialize xstream-local storage */
621  rc = pthread_key_create(&dss_tls_key, NULL);
622  if (rc) {
623  D_ERROR("failed to create dtc: %d\n", rc);
624  return -DER_NOMEM;
625  }
626 
627  /* start the execution streams */
628  D_DEBUG(DB_TRACE, "%d cores detected, starting %d execution streams\n",
629  ncores, dss_nxstreams);
630  for (i = 1; i <= dss_nxstreams; i++) {
631  hwloc_obj_t obj;
632 
633  obj = hwloc_get_obj_by_depth(dss_topo, depth, i % ncores);
634  if (obj == NULL) {
635  D_ERROR("Null core returned by hwloc\n");
636  D_GOTO(failed, rc = -DER_INVAL);
637  }
638 
640  rc = dss_start_one_xstream(obj->allowed_cpuset, i);
641  if (rc)
642  D_GOTO(failed, rc);
643  }
644  D_DEBUG(DB_TRACE, "%d execution streams successfully started\n",
645  dss_nxstreams);
646 failed:
647  dss_xstreams_open_barrier();
648  if (dss_xstreams_empty()) /* started nothing */
649  pthread_key_delete(dss_tls_key);
650 
651  return rc;
652 }

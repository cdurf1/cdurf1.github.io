60 static int
61 credit_poll(struct dts_context *tsc, bool drain)
62 {
63  daos_event_t *evs[DTS_CRED_MAX];
64  int i;
65  int rc;
66 
67  if (tsc->tsc_cred_inuse == 0)
68  return 0; /* nothing inflight (sync mode never set inuse) */
69 
70  while (1) {
71  rc = daos_eq_poll(tsc->tsc_eqh, 0, DAOS_EQ_WAIT, DTS_CRED_MAX,
72  evs);
73  if (rc < 0) {
74  fprintf(stderr, "failed to pool event: %d\n", rc);
75  return rc;
76  }
77 
78  for (i = 0; i < rc; i++) {
79  int err = evs[i]->ev_error;
80 
81  if (err != 0) {
82  fprintf(stderr, "failed op: %d\n", err);
83  return err;
84  }
85  tsc->tsc_credits[tsc->tsc_cred_avail] =
86  container_of(evs[i], struct dts_io_credit, tc_ev);
87 
88  tsc->tsc_cred_inuse--;
89  tsc->tsc_cred_avail++;
90  }
91 
92  if (tsc->tsc_cred_avail == 0)
93  continue; /* still no available event */
94 
95  /* if caller wants to drain, is there any event inflight? */
96  if (tsc->tsc_cred_inuse != 0 && drain)
97  continue;
98 
99  return 0;
100  }
101 }

106 {

108

114  tsc->tsc_cred_avail--;

120  if (rc)
121  return NULL;

132 static int
133 credits_init(struct dts_context *tsc)
134 {
135  int i;
136  int rc;
137 
138  if (tsc->tsc_cred_nr > 0) {
139  rc = daos_eq_create(&tsc->tsc_eqh);
140  if (rc)
141  return rc;
142 
143  if (tsc->tsc_cred_nr > DTS_CRED_MAX)
144  tsc->tsc_cred_avail = tsc->tsc_cred_nr = DTS_CRED_MAX;
145  else
146  tsc->tsc_cred_avail = tsc->tsc_cred_nr;
147  } else { /* synchronous mode */
148  tsc->tsc_eqh = DAOS_HDL_INVAL;
149  tsc->tsc_cred_nr = 1; /* take one slot in the buffer */
150  tsc->tsc_cred_avail = -1; /* always available */
151  }
152 
153  for (i = 0; i < tsc->tsc_cred_nr; i++) {
154  struct dts_io_credit *cred = &tsc->tsc_cred_buf[i];
155 
156  memset(cred, 0, sizeof(*cred));
157  cred->tc_vbuf = calloc(1, tsc->tsc_cred_vsize);
158  if (!cred->tc_vbuf) {
159  fprintf(stderr, "Cannt allocate buffer size=%d\n",
160  tsc->tsc_cred_vsize);
161  return -1;
162  }
163 
164  if (!daos_handle_is_inval(tsc->tsc_eqh)) {
165  rc = daos_event_init(&cred->tc_ev, tsc->tsc_eqh, NULL);
166  D_ASSERTF(!rc, "rc=%d\n", rc);
167  cred->tc_evp = &cred->tc_ev;
168  }
169  tsc->tsc_credits[i] = cred;
170  }
171  return 0;
172 }

192 static int
193 pool_init(struct dts_context *tsc)
194 {
195  daos_handle_t poh = DAOS_HDL_INVAL;
196  int rc;
197 
198  if (tsc->tsc_pool_size == 0)
199  tsc->tsc_pool_size = (1ULL << 30);
200 
201  if (tsc->tsc_pmem_file) { /* VOS mode */
202  char *pmem_file = tsc->tsc_pmem_file;
203  int fd;
204 
205  if (!daos_file_is_dax(pmem_file)) {
206  rc = open(pmem_file, O_CREAT | O_TRUNC | O_RDWR, 0666);
207  if (rc < 0)
208  goto out;
209 
210  fd = rc;
211  rc = fallocate(fd, 0, 0, tsc->tsc_pool_size);
212  if (rc)
213  goto out;
214  }
215 
216  rc = vos_pool_create(pmem_file, tsc->tsc_pool_uuid, 0);
217  if (rc)
218  goto out;
219 
220  rc = vos_pool_open(pmem_file, tsc->tsc_pool_uuid, &poh);
221  if (rc)
222  goto out;
223 
224  } else if (tsc->tsc_mpi_rank == 0) { /* DAOS mode and rank zero */
225  d_rank_list_t *svc = &tsc->tsc_svc;
226 
227  rc = daos_pool_create(0731, geteuid(), getegid(),
228  NULL, NULL, "pmem",
229  tsc->tsc_pool_size, svc,
230  tsc->tsc_pool_uuid, NULL);
231  if (rc)
232  goto bcast;
233 
234  rc = daos_pool_connect(tsc->tsc_pool_uuid, NULL, svc,
235  DAOS_PC_EX, &poh, NULL, NULL);
236  if (rc)
237  goto bcast;
238  }
239  tsc->tsc_poh = poh;
240  bcast:
241  if (tsc->tsc_mpi_size <= 1)
242  goto out; /* don't need to share handle */
243 
244  MPI_Bcast(&rc, 1, MPI_INT, 0, MPI_COMM_WORLD);
245  if (rc)
246  goto out; /* open failed */
247 
248  handle_share(&tsc->tsc_poh, HANDLE_POOL, tsc->tsc_mpi_rank,
249  tsc->tsc_poh, 0);
250  out:
251  return rc;
252 }

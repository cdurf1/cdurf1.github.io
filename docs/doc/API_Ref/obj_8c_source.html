43 ds_obj_retry_cb(tse_task_t *task, void *arg)
44 {
45  daos_handle_t *oh = arg;
46  int rc;
47 
48  if (task->dt_result != -DER_NO_HDL || oh == NULL)
49  return 0;
50 
51  /* If the remote rebuild pool/container is not ready,
52  * or the remote target has been evicted from pool.
53  * Note: the pool map will redistributed by IV
54  * automatically, so let's just keep refreshing the
55  * layout.
56  */
57  D_DEBUG(DB_TRACE, "retry task %p\n", task);
58 
59  /* let's check if the pool_map has been changed */
60  dc_obj_layout_refresh(*oh);
61 
62  rc = dc_task_resched(task);
63  if (rc != 0) {
64  D_ERROR("Failed to re-init task (%p)\n", task);
65  return rc;
66  }
67 
68  /* Register the callback again, because it will be removed
69  * after callback.
70  */
71  rc = dc_task_reg_comp_cb(task, ds_obj_retry_cb, arg, sizeof(arg));
72  return rc;
73 }

531  pool_domain_name(parent),

534  }
535  }
536

538  break;
539  }
540

542

544  domain = &tree[0]) {

559  parent->do_target_nr += domain->do_target_nr;
560 
561  D_DEBUG(DB_MGMT, "Set %d target for %s[%d]\n",
562  parent->do_target_nr,
563  pool_comp_type2str(parent->do_comp.co_type),
564  parent->do_comp.co_id);
565  }
566  }

838  }
839  }

987 static int
988 pool_map_compat(struct pool_map *map, uint32_t version,
989  struct pool_domain *tree)
990 {
991  struct pool_domain *parent;
992  struct pool_domain *doms;
993  int dom_nr;
994  int rc;
995 
996  if (pool_map_empty(map)) {
997  D_DEBUG(DB_MGMT, "empty map, type of buffer root is %s\n",
998  pool_domain_name(&tree[0]));
999  return 0;
1000  }
1001 
1002  if (map->po_version >= version)
1003  return -DER_NO_PERM;
1004 
1005  /* pool_buf_parse should always generate root */
1006  if (tree[0].do_comp.co_type != PO_COMP_TP_ROOT)
1007  return -DER_INVAL;
1008 
1009  rc = pool_map_find_domain(map, tree[1].do_comp.co_type,
1010  PO_COMP_ID_ALL, &doms);
1011  if (rc == 0)
1012  return -DER_INVAL;
1013 
1014  if (doms - map->po_tree == 1) {
1015  /* the first component is indeed under the root */
1016  parent = &tree[0];
1017  } else {
1018  /* root of the new tree is dummy */
1019  parent = NULL;
1020  }
1021 
1022  D_DEBUG(DB_MGMT, "Check if buffer is compatible with pool map\n");
1023 
1024  dom_nr = tree[0].do_child_nr;
1025  for (tree++; tree != NULL; parent = &tree[0],
1026  tree = tree[0].do_children,
1027  doms = doms[0].do_children) {
1028  int child_nr = 0;
1029  int nr = 0;
1030  int i;
1031  int j;
1032 
1033  if (doms == NULL) {
1034  D_DEBUG(DB_MGMT, "tree has more layers than the map\n");
1035  return -DER_INVAL;
1036  }
1037 
1038  D_DEBUG(DB_MGMT, "checking %s/%s\n",
1039  pool_domain_name(&tree[0]),
1040  pool_domain_name(&doms[0]));
1041 
1042  for (i = 0; i < dom_nr; i++) {
1043  struct pool_component *dc = &tree[i].do_comp;
1044  bool existed;
1045 
1046  if (dc->co_type != doms[0].do_comp.co_type) {
1047  D_DEBUG(DB_MGMT,
1048  "domain type not match %s(%u) %s(%u)\n",
1049  pool_comp_name(dc), dc->co_type,
1050  pool_domain_name(&doms[0]),
1051  doms[0].do_comp.co_type);
1052  return -DER_INVAL;
1053  }
1054 
1055  existed = domain_exist(map, dc->co_type, dc->co_id);
1056  if (dc->co_status == PO_COMP_ST_NEW) {
1057  if (parent == NULL)
1058  return -DER_INVAL;
1059  if (existed)
1060  return -DER_NO_PERM;
1061 
1062  } else if (dc->co_status == PO_COMP_ST_UP) {
1063  if (!existed)
1064  return -DER_INVAL;
1065 
1066  if (parent->do_comp.co_status == PO_COMP_ST_NEW)
1067  return -DER_INVAL;
1068 
1069  } else {
1070  return -DER_INVAL;
1071  }
1072 
1073  if (tree[i].do_children != NULL) {
1074  child_nr += tree[i].do_child_nr;
1075  } else {
1076  /* the last layer domain */
1077  if (doms[0].do_children != NULL) {
1078  D_DEBUG(DB_MGMT, "unmatched tree\n");
1079  return -DER_INVAL;
1080  }
1081 
1082  for (j = 0; j < tree[i].do_target_nr; j++) {
1083  struct pool_component *tc;
1084 
1085  tc = &tree[i].do_targets[j].ta_comp;
1086  if (tc->co_status != PO_COMP_ST_NEW ||
1087  target_exist(map, tc->co_id))
1088  return -DER_INVAL;
1089  }
1090  }
1091 
1092  nr++;
1093  if (parent != NULL && parent->do_child_nr == nr) {
1094  parent++;
1095  nr = 0;
1096  }
1097  }
1098  dom_nr = child_nr;
1099  }
1100  return 0;
1101 }

1107 static int
1108 pool_map_merge(struct pool_map *map, uint32_t version,
1109  struct pool_domain *tree)
1110 {
1111  struct pool_map *src_map;
1112  struct pool_domain *dst_tree;
1113  struct pool_domain *dst_doms;
1114  struct pool_domain *cur_doms;
1115  void *addr;
1116  struct pool_comp_cntr cntr;
1117  unsigned int dom_nr;
1118  unsigned int size;
1119  int i;
1120  int rc;
1121 
1122  /* create scratch map for merging */
1123  D_ALLOC_PTR(src_map);
1124  if (src_map == NULL)
1125  return -DER_NOMEM;
1126 
1127  rc = pool_map_initialise(src_map, false, tree);
1128  if (rc != 0) {
1129  D_DEBUG(DB_MGMT, "Failed to create scratch map for buffer\n");
1130  goto failed;
1131  }
1132 
1133  /* destination buffer could has larger than the actually needed space,
1134  * but it is not big deal.
1135  */
1136  size = pool_tree_size(map->po_tree) + pool_tree_size(tree);
1137  D_ALLOC(dst_tree, size);
1138  if (dst_tree == NULL) {
1139  rc = -DER_NOMEM;
1140  goto failed;
1141  }
1142 
1143  /* copy current pool map to destination buffer */
1144  pool_tree_copy(dst_tree, map->po_tree);
1145 
1146  if (src_map->po_domain_layers != map->po_domain_layers) {
1147  /* source map may have less levels because it could be in
1148  * a subtree, skip the fake root in this case.
1149  */
1150  D_ASSERT(src_map->po_domain_layers < map->po_domain_layers);
1151  rc = pool_map_find_domain(map, tree[1].do_comp.co_type,
1152  PO_COMP_ID_ALL, &cur_doms);
1153  } else {
1154  rc = pool_map_find_domain(map, tree[0].do_comp.co_type,
1155  PO_COMP_ID_ALL, &cur_doms);
1156  }
1157  if (rc != 0)
1158  goto failed;
1159 
1160  dst_doms = dst_tree;
1161  dst_doms += cur_doms - map->po_tree;
1162  pool_tree_count(dst_doms, &cntr);
1163  dom_nr = cntr.cc_top_doms;
1164 
1165  /* overwrite the components after the top layer domains */
1166  addr = (void *)&dst_doms[dom_nr];
1167  pool_tree_count(dst_tree, &cntr);
1168 
1169  /* complex buffer manipulating... */
1170  for (; dst_doms != NULL;
1171  dst_doms = dst_doms[0].do_children,
1172  cur_doms = cur_doms[0].do_children) {
1173  struct pool_domain *cdom = &cur_doms[0];
1174  int child_nr = 0;
1175 
1176  for (i = 0; i < dom_nr; i++) {
1177  struct pool_domain *ddom = &dst_doms[i];
1178  struct pool_domain *sdom;
1179  int nb;
1180  int j;
1181 
1182  if (ddom->do_comp.co_ver == version) {
1183  ddom->do_children = NULL;
1184  ddom->do_targets = NULL;
1185  ddom->do_child_nr = 0;
1186  ddom->do_target_nr = 0;
1187  D_DEBUG(DB_MGMT, "Add new domain %s %d\n",
1188  pool_domain_name(cdom), dom_nr);
1189  } else {
1190  /* Domain existed, copy its children/targets
1191  * from current pool map.
1192  */
1193  D_ASSERT(ddom->do_comp.co_ver < version);
1194  D_ASSERT(ddom->do_comp.co_id ==
1195  cdom->do_comp.co_id);
1196 
1197  if (cdom->do_children != NULL) {
1198  ddom->do_children = addr;
1199  ddom->do_child_nr = cdom->do_child_nr;
1200  nb = cdom->do_child_nr *
1201  sizeof(struct pool_domain);
1202  memcpy(addr, cdom->do_children, nb);
1203  } else {
1204  ddom->do_targets = addr;
1205  ddom->do_target_nr = cdom->do_target_nr;
1206  nb = cdom->do_target_nr *
1207  sizeof(struct pool_target);
1208  memcpy(addr, cdom->do_targets, nb);
1209  }
1210  addr += nb;
1211  cdom++;
1212  }
1213 
1214  D_DEBUG(DB_MGMT, "Check changes for %s[%d]\n",
1215  pool_domain_name(ddom), ddom->do_comp.co_id);
1216 
1217  rc = pool_map_find_domain(src_map,
1218  ddom->do_comp.co_type,
1219  ddom->do_comp.co_id, &sdom);
1220  if (rc == 0) {
1221  child_nr += ddom->do_child_nr;
1222  continue; /* no change for this domain */
1223  }
1224 
1225  /* new buffer may have changes for this domain */
1226  if (sdom->do_children != NULL) {
1227  struct pool_domain *child = addr;
1228 
1229  D_DEBUG(DB_MGMT, "Scan children of %s[%d]\n",
1230  pool_domain_name(ddom),
1231  ddom->do_comp.co_id);
1232 
1233  if (ddom->do_children == NULL)
1234  ddom->do_children = child;
1235 
1236  /* copy new child domains to dest buffer */
1237  for (j = 0; j < sdom->do_child_nr; j++) {
1238  struct pool_component *dc;
1239 
1240  dc = &sdom->do_children[j].do_comp;
1241  /* ignore existent children */
1242  if (dc->co_status != PO_COMP_ST_NEW)
1243  continue;
1244 
1245  D_DEBUG(DB_MGMT, "New %s[%d]\n",
1246  pool_comp_type2str(dc->co_type),
1247  dc->co_id);
1248 
1249  dc->co_status = PO_COMP_ST_UP;
1250 
1251  *child = sdom->do_children[j];
1252  child++;
1253 
1254  ddom->do_child_nr++;
1255  cntr.cc_domains++;
1256  }
1257  addr = child;
1258  } else {
1259  struct pool_target *target = addr;
1260 
1261  D_DEBUG(DB_MGMT, "Scan targets of %s[%d]\n",
1262  pool_domain_name(ddom),
1263  ddom->do_comp.co_id);
1264 
1265  if (ddom->do_targets == NULL)
1266  ddom->do_targets = target;
1267 
1268  /* copy new targets to destination buffer */
1269  for (j = 0; j < sdom->do_target_nr; j++) {
1270  struct pool_component *tc;
1271 
1272  tc = &sdom->do_targets[j].ta_comp;
1273 
1274  if (tc->co_status != PO_COMP_ST_NEW)
1275  continue;
1276 
1277  D_DEBUG(DB_MGMT, "New target[%d]\n",
1278  tc->co_id);
1279 
1280  tc->co_status = PO_COMP_ST_UP;
1281 
1282  *target = sdom->do_targets[j];
1283  target++;
1284 
1285  ddom->do_target_nr++;
1286  cntr.cc_targets++;
1287  }
1288  addr = target;
1289  }
1290  child_nr += ddom->do_child_nr;
1291  }
1292  dom_nr = child_nr;
1293  }
1294  D_ASSERT(addr - (void *)dst_tree <= size);
1295  D_DEBUG(DB_MGMT, "Merged all components\n");
1296  /* At this point, I only have valid children pointers for the last
1297  * layer domains, and need to build target pointers for all layers.
1298  */
1299  pool_tree_build_ptrs(dst_tree, &cntr);
1300 
1301  /* release old buffers of pool map */
1302  pool_map_finalise(map);
1303 
1304  /* install new buffer for pool map */
1305  rc = pool_map_initialise(map, true, dst_tree);
1306  D_ASSERT(rc == 0 || rc == -DER_NOMEM);
1307 
1308  map->po_version = version;
1309  failed:
1310  pool_map_destroy(src_map);
1311  return rc;
1312 }

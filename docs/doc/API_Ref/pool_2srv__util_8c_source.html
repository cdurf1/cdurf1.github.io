114 map_ranks_merge(d_rank_list_t *src_ranks, d_rank_list_t *ranks_merge)
115 {
116  d_rank_t *rs;
117  int *indexes;
118  int num = 0;
119  int src_num;
120  int i;
121  int j;
122  int rc = 0;
123 
124  if (ranks_merge == NULL || src_ranks == NULL)
125  return 0;
126 
127  src_num = src_ranks->rl_nr;
128  D_ALLOC(indexes, sizeof(*indexes) * ranks_merge->rl_nr);
129  if (indexes == NULL)
130  return -DER_NOMEM;
131 
132  for (i = 0; i < ranks_merge->rl_nr; i++) {
133  bool included = false;
134 
135  for (j = 0; j < src_num; j++) {
136  if (src_ranks->rl_ranks[j] ==
137  ranks_merge->rl_ranks[i]) {
138  included = true;
139  break;
140  }
141  }
142 
143  if (!included) {
144  indexes[num] = i;
145  num++;
146  }
147  }
148 
149  if (num == 0)
150  D_GOTO(free, rc = 0);
151 
152  D_ALLOC(rs, sizeof(*rs) * (num + src_ranks->rl_nr));
153  if (rs == NULL)
154  D_GOTO(free, rc = -DER_NOMEM);
155 
156  for (i = 0; i < src_num; i++)
157  rs[i] = src_ranks->rl_ranks[i];
158 
159  for (i = src_num, j = 0; i < src_num + num; i++, j++) {
160  int idx = indexes[j];
161 
162  rs[i] = ranks_merge->rl_ranks[idx];
163  }
164 
165  map_ranks_fini(src_ranks);
166 
167  src_ranks->rl_nr = num + src_num;
168  src_ranks->rl_ranks = rs;
169 
170 free:
171  D_FREE(indexes);
172  return rc;
173 }

288  * changes have been made. If "tgts_failed" is not NULL, then targets that are

298  int i;

300  int rc;
301  uint32_t tgts_failed_out;
302  uint32_t tgts_sanitized_size = 0;
303 
304  D_ASSERT(tgts != NULL && tgts->rl_nr > 0 && tgts->rl_ranks != NULL);
305  D_ASSERT(tgts_failed == NULL ||
306  (tgts_failed->rl_nr >= tgts->rl_nr &&
307  tgts_failed->rl_ranks != NULL));
308 
309  if (tgts_failed != NULL)

311

314  if (rc != 0)

316 
317  version = pool_map_get_version(map) + 1;
318 
319  for (i = 0; i < tgts_sanitized->rl_nr; i++) {
320  struct pool_target *target;
321  d_rank_t rank = tgts_sanitized->rl_ranks[i];
322 
323  target = pool_map_find_target_by_rank(map, rank);
324  if (target == NULL) {

326  rank, map);

329

331  tgts_failed_out++;

333  continue;
334  }
335 
336  D_ASSERTF(target->ta_comp.co_status == PO_COMP_ST_UP ||
337  target->ta_comp.co_status == PO_COMP_ST_UPIN ||
338  target->ta_comp.co_status == PO_COMP_ST_DOWN ||
339  target->ta_comp.co_status == PO_COMP_ST_DOWNOUT,

343  target->ta_comp.co_status != PO_COMP_ST_DOWNOUT) {

347  target->ta_comp.co_fseq = version;
348  nchanges++;
349  } else if (opc == POOL_ADD &&
350  target->ta_comp.co_status != PO_COMP_ST_UP &&
351  target->ta_comp.co_status != PO_COMP_ST_UPIN) {
352  D_DEBUG(DF_DSMS, "changing rank %u to UP in map %p\n",
353  target->ta_comp.co_rank, map);
354  target->ta_comp.co_status = PO_COMP_ST_UP;
355  target->ta_comp.co_ver = version;
356  target->ta_comp.co_fseq = 0;
357  nchanges++;
358  } else if (opc == POOL_EXCLUDE_OUT &&
359  target->ta_comp.co_status == PO_COMP_ST_DOWN) {
360  D_DEBUG(DF_DSMS, "changing rank %u to DOWNOUT map %p\n",
361  target->ta_comp.co_rank, map);
362  target->ta_comp.co_status = PO_COMP_ST_DOWNOUT;
363  nchanges++;
364  }
365  }
366

368  if (nchanges > 0) {
369  D_DEBUG(DF_DSMS, "generating map %p version %u: nchanges=%d\n",
370  map, version, nchanges);
371  rc = pool_map_set_version(map, version);
372  D_ASSERTF(rc == 0, "%d\n", rc);
373  }
374 
375  if (tgts_failed != NULL)
376  tgts_failed->rl_nr = tgts_failed_out;
377  map_exclude_destroy_sanitized_tgts(tgts_sanitized, tgts_sanitized_size);
378  return 0;
379 }
